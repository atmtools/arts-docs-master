51 template<typename T>
52 std::array<String, Index(T::FINAL)> enum_strarray(const String& strchars)
53 {
54  std::array<String, Index(T::FINAL)> out;
55  std::istringstream x(strchars);
56  for (Index i=0; i<Index(T::FINAL); i++) {
57  std::getline(x, out[i], ',');

95 #define ENUMARTS(ENUMTYPE, ...) \
96  enum class ENUMTYPE : Index {__VA_ARGS__, FINAL} ; \
97  \
98  namespace enumstrs { \
99  static const auto ENUMTYPE ## Names = \
100  enum_strarray<ENUMTYPE>(#__VA_ARGS__); \
101  }; \
102  \
103  inline String toString(ENUMTYPE x) noexcept { \
104  if (good_enum(x)) \
105  return enumstrs::ENUMTYPE ## Names [Index(x)]; \
106  else \
107  return "BAD " # ENUMTYPE; \
108  } \
109  \
110  inline ENUMTYPE to ## ENUMTYPE(const String& x) noexcept { \
111  for (Index i=0; i<Index(ENUMTYPE::FINAL); i++) \
112  if (enumstrs::ENUMTYPE ## Names [i] == x) \
113  return ENUMTYPE(i); \
114  return ENUMTYPE::FINAL; \
115  } \
116  \
117  inline std::ostream& operator<<(std::ostream& os, const ENUMTYPE x) { \
118  return os << toString(x); \
119  } \
120  \
121  inline std::istream& operator<<(std::istream& is, ENUMTYPE& x) { \
122  String val; is >> val; x = to ## ENUMTYPE(val); return is; \
123  }

194 inline TypeOfTarget toTypeOfTarget(const String& s, Type x) noexcept {
195  switch (x) {
196  case Type::Special:
197  return TypeOfTarget(toSpecial(s));
198  case Type::Sensor:
199  return TypeOfTarget(toSensor(s));
200  case Type::Line:
201  return TypeOfTarget(toLine(s));
202  case Type::Atm:
203  return TypeOfTarget(toAtm(s));
204  case Type::FINAL: /* leave last, don't use default */ ;
205  }
206  return TypeOfTarget();
207 }
