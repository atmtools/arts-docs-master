113 constexpr TypeOfTarget toTypeOfTarget(const std::string_view& s, Type x) noexcept {
114  switch (x) {
115  case Type::Special:
116  return TypeOfTarget(toSpecial(s));
117  case Type::Sensor:
118  return TypeOfTarget(toSensor(s));
119  case Type::Line:
120  return TypeOfTarget(toLine(s));
121  case Type::Atm:
122  return TypeOfTarget(toAtm(s));
123  case Type::FINAL: /* leave last, don't use default */ ;
124  }
125  return TypeOfTarget();
