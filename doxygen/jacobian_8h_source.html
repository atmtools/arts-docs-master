252  void TargetSubType(const std::string_view& s) noexcept {
253  matm = Atm::FINAL;
254  mline = Line::FINAL;
255  msensor = Sensor::FINAL;
256  mspecial = Special::FINAL;
257 
258  switch (mtype) {
259  case Type::Atm: matm = toAtm(s); break;
260  case Type::Line: mline = toLine(s); break;
261  case Type::Sensor: msensor = toSensor(s); break;
262  case Type::Special: mspecial = toSpecial(s); break;
263  case Type::FINAL: { /* leave last, don't use default */ }
264  }
265  }

267  std::string_view TargetSubType() const noexcept {
268  switch (mtype) {
269  case Type::Atm: return toString(matm);
270  case Type::Line: return toString(mline);
271  case Type::Sensor: return toString(msensor);
272  case Type::Special: return toString(mspecial);
273  case Type::FINAL: { /* leave last, don't use default */ }
274  }
275  return "BAD SUBTYPE";

284  if (1 == (good_enum(mspecial) + good_enum(msensor) + good_enum(mline) + good_enum(matm))) {
285  switch (mtype) {
286  case Type::Special:
287  return good_enum(mspecial);
288  case Type::Sensor:
289  return good_enum(msensor);
290  case Type::Line:
291  return good_enum(mline);
292  case Type::Atm:
293  return good_enum(matm);
