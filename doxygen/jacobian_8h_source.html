248  void TargetSubType(const std::string_view& s) noexcept {
249  matm = Atm::FINAL;
250  mline = Line::FINAL;
251  msensor = Sensor::FINAL;
252  mspecial = Special::FINAL;
253 
254  switch (mtype) {
255  case Type::Atm: matm = toAtm(s); break;
256  case Type::Line: mline = toLine(s); break;
257  case Type::Sensor: msensor = toSensor(s); break;
258  case Type::Special: mspecial = toSpecial(s); break;
259  case Type::FINAL: { /* leave last, don't use default */ }
260  }
261  }

263  std::string_view TargetSubType() const noexcept {
264  switch (mtype) {
265  case Type::Atm: return toString(matm);
266  case Type::Line: return toString(mline);
267  case Type::Sensor: return toString(msensor);
268  case Type::Special: return toString(mspecial);
269  case Type::FINAL: { /* leave last, don't use default */ }
270  }
271  return "BAD SUBTYPE";
272  }

278  bool TargetSubTypeOK() const noexcept {
279  // We can only hold one valid enum at a time, and it must be of the correct type
280  if (1 == (good_enum(mspecial) + good_enum(msensor) + good_enum(mline) + good_enum(matm))) {
281  switch (mtype) {
282  case Type::Special:
283  return good_enum(mspecial);
284  case Type::Sensor:
285  return good_enum(msensor);
286  case Type::Line:
287  return good_enum(mline);
288  case Type::Atm:
289  return good_enum(matm);
290  case Type::FINAL: { /* leave last, don't use default */ }
291  }
292  }

294  return false;
