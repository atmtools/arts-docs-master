314 template <typename T>
315 void xml_read_from_file(const String& filename,
316  T& type,
317  const Verbosity& verbosity) {
318  CREATE_OUT2;
319 
320  String xml_file = filename;
321  find_xml_file(xml_file, verbosity);
322  out2 << " Reading " + xml_file + '\n';
323 
324  // Open input stream:
325  std::unique_ptr<istream> ifs;
326  if (xml_file.nelem() > 2 &&
327  xml_file.substr(xml_file.length() - 3, 3) == ".gz")
328 #ifdef ENABLE_ZLIB
329  {
330  ifs = std::make_unique<igzstream>();
331  xml_open_input_file(
332  *static_cast<igzstream*>(ifs.get()), xml_file, verbosity);
333  }
334 #else
335  {
336  throw runtime_error(
337  "This arts version was compiled without zlib support.\n"
338  "Thus zipped xml files cannot be read.");
339  }
340 #endif /* ENABLE_ZLIB */
341  else {
342  ifs = std::make_unique<ifstream>();
343  xml_open_input_file(
344  *static_cast<ifstream*>(ifs.get()), xml_file, verbosity);
345  }
346 
347  // No need to check for error, because xml_open_input_file throws a
348  // runtime_error with an appropriate error message.
349 
350  // Read the matrix from the stream. Here we catch the exception,
351  // because then we can issue a nicer error message that includes the
352  // filename.
353  try {
354  FileType ftype;
355  NumericType ntype;
356  EndianType etype;
357 
358  xml_read_header_from_stream(*ifs, ftype, ntype, etype, verbosity);
359  if (ftype == FILE_TYPE_ASCII) {
360  xml_read_from_stream(*ifs, type, NULL, verbosity);
361  } else {
362  String bfilename = xml_file + ".bin";
363  bifstream bifs(bfilename.c_str());
364  xml_read_from_stream(*ifs, type, &bifs, verbosity);
365  }
366  xml_read_footer_from_stream(*ifs, verbosity);
367  } catch (const std::runtime_error& e) {
368  ostringstream os;
369  os << "Error reading file: " << xml_file << '\n' << e.what();
370  throw runtime_error(os.str());
371  }
372 }

384 template <typename T>
385 void xml_write_to_file(const String& filename,
386  const T& type,
387  const FileType ftype,
388  const Index no_clobber,
389  const Verbosity& verbosity) {
390  CREATE_OUT2;
391 
392  String efilename = add_basedir(filename);
393 
394  std::unique_ptr<ostream> ofs;
395 
396  if (no_clobber) make_filename_unique(efilename, ".xml");
397 
398  out2 << " Writing " << efilename << '\n';
399  if (ftype == FILE_TYPE_ZIPPED_ASCII)
400 #ifdef ENABLE_ZLIB
401  {
402  ofs = std::make_unique<ogzstream>();
403  xml_open_output_file(*static_cast<ogzstream*>(ofs.get()), efilename);
404  }
405 #else
406  {
407  throw runtime_error(
408  "This arts version was compiled without zlib support.\n"
409  "Thus zipped xml files cannot be written.");
410  }
411 #endif /* ENABLE_ZLIB */
412  else {
413  ofs = std::make_unique<ofstream>();
414  xml_open_output_file(*static_cast<ofstream*>(ofs.get()), efilename);
415  }
416 
417  try {
418  xml_write_header_to_stream(*ofs, ftype, verbosity);
419  if (ftype == FILE_TYPE_ASCII || ftype == FILE_TYPE_ZIPPED_ASCII) {
420  xml_write_to_stream(*ofs, type, NULL, "", verbosity);
421  } else {
422  String bfilename = efilename + ".bin";
423  bofstream bofs(bfilename.c_str());
424  xml_write_to_stream(*ofs, type, &bofs, "", verbosity);
425  }
426 
427  xml_write_footer_to_stream(*ofs, verbosity);
428  } catch (const std::runtime_error& e) {
429  ostringstream os;
430  os << "Error writing file: " << efilename << '\n' << e.what();
431  throw runtime_error(os.str());
432  }
433 }
