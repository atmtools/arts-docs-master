91 void QuantumIdentifier::SetFromString(String str) {
92  // Global species lookup data:
93  using global_data::species_data;
94 
95  // We need a species index sorted by Arts identifier. Keep this in a
96  // static variable, so that we have to do this only once. The ARTS
97  // species index is ArtsMap[<Arts String>].
98  static map<String, SpecIsoMap> ArtsMap;
99 
100  // Remember if this stuff has already been initialized:
101  static bool hinit = false;
102 
103  if (!hinit) {
104  for (Index i = 0; i < species_data.nelem(); ++i) {
105  const SpeciesRecord& sr = species_data[i];
106 
107  for (Index j = 0; j < sr.Isotopologue().nelem(); ++j) {
108  SpecIsoMap indicies(i, j);
109  String buf = sr.Name() + "-" + sr.Isotopologue()[j].Name();
110 
111  ArtsMap[buf] = indicies;
112  }
113  }
114  hinit = true;
115  }
116 
117  std::istringstream is(str);
118  String token;
119 
120  is >> token;
121 
122  // ok, now for the cool index map:
123  // is this arts identifier valid?
124  const map<String, SpecIsoMap>::const_iterator i = ArtsMap.find(token);
125  if (i == ArtsMap.end()) {
126  ostringstream os;
127  os << "ARTS Tag: " << token << " is unknown.";
128  throw runtime_error(os.str());
129  }
130 
131  SpecIsoMap id = i->second;
132  Species(id.Speciesindex());
133  Isotopologue(id.Isotopologueindex());
134 
135  is >> token;
136  if (token == "TR") {
137  mqtype = QuantumIdentifier::TRANSITION;
138  is >> token;
139  if (token != "UP") {
140  std::ostringstream os;
141  os << "Expected 'UP', but got: " << token;
142  throw std::runtime_error(os.str());
143  }
144 
145  is >> token;
146  Rational r;
147  while (is) {
148  ThrowIfQuantumNumberNameInvalid(token);
149  is >> r;
150  QuantumMatch()[QuantumIdentifier::TRANSITION_UPPER_INDEX].Set(token, r);
151  is >> token;
152  if (token == "LO") break;
153  }
154 
155  if (!is) {
156  std::ostringstream os;
157  os << "Premature end of data, expected 'LO'.";
158  throw std::runtime_error(os.str());
159  }
160  is >> token;
161  while (is) {
162  ThrowIfQuantumNumberNameInvalid(token);
163  is >> r;
164  QuantumMatch()[QuantumIdentifier::TRANSITION_LOWER_INDEX].Set(token, r);
165  is >> token;
166  }
167  } else if (token == "EN") {
168  mqtype = QuantumIdentifier::ENERGY_LEVEL;
169 
170  is >> token;
171  Rational r;
172  while (is) {
173  ThrowIfQuantumNumberNameInvalid(token);
174  is >> r;
175  QuantumMatch()[QuantumIdentifier::TRANSITION_UPPER_INDEX].Set(token, r);
176  is >> token;
177  }
178  } else if (token == "ALL") {
179  mqtype = QuantumIdentifier::ALL;
180  } else if (token == "NONE") {
181  mqtype = QuantumIdentifier::NONE;
182  } else {
183  std::ostringstream os;
184  os << "Error parsing QuantumIdentifier. Expected TR or EN, but got: "
185  << token << "\n"
186  << "QI: " << str;
187  throw std::runtime_error(os.str());
188  }
189 }

235 Rational interpret_stringdata(const QuantumNumberType key, const String& val) {
236  if (key == QuantumNumberType::parity) {
237  if (val == "+")
238  return 1_rat;
239  else if (val == "-")
240  return -1_rat;
241  } else if (key == QuantumNumberType::ElectronState) {
242  if (val == "X")
243  return Rational(int('X'));
244  } else if (key == QuantumNumberType::kronigParity) {
245  if (val == "f")
246  return Rational(int('f'));
247  else if (val == "e")
248  return Rational(int('e'));
249  } else {
250  return Rational(val);
251  }
252 
253  return RATIONAL_UNDEFINED;
254 }

256 void update_id(QuantumIdentifier& qid, const std::vector<std::array<String, 2> >& upper_list, const std::vector<std::array<String, 2> >& lower_list)

264  } else {
265  auto val = interpret_stringdata(key, keyval[1]);

267  qid.UpperQuantumNumber(key) = val;
268  } else {

272  }
273  }
274  }
