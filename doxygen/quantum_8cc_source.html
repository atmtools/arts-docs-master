65 void Quantum::Identifier::SetFromString(String str) {
66  std::istringstream is(str);
67  String token;
68 
69  is >> token;
70 
71  spec_ind = Species::Tag(token).spec_ind;
72 
73  is >> token;
74  if (token == "TR") {
75  type = Quantum::IdentifierType::Transition;
76  is >> token;
77  ARTS_USER_ERROR_IF (token != "UP",
78  "Expected 'UP', but got: ", token)
79 
80  is >> token;
81  Rational r;
82  while (is) {
83  if (token == "LO") break;
84  ThrowIfQuantumNumberNameInvalid(token);
85  is >> r;
86  Upper().Set(token, r);
87  is >> token;
88  }
89 
90  ARTS_USER_ERROR_IF (!is,
91  "Premature end of data, expected 'LO'.")
92  is >> token;
93  while (is) {
94  ThrowIfQuantumNumberNameInvalid(token);
95  is >> r;
96  Lower().Set(token, r);
97  is >> token;
98  }
99  } else if (token == "EN") {
100  type = Quantum::IdentifierType::EnergyLevel;
101 
102  is >> token;
103  Rational r;
104  while (is) {
105  ThrowIfQuantumNumberNameInvalid(token);
106  is >> r;
107  Level().Set(token, r);
108  is >> token;
109  }
110  } else if (token == "ALL") {
111  type = Quantum::IdentifierType::All;
112  } else if (token == "NONE") {
113  type = Quantum::IdentifierType::None;
114  } else {
115  ARTS_USER_ERROR (
116  "Error parsing QuantumIdentifier. Expected TR or EN, but got: ",
117  token, "\n"
118  "QI: ", str)
119  }
120 }

130 Rational interpret_stringdata(const QuantumNumberType key, const String& val) {
131  if (key == QuantumNumberType::parity) {
132  if (val == "+")
133  return 1;
134  else if (val == "-")
135  return -1;
136  } else if (key == QuantumNumberType::ElectronState) {
137  if (val == "X")
138  return Rational(int('X'));
139  } else if (key == QuantumNumberType::kronigParity) {
140  if (val == "f")
141  return Rational(int('f'));
142  else if (val == "e")
143  return Rational(int('e'));
144  } else {
145  return Rational(val);
146  }
147 
148  return RATIONAL_UNDEFINED;
149 }
