461 void find_nonlinear_continua(ArrayOfIndex& cont,
462  const ArrayOfArrayOfSpeciesTag& abs_species,
463  const Verbosity& verbosity) {
464  CREATE_OUT3;
465 
466  cont.resize(0);
467 
468  // This is quite complicated, unfortunately. The approach here
469  // is copied from abs_xsec_per_speciesAddConts. For explanation,
470  // see there.
471 
472  // Loop tag groups:
473  for (Index i = 0; i < abs_species.nelem(); ++i) {
474  // Loop tags in tag group
475  for (Index s = 0; s < abs_species[i].nelem(); ++s) {
476  // Check for continuum tags
477  if (abs_species[i][s].type == Species::TagType::Predefined ||
478  abs_species[i][s].type == Species::TagType::Cia) {
479  const String thisname = abs_species[i][s].Name();
480  // Ok, now we know this is a continuum tag.
481  out3 << " Continuum tag: " << thisname;
482 
483  // Check whether we want nonlinear treatment for
484  // this or not. We have three classes of continua:
485  // 1. Those that we know do not require it
486  // 2. Those that we know require h2o_abs
487  // 3. Those for which we do not know
488 
489  // The list here is not at all perfect, just a first
490  // guess. If you need particular models, you better
491  // check that the right thing is done with your model.
492 
493  // 1. Continua known to not use h2o_abs
494  // We take also H2O itself here, since this is
495  // handled separately
496  if (Species::fromShortName("H2O") == abs_species[i][s].Spec() ||
497  "N2-" == thisname.substr(0, 3) || "CO2-" == thisname.substr(0, 4) ||
498  "O2-CIA" == thisname.substr(0, 6) ||
499  "O2-v0v" == thisname.substr(0, 6) ||
500  "O2-v1v" == thisname.substr(0, 6) ||
501  "H2-CIA" == thisname.substr(0, 6) ||
502  "He-CIA" == thisname.substr(0, 6) ||
503  "CH4-CIA" == thisname.substr(0, 7) ||
504  "liquidcloud-MPM93" == thisname.substr(0, 17) ||
505  "liquidcloud-ELL07" == thisname.substr(0, 17)) {
506  out3 << " --> not added.\n";
507  break;
508  }
509 
510  // 2. Continua known to use h2o_abs
511  if ("O2-" == thisname.substr(0, 3)) {
512  cont.push_back(i);
513  out3 << " --> added to abs_nls.\n";
514  break;
515  }
516 
517  // 3. abs_species tags that are NOT allowed in LUT
518  // calculations
519  if ("icecloud-" == thisname.substr(0, 9) ||
520  "rain-" == thisname.substr(0, 5)) {
521  ostringstream os;
522  os << "Tag " << thisname << " not allowed in absorption "
523  << "lookup tables.";
524  throw runtime_error(os.str());
525  }
526 
527  // If we get here, then the tag was neither in the
528  // posivitive nor in the negative list. We through a
529  // runtime error.
530  out3 << " --> unknown.\n";
531  ostringstream os;
532  os << "Unknown whether tag " << thisname
533  << " is a nonlinear species (i.e. uses h2o_abs) or not.\n"
534  << "Cannot set abs_nls automatically.";
535  throw runtime_error(os.str());
536  }
537  }
538  }

601 void choose_abs_t_pert(Vector& abs_t_pert,
602  ConstVectorView abs_t,
603  ConstVectorView tmin,
604  ConstVectorView tmax,
605  const Numeric& step,
606  const Index& p_interp_order,
607  const Index& t_interp_order,
608  const Verbosity& verbosity) {
609  CREATE_OUT2;
610  CREATE_OUT3;
611 
612  // The code to find out the range for perturbation is a bit
613  // complicated. The problem is that, since we use higher order
614  // interpolation for p, we may require temperatures well outside the
615  // local min/max range at the reference level. We solve this by
616  // really looking at the min/max range for those levels required by
617  // the p_interp_order.
618 
619  Numeric mindev = 1e9;
620  Numeric maxdev = -1e9;
621 
622  Vector the_grid(0, abs_t.nelem(), 1);
623  for (Index i = 0; i < the_grid.nelem(); ++i) {
624  const Index idx0 = Interpolation::pos_finder(i, Numeric(i), the_grid, p_interp_order, false, true);
625 
626  for (Index j = 0; j < p_interp_order+1; ++j) {
627  // Our pressure grid for the lookup table may be coarser than
628  // the original one for the batch cases. This may lead to max/min
629  // values in the original data that exceed those we assumed
630  // above. We add some extra margin here to account for
631  // that. (The margin is +-10K)
632 
633  Numeric delta_min = tmin[i] - abs_t[idx0 + j] - 10;
634  Numeric delta_max = tmax[i] - abs_t[idx0 + j] + 10;
635 
636  if (delta_min < mindev) mindev = delta_min;
637  if (delta_max > maxdev) maxdev = delta_max;
638  }
639  }
640 
641  out3 << " abs_t_pert: mindev/maxdev : " << mindev << " / " << maxdev << "\n";
642 
643  // We divide the interval between mindev and maxdev, so that the
644  // steps are of size *step* or smaller. But we also need at least
645  // *t_interp_order*+1 points.
646  Index div = t_interp_order;
647  Numeric effective_step;
648  do {
649  effective_step = (maxdev - mindev) / (Numeric)div;

674 void choose_abs_nls_pert(Vector& abs_nls_pert,
675  ConstVectorView refprof,
676  ConstVectorView minprof,
677  ConstVectorView maxprof,
678  const Numeric& step,
679  const Index& p_interp_order,
680  const Index& nls_interp_order,
681  const Verbosity& verbosity) {
682  CREATE_OUT2;
683  CREATE_OUT3;
684 
685  // The code to find out the range for perturbation is a bit
686  // complicated. The problem is that, since we use higher order
687  // interpolation for p, we may require humidities well outside the
688  // local min/max range at the reference level. We solve this by
689  // really looking at the min/max range for those levels required by
690  // the p_interp_order.
691 
692  Numeric mindev = 0;
693  Numeric maxdev = -1e9;
694 
695  // mindev is set to zero from the start, since we always want to
696  // include 0.
697 
698  Vector the_grid(0, refprof.nelem(), 1);
699  for (Index i = 0; i < the_grid.nelem(); ++i) {
700  // cout << "!!!!!! i = " << i << "\n";
701  // cout << " min/ref/max = " << minprof[i] << " / "
702  // << refprof[i] << " / "
703  // << maxprof[i] << "\n";
704 
705  const Index idx0 = Interpolation::pos_finder(i, Numeric(i), the_grid, p_interp_order, false, true);
706 
707  for (Index j = 0; j < p_interp_order+1; ++j) {
708  // cout << "!!!!!! j = " << j << "\n";
709  // cout << " ref[j] = " << refprof[gp.idx[j]] << " ";
710  // cout << " minfrac[j] = " << minprof[i] / refprof[gp.idx[j]] << " ";
711  // cout << " maxfrac[j] = " << maxprof[i] / refprof[gp.idx[j]] << " \n";
712 
713  // Our pressure grid for the lookup table may be coarser than
714  // the original one for the batch cases. This may lead to max/min
715  // values in the original data that exceed those we assumed
716  // above. We add some extra margin to the max value here to account for
717  // that. (The margin is a factor of 2.)
718 
719  Numeric delta_min = minprof[i] / refprof[idx0 + j];
720  Numeric delta_max = 2 * maxprof[i] / refprof[idx0 + j];
721 
722  if (delta_min < mindev) mindev = delta_min;
723  // do not update maxdev, when delta_max is infinity (this results from
724  // refprof being 0)
725  if (!std::isinf(delta_max) && (delta_max > maxdev)) maxdev = delta_max;
726  }
727  }
728 
729  out3 << " abs_nls_pert: mindev/maxdev : " << mindev << " / " << maxdev
730  << "\n";
731 
732  bool allownegative = false;
733  if (mindev < 0) {
734  out2
735  << " Warning: I am getting a negative fractional distance to the H2O\n"
736  << " reference profile. Some of your H2O fields may contain negative values.\n"
737  << " Will allow negative values also for abs_nls_pert.\n";
738  allownegative = true;
739  }
740 
741  if (!allownegative) {
742  mindev = 0;
743  out3 << " Adjusted mindev : " << mindev << "\n";
744  }
745 
746  if (std::isinf(maxdev)) {
747  ostringstream os;
748  os << "Perturbation upper limit is infinity (likely due to the reference\n"
749  << "profile being 0 at at least one pressure level). Can not work\n"
750  << "with that.";
751  throw runtime_error(os.str());
752  }
753 
754  // We divide the interval between mindev and maxdev, so that the
755  // steps are of size *step* or smaller. But we also need at least
756  // *nls_interp_order*+1 points.
757  Index div = nls_interp_order;
758  Numeric effective_step;
759  do {
760  effective_step = (maxdev - mindev) / (Numeric)div;
761  ++div;
762  } while (effective_step > step);
763 
764  abs_nls_pert = Vector(mindev, div, effective_step);
765 
766  // If there are negative values, we also add 0. The reason for this
767  // is that 0 is a turning point.
768  if (allownegative) {
769  VectorInsertGridPoints(abs_nls_pert, abs_nls_pert, {0}, verbosity);
770  out2
771  << " I am including also 0 in the abs_nls_pert, because it is a turning \n"
772  << " point. Consider to use a higher abs_nls_interp_order, for example 4.\n";
773  }
774 
775  out2 << " abs_nls_pert: " << abs_nls_pert[0] << " to "
776  << abs_nls_pert[abs_nls_pert.nelem() - 1]
777  << " (fractional units) in steps of "
778  << abs_nls_pert[1] - abs_nls_pert[0] << " (" << abs_nls_pert.nelem()
779  << " grid points)\n";
780 }
