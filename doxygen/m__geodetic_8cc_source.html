54 void refellipsoidForAzimuth(Vector& refellipsoid,
55  const Numeric& latitude,
56  const Numeric& azimuth,
57  const Verbosity&) {
58  if (refellipsoid.nelem() != 2)
59  throw runtime_error("Input *refellispoid must be a vector of length 2*.");
60 
61  if (refellipsoid[1] > 0) {
62  const Numeric e2 = refellipsoid[1] * refellipsoid[1];
63  const Numeric a = 1 - e2 * pow(sin(DEG2RAD * latitude), 2.0);
64 
65  const Numeric rn = 1 / sqrt(a);
66  const Numeric rm = (1 - e2) * (rn / a);
67 
68  const Numeric v = DEG2RAD * azimuth;
69 
70  refellipsoid[0] =
71  refellipsoid[0] / (pow(cos(v), 2.0) / rm + pow(sin(v), 2.0) / rn);
72  refellipsoid[1] = 0;
73  }
74 }

79 void refellipsoidOrbitPlane(Vector& refellipsoid,
80  const Numeric& orbitinc,
81  const Verbosity&) {
82  if (refellipsoid.nelem() != 2)
83  throw runtime_error("Input *refellispoid must be a vector of length 2*.");
84  chk_if_in_range("orbitinc", orbitinc, 0, 180);
85 
86  // Radius at maximum latitude
87  const Numeric rp = refell2r(refellipsoid, orbitinc);
88 
89  // New eccentricity
90  refellipsoid[1] = sqrt(1 - pow(rp / refellipsoid[0], 2.0));
91 }

300  y_geo(i,1), y_geo(i,2), y_geo(i,3), y_geo(i,4));
301  Numeric h, lat, lon, za, aa;
