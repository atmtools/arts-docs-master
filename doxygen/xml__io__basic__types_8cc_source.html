130 
131  if (jt == Jacobian::Line::VMR) {
132  jt.QuantumIdentity().SetAll();
133  jt.QuantumIdentity().Species() = spec.Species();
134  jt.QuantumIdentity().Isotopologue() = spec.Isotopologue();
135  } else if (jt == Jacobian::Line::NLTE) {
136  QuantumNumbers levelquanta;
137  tag.get_attribute_value("levelquanta", levelquanta);

139  } else {

142  tag.get_attribute_value("lowerglobalquanta", lowerglobalquanta);
143  jt.QuantumIdentity().SetTransition(upperglobalquanta, lowerglobalquanta);
144  }
145  }
146

148  String key;
149  tag.get_attribute_value("species", key);

151  }
152

155  }
156

158  *pbifs >> jt.Perturbation();
159  if (pbifs->fail()) {
160  xml_data_parse_error(tag, "");
161  }
162  } else {
163  is_xml >> jt.Perturbation();
164  if (is_xml.fail()) {
165  xml_data_parse_error(tag, "");
166  }
167  }
168 
169  tag.read_from_stream(is_xml);
170  tag.check_name("/JacobianTarget");
171 
172  ARTS_USER_ERROR_IF (not jt.TargetSubTypeOK(),
173  "Bad input: ", typestr, " or ", subtypestr, '\n', "\tCannot be interpreted as a type or substype...\n")
174 }

663 /* param pbifs Pointer to binary input stream. NULL in case of ASCII file.
664  Ignored because strings are always stored in ASCII format. */
665 void xml_read_from_stream(istream& is_xml,
666  String& str,
667  bifstream* /* pbifs */,
668  const Verbosity& verbosity) {
669  ArtsXMLTag tag(verbosity);
670  char dummy;
671 
672  tag.read_from_stream(is_xml);
673  tag.check_name("String");
674 
675  // Skip whitespaces
676  bool string_starts_with_quotes = true;
677  do {
678  is_xml >> dummy;
679  switch (dummy) {
680  case ' ':
681  case '\"':
682  case '\n':
683  case '\r':
684  case '\t':
685  break;
686  default:
687  string_starts_with_quotes = false;
688  }
689  } while (is_xml.good() && dummy != '"' && string_starts_with_quotes);
690 
691  // Throw exception if first char after whitespaces is not a quote
692  if (!string_starts_with_quotes) {
693  xml_parse_error("String must begin with \"");
694  }
695 
696  //catch case where string is empty. CPD 29/8/05
697  dummy = (char)is_xml.peek();
698  if (dummy == '"') {
699  str = "";
700  } else {
701  stringbuf strbuf;
702 
703  is_xml.get(strbuf, '"');
704  if (is_xml.fail()) {
705  xml_parse_error("String must end with \"");
706  }
707  str = strbuf.str();
708  }
709 
710  // Ignore quote
711  is_xml >> dummy;
712 
713  tag.read_from_stream(is_xml);
714  tag.check_name("/String");
715 }
