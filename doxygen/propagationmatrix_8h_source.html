145  const Numeric& b,
146  const Numeric& c,
147  const Numeric& u) noexcept {
148  return (Eigen::Matrix3d() << a * a + u * u,
149  -a * b - c * u,
150  -a * c + b * u,
151  -a * b + c * u,
