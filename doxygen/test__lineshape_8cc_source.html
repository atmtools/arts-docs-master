136

138  Vector vmr({0.3, 0.7});
139  if constexpr (mod == ModificationInternal::SelfVMR) vmr[0] += ::dVMR;
140  else if constexpr (mod == ModificationInternal::AirVMR) vmr[1] += ::dVMR;
141 
142  const Numeric P = 1e3;

144  const Numeric H = (mod == ModificationInternal::H) ? 50e-6 + ::dH : 50e-6;

146  return {qid, band, f_grid, vmr, P, T, H};
