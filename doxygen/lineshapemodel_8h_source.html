258 constexpr bool modelparameterEmpty(const ModelParameters mp) noexcept {
259  switch(mp.type) {
260  case TemperatureModel::None: // 0
261  return true;
262  case TemperatureModel::T0: // Constant, X0
263  return (mp.X0 == 0);
264  case TemperatureModel::T1: // Standard, X0 * (T0/T) ^ X1
265  return (mp.X0 == 0);
266  case TemperatureModel::T2: // X0 * (T0/T) ^ X1 * (1 + X2 * log(T/T0));
267  return (mp.X0 == 0);
268  case TemperatureModel::T3: // X0 + X1 * (T - T0)
269  return (mp.X0 == 0 and mp.X1 == 0);
270  case TemperatureModel::T4: // (X0 + X1 * (T0/T - 1)) * (T0/T)^X2;
271  return (mp.X0 == 0 and mp.X1 == 0);
272  case TemperatureModel::T5: // X0 * (T0/T)^(0.25 + 1.5*X1)
273  return (mp.X0 == 0);
274  case TemperatureModel::LM_AER: // X(200) = X0; X(250) = X1; X(298) = X2; X(340) = X3; Linear interpolation in between
275  return (mp.X0 == 0 and mp.X1 == 0 and mp.X2 == 0 and mp.X3 == 0);
276  case TemperatureModel::DPL: // X0 * (T0/T) ^ X1 + X2 * (T0/T) ^ X3
277  return (mp.X0 == 0 and mp.X2 == 0);
278  case TemperatureModel::POLY:
279  return (mp.X0 == 0 and mp.X1 == 0 and mp.X2 == 0 and mp.X3 == 0);
280  case TemperatureModel::FINAL:
281  return true;
282  }
283  return true;
284 }

286 constexpr Numeric modelparameterFirstExponent(const ModelParameters mp) noexcept {
287  switch(mp.type) {
288  case TemperatureModel::None: // 0
289  return 0;
290  case TemperatureModel::T0: // Constant, X0
291  return 0;
292  case TemperatureModel::T1: // Standard, X0 * (T0/T) ^ X1
293  return mp.X1;
294  case TemperatureModel::T2: // X0 * (T0/T) ^ X1 * (1 + X2 * log(T/T0));
295  return mp.X1;
296  case TemperatureModel::T3: // X0 + X1 * (T - T0)
297  return 0;
298  case TemperatureModel::T4: // (X0 + X1 * (T0/T - 1)) * (T0/T)^X2;
299  return mp.X2;
300  case TemperatureModel::T5: // X0 * (T0/T)^(0.25 + 1.5*X1)
301  return (0.25 + 1.5*mp.X1);
302  case TemperatureModel::LM_AER: // X(200) = X0; X(250) = X1; X(298) = X2; X(340) = X3; Linear interpolation in between
303  return 0;
304  case TemperatureModel::DPL: // X0 * (T0/T) ^ X1 + X2 * (T0/T) ^ X3
305  return mp.X1;
306  case TemperatureModel::POLY:
307  return 0;
308  case TemperatureModel::FINAL:
309  return std::numeric_limits<Numeric>::quiet_NaN();
310  }
311  return std::numeric_limits<Numeric>::quiet_NaN();
312 }
