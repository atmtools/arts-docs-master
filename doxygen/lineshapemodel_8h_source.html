138 constexpr bool modelparameterEmpty(const ModelParameters mp) noexcept {
139  switch(mp.type) {
140  case TemperatureModel::None: // 0
141  return true;
142  case TemperatureModel::T0: // Constant, X0
143  return (mp.X0 == 0);
144  case TemperatureModel::T1: // Standard, X0 * (T0/T) ^ X1
145  return (mp.X0 == 0);
146  case TemperatureModel::T2: // X0 * (T0/T) ^ X1 * (1 + X2 * log(T/T0));
147  return (mp.X0 == 0);
148  case TemperatureModel::T3: // X0 + X1 * (T - T0)
149  return (mp.X0 == 0 and mp.X1 == 0);
150  case TemperatureModel::T4: // (X0 + X1 * (T0/T - 1)) * (T0/T)^X2;
151  return (mp.X0 == 0 and mp.X1 == 0);
152  case TemperatureModel::T5: // X0 * (T0/T)^(0.25 + 1.5*X1)
153  return (mp.X0 == 0);
154  case TemperatureModel::LM_AER: // X(200) = X0; X(250) = X1; X(298) = X2; X(340) = X3; Linear interpolation in between
155  return (mp.X0 == 0 and mp.X1 == 0 and mp.X2 == 0 and mp.X3 == 0);
156  case TemperatureModel::DPL: // X0 * (T0/T) ^ X1 + X2 * (T0/T) ^ X3
157  return (mp.X0 == 0 and mp.X2 == 0);
158  case TemperatureModel::FINAL:
159  return true;
160  }
161  return true;
162 }

164 constexpr Numeric modelparameterFirstExponent(const ModelParameters mp) noexcept {
165  switch(mp.type) {
166  case TemperatureModel::None: // 0
167  return 0;
168  case TemperatureModel::T0: // Constant, X0
169  return 0;
170  case TemperatureModel::T1: // Standard, X0 * (T0/T) ^ X1
171  return mp.X1;
172  case TemperatureModel::T2: // X0 * (T0/T) ^ X1 * (1 + X2 * log(T/T0));
173  return mp.X1;
174  case TemperatureModel::T3: // X0 + X1 * (T - T0)
175  return 0;
176  case TemperatureModel::T4: // (X0 + X1 * (T0/T - 1)) * (T0/T)^X2;
177  return mp.X2;
178  case TemperatureModel::T5: // X0 * (T0/T)^(0.25 + 1.5*X1)
179  return (0.25 + 1.5*mp.X1);
180  case TemperatureModel::LM_AER: // X(200) = X0; X(250) = X1; X(298) = X2; X(340) = X3; Linear interpolation in between
181  return 0;
182  case TemperatureModel::DPL: // X0 * (T0/T) ^ X1 + X2 * (T0/T) ^ X3
183  return mp.X1;
184  case TemperatureModel::FINAL:
185  return std::numeric_limits<Numeric>::quiet_NaN();
186  }
187  return std::numeric_limits<Numeric>::quiet_NaN();
188 }
