132 constexpr bool modelparameterEmpty(const ModelParameters mp) noexcept {
133  switch(mp.type) {
134  case TemperatureModel::None: // 0
135  return true;
136  case TemperatureModel::T0: // Constant, X0
137  return (mp.X0 == 0);
138  case TemperatureModel::T1: // Standard, X0 * (T0/T) ^ X1
139  return (mp.X0 == 0);
140  case TemperatureModel::T2: // X0 * (T0/T) ^ X1 * (1 + X2 * log(T/T0));
141  return (mp.X0 == 0);
142  case TemperatureModel::T3: // X0 + X1 * (T - T0)
143  return (mp.X0 == 0 and mp.X1 == 0);
144  case TemperatureModel::T4: // (X0 + X1 * (T0/T - 1)) * (T0/T)^X2;
145  return (mp.X0 == 0 and mp.X1 == 0);
146  case TemperatureModel::T5: // X0 * (T0/T)^(0.25 + 1.5*X1)
147  return (mp.X0 == 0);
148  case TemperatureModel::LM_AER: // X(200) = X0; X(250) = X1; X(298) = X2; X(340) = X3; Linear interpolation in between
149  return (mp.X0 == 0 and mp.X1 == 0 and mp.X2 == 0 and mp.X3 == 0);
150  case TemperatureModel::DPL: // X0 * (T0/T) ^ X1 + X2 * (T0/T) ^ X3
151  return (mp.X0 == 0 and mp.X2 == 0);
152  case TemperatureModel::FINAL:
153  return true;
154  }
155  return true;
156 }

158 constexpr Numeric modelparameterFirstExponent(const ModelParameters mp) noexcept {
159  switch(mp.type) {
160  case TemperatureModel::None: // 0
161  return 0;
162  case TemperatureModel::T0: // Constant, X0
163  return 0;
164  case TemperatureModel::T1: // Standard, X0 * (T0/T) ^ X1
165  return mp.X1;
166  case TemperatureModel::T2: // X0 * (T0/T) ^ X1 * (1 + X2 * log(T/T0));
167  return mp.X1;
168  case TemperatureModel::T3: // X0 + X1 * (T - T0)
169  return 0;
170  case TemperatureModel::T4: // (X0 + X1 * (T0/T - 1)) * (T0/T)^X2;
171  return mp.X2;
172  case TemperatureModel::T5: // X0 * (T0/T)^(0.25 + 1.5*X1)
173  return (0.25 + 1.5*mp.X1);
174  case TemperatureModel::LM_AER: // X(200) = X0; X(250) = X1; X(298) = X2; X(340) = X3; Linear interpolation in between
175  return 0;
176  case TemperatureModel::DPL: // X0 * (T0/T) ^ X1 + X2 * (T0/T) ^ X3
177  return mp.X1;
178  case TemperatureModel::FINAL:
179  return std::numeric_limits<Numeric>::quiet_NaN();
180  }
181  return std::numeric_limits<Numeric>::quiet_NaN();
182 }
