509  InternalData(Index nf, Index nj) {
510  F.setZero(nf);
511  N.setZero(nf);
512  dF.setZero(nf, nj);
513  dN.setZero(nf, nj);
514  data.setZero(nf, Linefunctions::ExpectedDataSize());
515 
516  Fc.setZero(1);
517  Nc.setZero(1);
518  dFc.setZero(1, nj);
519  dNc.setZero(1, nj);
520  datac.setZero(1, Linefunctions::ExpectedDataSize());
521  }
522 
523  void SetZero() {
524  F.setZero(F.rows());
525  N.setZero(N.rows());
526  dF.setZero(dF.rows(), dF.cols());
527  dN.setZero(dN.rows(), dN.cols());
