62 Numeric barometric_heightformula( //output is p1
63  //input
64  const Numeric& p,
65  const Numeric& dh)
66 
67 {
68  /* taken from: Seite „Barometrische Höhenformel“. In: Wikipedia,
69  * Die freie Enzyklopädie. Bearbeitungsstand: 3. April 2011, 20:28 UTC.
70  * URL: http://de.wikipedia.org/w/index.php?title=Barometrische_H%C3%B6henformel&oldid=87257486

74  //barometric height formula

98 Numeric dinvplanckdI(const Numeric& i, const Numeric& f) {
99  ARTS_USER_ERROR_IF (i <= 0, "Non-positive radiance")
100  ARTS_USER_ERROR_IF (f <= 0, "Non-positive frequency")
101 
102  static const Numeric a = PLANCK_CONST / BOLTZMAN_CONST;
103  static const Numeric b = 2 * PLANCK_CONST / (SPEED_OF_LIGHT * SPEED_OF_LIGHT);
104  const Numeric d = b * f * f * f / i;
105  const Numeric binv = a * f / log1p(d);
106 
107  return binv * binv / (a * f * i * (1 / d + 1));
108 }

130 void fresnel(Complex& Rv,
131  Complex& Rh,
132  const Complex& n1,
133  const Complex& n2,
134  const Numeric& theta) {
135  const Numeric theta1 = DEG2RAD * theta;
136  const Numeric costheta1 = cos(theta1);
137  const Numeric costheta2 = cos(asin(n1.real() * sin(theta1) / n2.real()));
138 
139  Complex a, b;
140  a = n2 * costheta1;
141  b = n1 * costheta2;
142  Rv = (a - b) / (a + b);
143  a = n1 * costheta1;
144  b = n2 * costheta2;
145  Rh = (a - b) / (a + b);
146 }

271 Numeric dplanck_dt(const Numeric& f, const Numeric& t) {
272  ARTS_USER_ERROR_IF (t <= 0, "Non-positive temperature")
273  ARTS_USER_ERROR_IF (f <= 0, "Non-positive frequency")
274 
275  constexpr Numeric a = 2 * Constant::h / Constant::pow2(Constant::c);;
276  constexpr Numeric b = Constant::h / Constant::k;
277 
278  // nb. expm1(x) should be more accurate than exp(x) - 1, so use it
279  const Numeric inv_exp_t_m1 = 1.0 / std::expm1(b * f / t);
280 
281  return a * b * Constant::pow4(f) * inv_exp_t_m1 * (1 + inv_exp_t_m1) / Constant::pow2(t);
282 }

335 Numeric dplanck_df(const Numeric& f, const Numeric& t) {
336  ARTS_USER_ERROR_IF (t <= 0, "Non-positive temperature")
337  ARTS_USER_ERROR_IF (f <= 0, "Non-positive frequency")

339  constexpr Numeric a = 2 * Constant::h / Constant::pow2(Constant::c);;
