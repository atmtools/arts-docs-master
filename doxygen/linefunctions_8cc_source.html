89  const Eigen::Ref<const Eigen::VectorXd> f_grid,

91  const Numeric& temperature,

98  const Absorption::NormalizationType norm_type)
99 {
100  Eigen::MatrixXcd dF(0, 0), data(F.size(), Linefunctions::ExpectedDataSize());
101 
102  switch (lineshape_type) {
103  case LineShape::Type::HTP:
104  case LineShape::Type::SDVP:
105  set_htp(F,
106  dF,
107  f_grid,
108  zeeman_df,
109  magnetic_magnitude,
110  line.F0(),
111  doppler_constant,
112  X);
113  break;
114  case LineShape::Type::VP:
115  set_voigt(F,
116  dF,
117  data,
118  f_grid,

120  magnetic_magnitude,
121  line.F0(),
122  doppler_constant,
123  X);
124  break;
125  case LineShape::Type::DP:
126  set_doppler(F,
127  dF,
128  data,
129  f_grid,
130  zeeman_df,
131  magnetic_magnitude,
132  line.F0(),
133  doppler_constant);
134  break;
135  case LineShape::Type::LP:
136  set_lorentz(

138  break;
139  case LineShape::Type::FINAL:
140  break;
141  }
142 
143  switch (mirroring_type) {
144  case Absorption::MirroringType::None:

156  magnetic_magnitude,

159 
160  // Apply mirroring; FIXME: Add conjugate?
161  F.noalias() += Fm;
162  } break;
163  case Absorption::MirroringType::SameAsLineShape: {
164  // Set the mirroring computational vectors and size them as needed
165  Eigen::VectorXcd Fm(F.size());
166 
167  switch (lineshape_type) {
168  case LineShape::Type::DP:
169  set_doppler(Fm,
170  dF,
171  data,
172  f_grid,
173  -zeeman_df,
174  magnetic_magnitude,
175  -line.F0(),
176  -doppler_constant);
177  break;
178  case LineShape::Type::LP:
179  set_lorentz(Fm,
180  dF,
181  data,
182  f_grid,
183  -zeeman_df,
184  magnetic_magnitude,
185  -line.F0(),
186  LineShape::mirroredOutput(X));
187  break;
188  case LineShape::Type::VP:
189  set_voigt(Fm,
190  dF,
191  data,
192  f_grid,
193  -zeeman_df,
194  magnetic_magnitude,
195  -line.F0(),
196  -doppler_constant,
197  LineShape::mirroredOutput(X));
198  break;
199  case LineShape::Type::HTP:
200  case LineShape::Type::SDVP:
201  // WARNING: This mirroring is not tested and it might require, e.g., FVC to be treated differently
202  set_htp(Fm,
203  dF,
204  f_grid,
205  -zeeman_df,
206  magnetic_magnitude,
207  -line.F0(),
208  -doppler_constant,
209  LineShape::mirroredOutput(X));
210  break;
211  case LineShape::Type::FINAL:
212  break;
213  }
214 
215  F.noalias() += Fm;
216  break;
217  }
218  case Absorption::MirroringType::FINAL: break;
219  }
220 
221  // Line normalization if necessary
222  // The user sets this by setting LSM LNT followed by a type

232  break;
233  case Absorption::NormalizationType::RQ:
234  apply_rosenkranz_quadratic_scaling(F, dF, f_grid, line.F0(), temperature);
235  break;
236  case Absorption::NormalizationType::FINAL: break;
237  }

240 void Linefunctions::set_lorentz(
241  Eigen::Ref<Eigen::VectorXcd> F,
242  Eigen::Ref<Eigen::MatrixXcd> dF,
243  Eigen::Ref<Eigen::Matrix<Complex, Eigen::Dynamic, ExpectedDataSize()>> data,
244  const Eigen::Ref<const Eigen::VectorXd> f_grid,
245  const Numeric& zeeman_df,
246  const Numeric& magnetic_magnitude,
247  const Numeric& F0_noshift,
248  const LineShape::Output& lso,
249  const AbsorptionLines& band,
250  const Index& line_ind,
251  const ArrayOfRetrievalQuantity& derivatives_data,
252  const LineShape::Output& dT,
253  const LineShape::Output& dVMR) {
254  constexpr Complex cpi(0, Constant::pi);
255  constexpr Complex iz(0.0, 1.0);
256 
257  // Size of the problem
258  auto nppd = derivatives_data.nelem();
259 
260  // The central frequency
261  const Numeric F0 = F0_noshift + lso.D0 + zeeman_df * magnetic_magnitude + lso.DV;
262 
263  // Naming req data blocks
264  auto z = data.col(0);
265  auto dw = data.col(1);
266 
267  z.noalias() =
268  (Constant::pi * Complex(lso.G0, F0) - cpi * f_grid.array()).matrix();
269  F.noalias() = z.cwiseInverse();
270 
271  if (nppd) {
272  dw.noalias() = -Constant::pi * F.array().square().matrix();
273 
274  for (auto iq = 0; iq < nppd; iq++) {
275  const auto& deriv = derivatives_data[iq];
276 
277  if (not propmattype(deriv)) continue;
278 
279  if (deriv == Jacobian::Atm::Temperature) {
280  dF.col(iq).noalias() = Complex(dT.G0, dT.D0 + dT.DV) * dw;
281  } else if (is_frequency_parameter(deriv)) {
282  dF.col(iq).noalias() = -iz * dw;
283  } else if (is_magnetic_parameter(deriv)) {
284  dF.col(iq).noalias() = iz * zeeman_df * dw;
285  } else if (deriv.Target().needQuantumIdentity()) {
286  const Absorption::QuantumIdentifierLineTarget lt(deriv.Target().QuantumIdentity(), band, line_ind);
287  if (lt not_eq Absorption::QuantumIdentifierLineTargetType::None) {
288  if (deriv == Jacobian::Line::VMR and lt == Absorption::QuantumIdentifierLineTargetType::Isotopologue) {
289  dF.col(iq).noalias() = Complex(dVMR.G0, dVMR.D0 + dVMR.DV) * dw;
290  } else if (deriv == Jacobian::Line::Center and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
291  dF.col(iq).noalias() = iz * dw;
292  } else if (is_pressure_broadening_G0(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
293  dF.col(iq).noalias() = dw;
294  } else if (is_pressure_broadening_D0(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
295  dF.col(iq).noalias() = iz * dw;
296  } else {
297  dF.col(iq).setZero();
298  }
299  }
300  }
301  }
302  }

305 void Linefunctions::set_voigt(
306  Eigen::Ref<Eigen::VectorXcd> F,
307  Eigen::Ref<Eigen::MatrixXcd> dF,
308  Eigen::Ref<Eigen::Matrix<Complex, Eigen::Dynamic, ExpectedDataSize()>> data,
309  const Eigen::Ref<const Eigen::VectorXd> f_grid,
310  const Numeric& zeeman_df,
311  const Numeric& magnetic_magnitude,
312  const Numeric& F0_noshift,
313  const Numeric& GD_div_F0,
314  const LineShape::Output& lso,
315  const AbsorptionLines& band,
316  const Index& line_ind,
317  const ArrayOfRetrievalQuantity& derivatives_data,
318  const Numeric& dGD_div_F0_dT,
319  const LineShape::Output& dT,
320  const LineShape::Output& dVMR) {
321  constexpr Complex iz(0.0, 1.0);
322 
323  // Size of problem
324  auto nppd = derivatives_data.nelem();
325 
326  // Doppler broadening and line center
327  const Numeric F0 = F0_noshift + zeeman_df * magnetic_magnitude + lso.D0 + lso.DV;
328  const Numeric GD = GD_div_F0 * F0;
329  const Numeric invGD = 1.0 / std::abs(GD);
330  const Numeric dGD_dT = dGD_div_F0_dT * F0 - GD_div_F0 * (dT.D0 + dT.DV);
331 
332  // constant normalization factor for Voigt
333  const Numeric fac = Constant::inv_sqrt_pi * invGD;
334 
335  // Naming req data blocks
336  auto z = data.col(0);
337  auto dw = data.col(1);
338 
339  // Frequency grid
340  z.noalias() = invGD * (Complex(-F0, lso.G0) + f_grid.array()).matrix();
341 
342  // Line shape
343  F.noalias() = fac * z.unaryExpr(&w);
344 
345  if (nppd) {
346  dw.noalias() = 2 * (Complex(0, fac * Constant::inv_sqrt_pi) -
347  z.cwiseProduct(F).array())
348  .matrix();
349 
350  for (auto iq = 0; iq < nppd; iq++) {
351  const auto& deriv = derivatives_data[iq];
352 
353  if (not propmattype(deriv)) continue;
354 
355  if (is_frequency_parameter(deriv)) {
356  dF.col(iq).noalias() = invGD * dw;
357  } else if (deriv == Jacobian::Atm::Temperature) {
358  dF.col(iq).noalias() =
359  dw * Complex(-dT.D0 - dT.DV, dT.G0) * invGD -
360  F * dGD_dT * invGD - dw.cwiseProduct(z) * dGD_dT * invGD;
361  } else if (is_magnetic_parameter(deriv)) {
362  dF.col(iq).noalias() = dw * (-zeeman_df * invGD);
363  } else if (deriv.Target().needQuantumIdentity()) {
364  const Absorption::QuantumIdentifierLineTarget lt(deriv.Target().QuantumIdentity(), band, line_ind);
365  if (lt not_eq Absorption::QuantumIdentifierLineTargetType::None) {
366  if (deriv == Jacobian::Line::VMR and lt == Absorption::QuantumIdentifierLineTargetType::Isotopologue) {
367  dF.col(iq).noalias() = dw * Complex(-dVMR.D0 - dVMR.DV, dVMR.G0) * invGD;
368  } else if (deriv == Jacobian::Line::Center and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
369  dF.col(iq).noalias() = -F / F0 - dw * invGD - dw.cwiseProduct(z) / F0;
370  } else if (is_pressure_broadening_G0(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
371  dF.col(iq).noalias() = iz * dw * invGD;
372  } else if (is_pressure_broadening_D0(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
373  dF.col(iq).noalias() = -dw * invGD;
374  } else {
375  dF.col(iq).setZero();
376  }
377  }
378  }
379  }
380  }

383 void Linefunctions::set_doppler(
384  Eigen::Ref<Eigen::VectorXcd> F,
385  Eigen::Ref<Eigen::MatrixXcd> dF,
386  Eigen::Ref<Eigen::Matrix<Complex, Eigen::Dynamic, ExpectedDataSize()>> data,
387  const Eigen::Ref<const Eigen::VectorXd> f_grid,
388  const Numeric& zeeman_df,
389  const Numeric& magnetic_magnitude,
390  const Numeric& F0_noshift,
391  const Numeric& GD_div_F0,
392  const AbsorptionLines& band,
393  const Index& line_ind,
394  const ArrayOfRetrievalQuantity& derivatives_data,
395  const Numeric& dGD_div_F0_dT) {
396  auto nppd = derivatives_data.nelem();
397 
398  // Doppler broadening and line center
399  const Numeric F0 = F0_noshift + zeeman_df * magnetic_magnitude;
400  const Numeric invGD = 1.0 / std::abs(GD_div_F0 * F0);
401 
402  // Naming data blocks
403  auto x = data.col(0);
404  auto mx2 = data.col(1);
405 
406  x.noalias() = (f_grid.array() - F0).matrix() * invGD;
407  mx2.noalias() = -data.col(0).cwiseAbs2();
408  F.noalias() = invGD * Constant::inv_sqrt_pi * mx2.array().exp().matrix();
409 
410  for (auto iq = 0; iq < nppd; iq++) {
411  const auto& deriv = derivatives_data[iq];
412 
413  if (not propmattype(deriv)) continue;
414 
415  if (is_frequency_parameter(deriv)) {
416  dF.col(iq).noalias() = -2 * invGD * F.cwiseProduct(x);
417  } else if (deriv == Jacobian::Atm::Temperature) {
418  dF.col(iq).noalias() =
419  -dGD_div_F0_dT * F0 * invGD * (2.0 * F.cwiseProduct(mx2) + F);
420  } else if (deriv.Target().needQuantumIdentity()) {
421  const Absorption::QuantumIdentifierLineTarget lt(deriv.Target().QuantumIdentity(), band, line_ind);
422  if (deriv == Jacobian::Line::Center and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
423  dF.col(iq).noalias() = -F.cwiseProduct(mx2) * (2 / F0) + F.cwiseProduct(x) * 2 * (invGD - 1 / F0);
424  } else {
425  dF.col(iq).setZero();
426  }
427  }
428  }

431 void Linefunctions::apply_linemixing_scaling_and_mirroring(
432  Eigen::Ref<Eigen::VectorXcd> F,
433  Eigen::Ref<Eigen::MatrixXcd> dF,
434  const Eigen::Ref<Eigen::VectorXcd> Fm,
435  const Eigen::Ref<Eigen::MatrixXcd> dFm,
436  const LineShape::Output& X,
437  const bool with_mirroring,
438  const AbsorptionLines& band,
439  const Index& line_ind,
440  const ArrayOfRetrievalQuantity& derivatives_data,
441  const LineShape::Output& dT,
442  const LineShape::Output& dVMR) {
443  auto nppd = derivatives_data.nelem();
444 
445  const Complex LM = Complex(1.0 + X.G, -X.Y);
446 
447  dF *= LM;
448  if (with_mirroring) {
449  dF.noalias() += dFm * conj(LM);
450 
451  for (auto iq = 0; iq < nppd; iq++) {
452  const auto& deriv = derivatives_data[iq];
453 
454  if (not propmattype(deriv)) continue;
455 
456  if (deriv == Jacobian::Atm::Temperature) {
457  const auto c = Complex(dT.G, -dT.Y);
458  dF.col(iq).noalias() += F * c + Fm * conj(c);
459  } else if (deriv.Target().needQuantumIdentity()) {
460  const Absorption::QuantumIdentifierLineTarget lt(deriv.Target().QuantumIdentity(), band, line_ind);
461  if (lt not_eq Absorption::QuantumIdentifierLineTargetType::None) {
462  if (deriv == Jacobian::Line::VMR and lt == Absorption::QuantumIdentifierLineTargetType::Isotopologue) {
463  const auto c = Complex(dVMR.G, -dVMR.Y);
464  dF.col(iq).noalias() += F * c + Fm * conj(c);
465  } else if (is_pressure_broadening_G(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
466  dF.col(iq).noalias() = F + Fm;
467  } else if (is_pressure_broadening_Y(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
468  dF.col(iq).noalias() = Complex(0, -1) * (F - Fm);
469  }
470  }
471  }
472  }
473  } else {
474  for (auto iq = 0; iq < nppd; iq++) {
475  const auto& deriv = derivatives_data[iq];
476 
477  if (not propmattype(deriv)) continue;
478 
479  if (deriv == Jacobian::Atm::Temperature) {
480  dF.col(iq).noalias() += F * Complex(dT.G, -dT.Y);
481  } else if (deriv.Target().needQuantumIdentity()) {
482  const Absorption::QuantumIdentifierLineTarget lt(deriv.Target().QuantumIdentity(), band, line_ind);
483  if (lt not_eq Absorption::QuantumIdentifierLineTargetType::None) {
484  if (deriv == Jacobian::Line::VMR and lt == Absorption::QuantumIdentifierLineTargetType::Isotopologue) {
485  dF.col(iq).noalias() += F * Complex(dVMR.G, -dVMR.Y);
486  } else if (is_pressure_broadening_G(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
487  dF.col(iq).noalias() = F;
488  } else if (is_pressure_broadening_Y(deriv) and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
489  dF.col(iq).noalias() = Complex(0, -1) * F;
490  }
491  }
492  }
493  }
494  }
495 
496  F *= LM;
497  if (with_mirroring) F.noalias() += Fm * std::conj(LM);
498 }

500 void Linefunctions::apply_rosenkranz_quadratic_scaling(
501  Eigen::Ref<Eigen::VectorXcd> F,
502  Eigen::Ref<Eigen::MatrixXcd> dF,
503  const Eigen::Ref<const Eigen::VectorXd> f_grid,
504  const Numeric& F0,
505  const Numeric& T,
506  const AbsorptionLines& band,
507  const Index& line_ind,
508  const ArrayOfRetrievalQuantity& derivatives_data) {
509  auto nf = f_grid.size(), nppd = derivatives_data.nelem();
510 
511  const Numeric invF0 = 1.0 / F0;
512  const Numeric mafac =
513  (Constant::h) / (2.0 * Constant::k * T) /
514  std::sinh((Constant::h * F0) / (2.0 * Constant::k * T)) * invF0;
515 
516  Numeric dmafac_dT_div_fun = 0;
517  if (do_temperature_jacobian(derivatives_data))
518  dmafac_dT_div_fun =
519  -(Constant::k * T -
520  F0 * Constant::h /
521  (2.0 * std::tanh(F0 * Constant::h / (2.0 * Constant::k * T)))) /
522  (Constant::k * T * T);
523 
524  Numeric fun;
525 
526  for (auto iv = 0; iv < nf; iv++) {
527  fun = mafac * (f_grid[iv] * f_grid[iv]);
528  F[iv] *= fun;
529 
530  for (auto iq = 0; iq < nppd; iq++) {
531  const auto& deriv = derivatives_data[iq];
532 
533  if (not propmattype(deriv)) continue;
534 
535  dF(iv, iq) *= fun;
536  if (deriv == Jacobian::Atm::Temperature) {
537  dF(iv, iq) += dmafac_dT_div_fun * F[iv];
538  } else if (is_frequency_parameter(deriv)) {
539  dF(iv, iq) += (2.0 / f_grid[iv]) * F[iv];
540  } else if (deriv.Target().needQuantumIdentity()) {
541  const Absorption::QuantumIdentifierLineTarget lt(deriv.Target().QuantumIdentity(), band, line_ind);
542  if (deriv == Jacobian::Line::Center and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
543  dF(iv, iq) +=
544  (-invF0 - Constant::h / (2.0 * Constant::k * T *
545  std::tanh(F0 * Constant::h /
546  (2.0 * Constant::k * T)))) *
547  F[iv];
548  }
549  }
550  }
551  }

554 void Linefunctions::apply_VVH_scaling(
555  Eigen::Ref<Eigen::VectorXcd> F,
556  Eigen::Ref<Eigen::MatrixXcd> dF,
557  Eigen::Ref<Eigen::Matrix<Complex, Eigen::Dynamic, ExpectedDataSize()>> data,
558  const Eigen::Ref<const Eigen::VectorXd> f_grid,
559  const Numeric& F0,
560  const Numeric& T,
561  const AbsorptionLines& band,
562  const Index& line_ind,
563  const ArrayOfRetrievalQuantity& derivatives_data) {
564  auto nppd = derivatives_data.nelem();
565 
566  // 2kT is constant for the loop
567  const Numeric kT = 2.0 * Constant::k * T;
568  const Numeric c1 = Constant::h / kT;
569 
570  // denominator is constant for the loop
571  const Numeric tanh_f0part = std::tanh(c1 * F0);
572  const Numeric denom = F0 * tanh_f0part;
573 
574  // Name the data
575  auto tanh_f = data.col(0);
576  auto ftanh_f = data.col(1);
577 
578  tanh_f.noalias() = (c1 * f_grid.array()).tanh().matrix();
579  ftanh_f.noalias() = f_grid.cwiseProduct(tanh_f) / denom;
580  F.array() *= ftanh_f.array();
581 
582  dF.array().colwise() *= ftanh_f.array();
583  for (auto iq = 0; iq < nppd; iq++) {
584  const auto& deriv = derivatives_data[iq];
585 
586  if (not propmattype(deriv)) continue;
587 
588  if (deriv == Jacobian::Atm::Temperature) {
589  dF.col(iq).noalias() +=
590  -c1 / T *
591  ((denom - F0 / tanh_f0part) * F - denom * ftanh_f.cwiseProduct(F) +
592  f_grid.cwiseProduct(F).cwiseQuotient(tanh_f));
593  } else if (is_frequency_parameter(deriv)) {
594  dF.col(iq).noalias() +=
595  F.cwiseQuotient(f_grid) +

751  const Numeric S_src = line.I0() * dS_dS0_src;

756  const auto& deriv = derivatives_data[iq];
757

759

762  S_abs *
763  (dstimulated_relative_emission_dT(gamma, gamma_ref, line.F0(), T) /
764  K2 +

772  K2 +

774  dboltzman_ratio_dT(K4, T, Evu) / K4 - invQT * dQT_dT);
775 
776  dN.col(iq).noalias() += F * (dS_dT_src - dS_dT_abs);
777  dF.col(iq).noalias() += F * dS_dT_abs;

780  if (deriv == Jacobian::Line::Strength and lt == Absorption::QuantumIdentifierLineTargetType::Line) {

783  } else if (deriv == Jacobian::Line::Center and lt == Absorption::QuantumIdentifierLineTargetType::Line) {

785  const Numeric dS_dF0_src = S_src * dstimulated_relative_emission_dF0(gamma, gamma_ref, T, T0) / K2;
786 
787  dN.col(iq).noalias() += F * (dS_dF0_src - dS_dF0_abs);
788  dF.col(iq).noalias() += F * dS_dF0_abs;
789  } else if (deriv == Jacobian::Line::NLTE and lt == Absorption::QuantumIdentifierLineTargetType::Level) {
790  if (lt.lower) {
791  const Numeric dS_dTl_abs = S_abs * dabsorption_nlte_rate_dTl(gamma, T, Tl, Evl, r_low) / K3;
792 
793  dN.col(iq).noalias() = -F * dS_dTl_abs;
794  dF.col(iq).noalias() = -dN.col(iq);
795  }
796 
797  if (lt.upper) {
798  const Numeric dS_dTu_abs = S_abs * dabsorption_nlte_rate_dTu(gamma, T, Tu, Evu, K4) / K3;
799  const Numeric dS_dTu_src = S_src * dboltzman_ratio_dT(K4, Tu, Evu) / K4;
800 
801  dN.col(iq).noalias() = F * (dS_dTu_src - dS_dTu_abs);
802  dF.col(iq).noalias() = F * dS_dTu_abs;
803  }
804  }
805  }
806  }
807 
808  N.noalias() = F * (S_src - S_abs);
809  F *= S_abs;
810 }

873 void Linefunctions::apply_linestrength_from_nlte_level_distributions(
874  Eigen::Ref<Eigen::VectorXcd> F,
875  Eigen::Ref<Eigen::MatrixXcd> dF,
876  Eigen::Ref<Eigen::VectorXcd> N,
877  Eigen::Ref<Eigen::MatrixXcd> dN,
878  const Numeric& r1,
879  const Numeric& r2,
880  const Numeric& g1,
881  const Numeric& g2,
882  const Numeric& A21,
883  const Numeric& F0,
884  const Numeric& T,
885  const AbsorptionLines& band,
886  const Index& line_ind,
887  const ArrayOfRetrievalQuantity& derivatives_data) {
888  // Size of the problem
889  auto nppd = derivatives_data.nelem();
890 
891  // Physical constants
892  constexpr Numeric c0 = 2.0 * Constant::h / Constant::pow2(Constant::c);
893  constexpr Numeric c1 = Constant::h / (4 * Constant::pi);
894 
895  // Constants based on input
896  const Numeric c2 = c0 * F0 * F0 * F0;
897  const Numeric c3 = c1 * F0;
898  const Numeric x = g2 / g1;
899 
900  /*
901  Einstein 'active' coefficients are as follows:
902  const Numeric B21 = A21 / c2;
903  const Numeric B12 = x * B21;
904  */
905 
906  // Absorption strength
907  const Numeric k = c3 * (r1 * x - r2) * (A21 / c2);
908 
909  // Emission strength
910  const Numeric e = c3 * r2 * A21;
911 
912  // Planck function of this line
913  const Numeric exp_T = std::exp(Constant::h / Constant::k * F0 / T),
914  b = c2 / (exp_T - 1);
915 
916  // Ratio between emission and absorption constant
917  const Numeric ratio = e / b - k;
918 
919  // Constants ALMOST everywhere inside these loops
920  dN.noalias() = dF * ratio;
921  dF *= k;
922 
923  // Partial derivatives
924  for (auto iq = 0; iq < nppd; iq++) {
925  const auto& deriv = derivatives_data[iq];
926 
927  if (not propmattype(deriv)) continue;
928 
929  if (deriv == Jacobian::Atm::Temperature) {
930  dN.col(iq).noalias() += F * e * Constant::h * F0 * exp_T / (c2 * Constant::k * T * T);
931  } else if (deriv.Target().needQuantumIdentity()) {
932  const Absorption::QuantumIdentifierLineTarget lt(deriv.Target().QuantumIdentity(), band, line_ind);
933  if (deriv == Jacobian::Line::Center and lt == Absorption::QuantumIdentifierLineTargetType::Line) {
934  Numeric done_over_b_df0 = Constant::h * exp_T / (c2 * Constant::k * T) - 3.0 * b / F0;
935  Numeric de_df0 = c1 * r2 * A21;
936  Numeric dk_df0 = c1 * (r1 * x - r2) * (A21 / c2) - 3.0 * k / F0;
937 
938  dN.col(iq).noalias() += F * (e * done_over_b_df0 + de_df0 / b - dk_df0);
939  dF.col(iq).noalias() += F * dk_df0;
940  } else if (deriv == Jacobian::Line::NLTE and lt == Absorption::QuantumIdentifierLineTargetType::Level) {
941  if (lt.lower) {
942  const Numeric dk_dr2 = -c3 * A21 / c2, de_dr2 = c3 * A21, dratio_dr2 = de_dr2 / b - dk_dr2;
943  dN.col(iq).noalias() = F * dratio_dr2;
944  dF.col(iq).noalias() = F * dk_dr2;
945  } else if (lt.upper) {
946  const Numeric dk_dr1 = c3 * x * A21 / c2;
947  dF.col(iq).noalias() = F * dk_dr1;
948  }
949  }
950  }
951  }
952 
953  // Set source function to be the relative amount emitted by the line divided by the Planck function
954  N.noalias() = F * ratio;
955 
956  // Set absorption
957  F *= k;
958 }
