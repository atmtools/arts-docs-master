144 
146  SpeciesRecord(const char name[],
147  const Index degfr,
148  const Array<IsotopologueRecord>& isotopologue)
149  : mname(name), mdegfr(degfr), misotopologue(isotopologue) {
150  // Thanks to Matpack, initialization of misotopologue with isotopologue
151  // should now work correctly.
152 
153 #ifndef NDEBUG
154  {
155  /* Check that the isotopologues are correctly sorted. */
156  for (Index i = 0; i < misotopologue.nelem() - 1; ++i) {
157  ARTS_ASSERT(std::isnan(misotopologue[i].Abundance()) ||
158  std::isnan(misotopologue[i + 1].Abundance()) ||
159  misotopologue[i].Abundance() >=
160  misotopologue[i + 1].Abundance());
161  }
162 
163  /* Check that the Mytran tags are correctly sorted. */
164  for (Index i = 0; i < misotopologue.nelem() - 1; ++i) {
165  if ((0 < misotopologue[i].MytranTag()) &&
166  (0 < misotopologue[i + 1].MytranTag())) {
167  ARTS_ASSERT(misotopologue[i].MytranTag() <
168  misotopologue[i + 1].MytranTag());
169 
170  // Also check that the tags have the same base number:
171  ARTS_ASSERT(misotopologue[i].MytranTag() / 10 ==
172  misotopologue[i].MytranTag() / 10);
173  }
174  }
175  }
176 #endif // #ifndef NDEBUG
177  }
178 
179  const String& Name() const { return mname; }

181  const Array<IsotopologueRecord>& Isotopologue() const {
