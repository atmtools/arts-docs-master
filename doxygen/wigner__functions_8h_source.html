99  static_assert(pos < 7 and pos > 0, "Only valid for pos := 1, 2, 3, 4, 5, and 6");
100 
101  if constexpr (pos == 1 or pos == 2) {
102  const Rational maxX = b - a;
103  std::pair<Rational, Rational> out {-maxX, a + b};
104  if (out.first > out.second) swap(out.first, out.second);
105  if (out.second > maxX) out.second = maxX;
106  if (out.first > maxX) out = {RATIONAL_UNDEFINED, RATIONAL_UNDEFINED};
107  return out;
108  } else if constexpr (pos == 3) {

111  if (maxX >= minX) return {minX, maxX};
112  else return {RATIONAL_UNDEFINED, RATIONAL_UNDEFINED};
113  } else {

115  pos == 5 ? abs(b) :
