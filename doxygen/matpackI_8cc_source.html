1245 void mult(MatrixView A, const ConstMatrixView& B, const ConstMatrixView& C) {
1246  // Check dimensions:
1247  ARTS_ASSERT(A.nrows() == B.nrows());
1248  ARTS_ASSERT(A.ncols() == C.ncols());
1249  ARTS_ASSERT(B.ncols() == C.nrows());
1250 
1251  // Catch trivial case if one of the matrices is empty.
1252  if ((B.nrows() == 0) || (B.ncols() == 0) || (C.ncols() == 0)) return;
1253 
1254  // Matrices B and C must be continuous in at least on dimension, C
1255  // must be continuous along the second dimension.
1256  if (((B.mrr.get_stride() == 1) || (B.mcr.get_stride() == 1)) &&
1257  ((C.mrr.get_stride() == 1) || (C.mcr.get_stride() == 1)) &&
1258  (A.mcr.get_stride() == 1)) {
1259  // BLAS uses column-major order while arts uses row-major order.
1260  // Hence instead of C = A * B we compute C^T = A^T * B^T!
1261 
1262  int k, m, n;
1263 
1264  k = (int)B.ncols();
1265  m = (int)C.ncols();
1266  n = (int)B.nrows();
1267 
1268  // Note also the clash in nomenclature: BLAS uses C = A * B while
1269  // arts uses A = B * C. Taking into accout this and the difference in
1270  // memory layouts, we need to map the MatrixViews A, B and C to the BLAS
1271  // arguments as follows:
1272  // A (arts) -> C (BLAS)
1273  // B (arts) -> B (BLAS)
1274  // C (arts) -> A (BLAS)
1275 
1276  // Char indicating whether A (BLAS) or B (BLAS) should be transposed.
1277  char transa, transb;
1278  // Sizes of the matrices along the direction in which they are
1279  // traversed.
1280  int lda, ldb, ldc;
1281 
1282  // Check if C (arts) is transposed.
1283  if (C.mrr.get_stride() == 1) {
1284  transa = 'T';
1285  lda = (int)C.mcr.get_stride();
1286  } else {
1287  transa = 'N';
1288  lda = (int)C.mrr.get_stride();
1289  }
1290 
1291  // Check if B (arts) is transposed.
1292  if (B.mrr.get_stride() == 1) {
1293  transb = 'T';
1294  ldb = (int)B.mcr.get_stride();
1295  } else {
1296  transb = 'N';
1297  ldb = (int)B.mrr.get_stride();
1298  }
1299 
1300  // In case B (arts) has only one column, column and row stride are 1.
1301  // We therefore need to set ldb to k, since dgemm_ requires lda to be at
1302  // least k / m if A is non-transposed / transposed.
1303  if ((B.mcr.get_stride() == 1) && (B.mrr.get_stride() == 1)) {
1304  transb = 'N';
1305  ldb = k;
1306  }
1307 
1308  // The same holds for C (arts).
1309  if ((C.mcr.get_stride() == 1) && (C.mrr.get_stride() == 1)) {
1310  transa = 'N';
1311  lda = m;
1312  }
1313 
1314  ldc = (int)A.mrr.get_stride();
1315  // The same holds for A (arts).
1316  if ((A.mcr.get_stride() == 1) && (A.mrr.get_stride() == 1)) {
1317  ldc = m;
1318  }
1319  double alpha = 1.0, beta = 0.0;
1320 
1321  dgemm_(&transa,
1322  &transb,
1323  &m,
1324  &n,
1325  &k,
1326  &alpha,
1327  C.mdata + C.mrr.get_start() + C.mcr.get_start(),
1328  &lda,
1329  B.mdata + B.mrr.get_start() + B.mcr.get_start(),
1330  &ldb,
1331  &beta,
1332  A.mdata + A.mrr.get_start() + A.mcr.get_start(),
1333  &ldc);
1334 
1335  } else {
1336  mult_general(A, B, C);
1337  }
