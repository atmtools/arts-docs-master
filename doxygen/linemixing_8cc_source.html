22  * in its derivatives.
23  *
24  * The problem in short:
25  * There is an Eigenvalue decomposition happening (W = V diag(e) V^-1)
26  *
27  * We use V and e in a strange way. We do not know how to get the

31  * How do we compute dV and de? We can get dW somewhat easily.

757 template <SpecialParam param>
758 ComplexVector ecs_absorption_zeeman_impl(const Numeric T,
759  const Numeric H,
760  const Numeric P,
761  const Numeric this_vmr,
762  const Vector& vmrs,
763  const Vector& mass,
764  const Vector& f_grid,
765  const Zeeman::Polarization zeeman_polarization,
766  const AbsorptionLines& band,
767  const SpeciesAuxData::AuxType& partition_type,
768  const ArrayOfGriddedField1& partition_data) {
769  constexpr Numeric sq_ln2pi = Constant::sqrt_ln_2 / Constant::sqrt_pi;
770 
771  // Weighted center of the band
772  const Numeric frenorm = band.F_mean();
773 
774  // Band Doppler broadening constant
775  const Numeric GD_div_F0 = Linefunctions::DopplerConstant(T, band.SpeciesMass());
776 
777  // Sorted population
778  const auto [sorting, tp] = sorted_population_and_dipole(T, band, partition_type, partition_data);
779 
780  // Relaxation matrix
781  const ComplexMatrix W = ecs_relaxation_matrix<param>(T, P, vmrs, mass, band, sorting, frenorm);
782 
783  // Equivalent lines computations
784  const EquivalentLines eqv(W, tp.pop, tp.dip);
785 
786  // Absorption of this band
787  ComplexVector absorption(f_grid.nelem(), 0);
788  for (Index i=0; i<band.NumLines(); i++) {
789  // Zeeman lines if necessary
790  const Index nz = band.ZeemanCount(i, zeeman_polarization);
791  for (Index j=0; j<nz; j++) {
792  const Numeric Sz = band.ZeemanStrength(i, zeeman_polarization, j);
793  const Numeric dzeeman = H * band.ZeemanSplitting(i, zeeman_polarization, j);
794 
795  const Numeric gamd = GD_div_F0 * (eqv.val[i].real() + frenorm + dzeeman);
796  const Numeric cte = Constant::sqrt_ln_2 / gamd;
797  for (Index iv=0; iv<f_grid.nelem(); iv++) {
798  const Complex z = (eqv.val[i] + frenorm + dzeeman - f_grid[iv]) * cte;
799  const Complex w = Faddeeva::w(z);
800  absorption[iv] += Sz * eqv.str[i] * w / gamd;
801  }
802  }
803  }
804 
805  // Adjust by frequency and number density
806  const Numeric numdens = this_vmr * number_density(P, T);
807  for (Index iv=0; iv<f_grid.nelem(); iv++) {
808  const Numeric f = f_grid[iv];
809  const Numeric fact = f * (1 - stimulated_emission(T, f));
810  absorption[iv] *= fact * numdens * sq_ln2pi;
811  }
812 
813  return absorption;
814 }
