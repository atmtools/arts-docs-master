21  * in its derivatives.
22  *
23  * The problem in short:
24  * There is an Eigenvalue decomposition happening (W = V diag(e) V^-1)

26  * We use V and e in a strange way. We do not know how to get the

357  const Rational Nf_p = band.LowerQuantumNumber(sorting[l], QuantumNumberType::N);

366  const Numeric b = wigner3j(Nf_p, Nf, L, 0, 0, 0);

368  const Numeric d = wigner6j(L, Jf, Jf_p, Sf, Nf_p, Nf);

373 
374  // Add to W and rescale to upwards element by the populations
375  W(k, l) = sum;
376  W(l, k) = sum * std::exp((erot(Nf_p, Jf_p) - erot(Nf, Jf)) / Conversion::kelvin2joule(T));
377  }

395  for (Index j=i+1; j<n; j++) {

727 ComplexVector ecs_absorption_impl(const Numeric T,
728  const Numeric P,
729  const Numeric this_vmr,
730  const Vector& vmrs,
731  const ErrorCorrectedSuddenData& ecs_data,
732  const Vector& f_grid,
733  const AbsorptionLines& band) {
734  constexpr Numeric sq_ln2pi = Constant::sqrt_ln_2 / Constant::sqrt_pi;
735 
736  // Weighted center of the band
737  const Numeric frenorm = band.F_mean();
738 
739  // Band Doppler broadening constant
740  const Numeric GD_div_F0 = Linefunctions::DopplerConstant(T, band.SpeciesMass());
741 
742  // Sorted population
743  auto [sorting, tp] = sorted_population_and_dipole(T, band);
744 
745  // Relaxation matrix
746  const ComplexMatrix W = ecs_relaxation_matrix(T, P, vmrs, ecs_data, band, sorting, frenorm);
747 
748  // Equivalent lines computations
749  const EquivalentLines eqv(W, tp.pop, tp.dip);
750 
751  // Absorption of this band
752  ComplexVector absorption(f_grid.nelem(), 0);
753  for (Index i=0; i<band.NumLines(); i++) {
754  const Numeric gamd = GD_div_F0 * (eqv.val[i].real() + frenorm);
755  const Numeric cte = Constant::sqrt_ln_2 / gamd;
756  for (Index iv=0; iv<f_grid.nelem(); iv++) {
757  const Complex z = (eqv.val[i] + frenorm - f_grid[iv]) * cte;
758  const Complex w = Faddeeva::w(z);
759  absorption[iv] += eqv.str[i] * w / gamd;
760  }
761  }
762 
763  // Adjust by frequency and number density
764  const Numeric numdens = this_vmr * number_density(P, T);
765  for (Index iv=0; iv<f_grid.nelem(); iv++) {
766  const Numeric f = f_grid[iv];
767  const Numeric fact = - f * std::expm1(- (Constant::h * f) / (Constant::k * T));
768  absorption[iv] *= fact * numdens * sq_ln2pi;
769 
770  ARTS_USER_ERROR_IF(isnan(absorption[iv]),
771  "There are NaNs in the absorption profile. The offending band\n"
772  "must not be fulfilling some of the conditions associated with\n"
773  "on-the-fly line mixing,\n\n"
774  "The full band metadata:\n", band.MetaData(), '\n',
775  "The full band data:\n", band)
776  }
777 
778  return absorption;
779 }
