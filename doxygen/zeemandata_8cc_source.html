41 Numeric case_b_g_coefficient_o2(Rational j,
42  Rational n,
43  Numeric GS,
44  Numeric GR,
45  Numeric GLE,
46  Numeric B,
47  Numeric D,
48  Numeric H,
49  Numeric gB,
50  Numeric gD,
51  Numeric gH,
52  Numeric lB,
53  Numeric lD,
54  Numeric lH) {
55  using Constant::pow2;
56  using Constant::pow3;
57  using std::atan2;
58  using std::cos;
59  using std::sin;
60  using std::sqrt;
61 
62  if (j.isUndefined() or n.isUndefined())
63  return NAN;
64  else if (j == 0)
65  return 0;
66 
67  auto J = j.toNumeric();
68 
69  auto nom = (lB + lD * (J * J + J + 1) + lH * pow2(J * J + J + 1)) * (2 * sqrt(J * J + J) / (2 * J + 1));
70 
71  auto denom =
72  B * J * (J - 1) - D * pow2(J * (J - 1)) + H * pow3(J * (J - 1)) +
73  (gB + gD * J * (J - 1) + gH * pow2(J * (J - 1))) * (J - 1) +
74  (lB + lD * J * (J - 1) + lH * pow2(J * (J - 1))) *
75  (2. / 3. - 2 * J / (2 * J + 1)) -
76  (B * (J + 2) * (J + 1) - D * pow2((J + 2) * (J + 1)) +
77  H * pow3((J + 2) * (J + 1)) -
78  (gB + gD * (J + 2) * (J + 1) + gH * pow2((J + 2) * (J + 1))) * (J + 2) +
79  (lB + lD * (J + 2) * (J + 1) + lH * pow2((J + 2) * (J + 1))) *
80  (2. / 3. - 2 * (J + 1) / (2 * J + 1)));
81 
82  auto phi = atan2(2 * nom, denom) / 2;
83 
84  if (j == n)
85  return (GS + GR) / (J * (J + 1)) - GR;
86  else if (j < n)
87  return (GS + GR) * (pow2(cos(phi)) / J - pow2(sin(phi)) / (J + 1)) +
88  2 * GLE * cos(2 * phi) / (2 * J + 1) - GR;
89  else /*if(j > n)*/
90  return (GS + GR) * (pow2(sin(phi)) / J - pow2(cos(phi)) / (J + 1)) -
91  2 * GLE * cos(2 * phi) / (2 * J + 1) - GR;
92 }

94 constexpr Numeric closed_shell_trilinear(Rational k,
95  Rational j,
96  Numeric gperp,
97  Numeric gpara)
98 {
99  if (k.isUndefined() or j.isUndefined() or j == 0)

229 Zeeman::Derived Zeeman::FromGrids(Numeric u, Numeric v, Numeric w, Numeric z, Numeric a) noexcept
230 {
231  Derived output;
232 
233  // If there is no magnetic field, bailout quickly
234  output.H = std::hypot(u, v, w);
235  if (output.H == 0) {
236  output = FromPreDerived(0, 0, 0);
237  } else {
238  // XYZ vectors normalized
239  const Eigen::Vector3d n = los_xyz_by_za_local(z, a);
240  const Eigen::Vector3d ev = ev_xyz_by_za_local(z, a);
241  const Eigen::Vector3d nH = los_xyz_by_uvw_local(u, v, w);
242 
243  // Normalized vector (which are also the magnetic field derivatives)
244  output.dH_dv = nH[0];
245  output.dH_du = nH[1];
246  output.dH_dw = nH[2];
247 
248  // Compute theta (and its derivatives if possible)
249  const Numeric cos_theta = n.dot(nH);
250  const Numeric sin_theta = std::sqrt(1 - Constant::pow2(cos_theta));
251  output.theta = std::acos(cos_theta);
252  if (sin_theta not_eq 0) {
253  const Eigen::Vector3d dtheta = (nH * cos_theta - n) / (output.H * sin_theta);
254  output.dtheta_dv = dtheta[0];
255  output.dtheta_du = dtheta[1];
256  output.dtheta_dw = dtheta[2];
257  } else {
258  output.dtheta_dv = 0;
259  output.dtheta_du = 0;
260  output.dtheta_dw = 0;
261  }
262 
263  // Compute eta (and its derivatives if possible)
264  const Eigen::Vector3d inplane = nH - nH.dot(n) * n;
265  const Numeric y = ev.cross(inplane).dot(n);
266  const Numeric x = ev.dot(inplane);
267  output.eta = std::atan2(y, x);
268  if (x not_eq 0 or y not_eq 0) {
269  const Eigen::Vector3d deta = n.cross(nH) / (output.H * (Constant::pow2(x) + Constant::pow2(y)));
270  output.deta_dv = deta[0];
271  output.deta_du = deta[1];
272  output.deta_dw = deta[2];
273  } else {
274  output.deta_dv = 0;
275  output.deta_du = 0;
276  output.deta_dw = 0;
277  }
278  }
279 
280  return output;
281 }

377 void sum(PropagationMatrix& pm, const ComplexVector& abs, const PolarizationVector& polvec) {
378  auto pol_real = polvec.attenuation();
379  auto pol_imag = polvec.dispersion();
380 
381  for (Index iv=0; iv<pm.NumberOfFrequencies(); iv++) {
382  pm.Kjj()[iv] += abs[iv].real() * pol_real[0];
383  pm.K12()[iv] += abs[iv].real() * pol_real[1];
384  pm.K13()[iv] += abs[iv].real() * pol_real[2];
385  pm.K14()[iv] += abs[iv].real() * pol_real[3];
386  pm.K23()[iv] += abs[iv].imag() * pol_imag[0];
387  pm.K24()[iv] += abs[iv].imag() * pol_imag[1];
388  pm.K34()[iv] += abs[iv].imag() * pol_imag[2];
389  }

392 void dsum(PropagationMatrix& pm,
393  const ComplexVector& abs,
394  const ComplexVector& dabs,
395  const PolarizationVector& polvec,
396  const PolarizationVector& dpolvec_dtheta,
397  const PolarizationVector& dpolvec_deta,
398  const Numeric dH,
399  const Numeric dt,
400  const Numeric de) {
401  auto pol_real = polvec.attenuation();
402  auto pol_imag = polvec.dispersion();
403  auto dp_dt_r = dpolvec_dtheta.attenuation();
404  auto dp_dt_i = dpolvec_dtheta.dispersion();
405  auto dp_de_r = dpolvec_deta.attenuation();
406  auto dp_de_i = dpolvec_deta.dispersion();
407 
408  for (Index iv=0; iv<pm.NumberOfFrequencies(); iv++) {
409  const auto da_r = (dt * dp_dt_r + de * dp_de_r);
410  const auto da_i = (dt * dp_dt_i + de * dp_de_i);
411  pm.Kjj()[iv] += dabs[iv].real() * dH * pol_real[0] + abs[iv].real() * da_r[0];
412  pm.K12()[iv] += dabs[iv].real() * dH * pol_real[1] + abs[iv].real() * da_r[1];
413  pm.K13()[iv] += dabs[iv].real() * dH * pol_real[2] + abs[iv].real() * da_r[2];
414  pm.K14()[iv] += dabs[iv].real() * dH * pol_real[3] + abs[iv].real() * da_r[3];
415  pm.K23()[iv] += dabs[iv].imag() * dH * pol_imag[0] + abs[iv].imag() * da_i[0];
416  pm.K24()[iv] += dabs[iv].imag() * dH * pol_imag[1] + abs[iv].imag() * da_i[1];
417  pm.K34()[iv] += dabs[iv].imag() * dH * pol_imag[2] + abs[iv].imag() * da_i[2];;
418  }
