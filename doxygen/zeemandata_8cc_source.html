40 Numeric case_b_g_coefficient_o2(Rational j,
41  Rational n,
42  Numeric GS,
43  Numeric GR,
44  Numeric GLE,
45  Numeric B,
46  Numeric D,
47  Numeric H,
48  Numeric gB,
49  Numeric gD,
50  Numeric gH,
51  Numeric lB,
52  Numeric lD,
53  Numeric lH) {
54  using Constant::pow2;
55  using Constant::pow3;
56  using std::atan2;
57  using std::cos;
58  using std::sin;
59  using std::sqrt;
60 
61  if (j.isUndefined() or n.isUndefined())
62  return NAN;
63  else if (j == 0)
64  return 0;
65 
66  auto J = j.toNumeric();
67 
68  auto nom = (lB + lD * (J * J + J + 1) + lH * pow2(J * J + J + 1)) * (2 * sqrt(J * J + J) / (2 * J + 1));
69 
70  auto denom =
71  B * J * (J - 1) - D * pow2(J * (J - 1)) + H * pow3(J * (J - 1)) +
72  (gB + gD * J * (J - 1) + gH * pow2(J * (J - 1))) * (J - 1) +
73  (lB + lD * J * (J - 1) + lH * pow2(J * (J - 1))) *
74  (2. / 3. - 2 * J / (2 * J + 1)) -
75  (B * (J + 2) * (J + 1) - D * pow2((J + 2) * (J + 1)) +
76  H * pow3((J + 2) * (J + 1)) -
77  (gB + gD * (J + 2) * (J + 1) + gH * pow2((J + 2) * (J + 1))) * (J + 2) +
78  (lB + lD * (J + 2) * (J + 1) + lH * pow2((J + 2) * (J + 1))) *
79  (2. / 3. - 2 * (J + 1) / (2 * J + 1)));
80 
81  auto phi = atan2(2 * nom, denom) / 2;
82 
83  if (j == n)
84  return (GS + GR) / (J * (J + 1)) - GR;
85  else if (j < n)
86  return (GS + GR) * (pow2(cos(phi)) / J - pow2(sin(phi)) / (J + 1)) +
87  2 * GLE * cos(2 * phi) / (2 * J + 1) - GR;
88  else /*if(j > n)*/
89  return (GS + GR) * (pow2(sin(phi)) / J - pow2(cos(phi)) / (J + 1)) -
90  2 * GLE * cos(2 * phi) / (2 * J + 1) - GR;
91 }

93 constexpr Numeric closed_shell_trilinear(Rational k,
94  Rational j,
95  Numeric gperp,
96  Numeric gpara)
97 {
98  if (k.isUndefined() or j.isUndefined() or j == 0)
99  return 0;
100  else
101  return gperp + (gperp + gpara) * Numeric(k*k / (j*(j+1)));

229 Zeeman::Derived Zeeman::FromGrids(Numeric u, Numeric v, Numeric w, Numeric z, Numeric a) noexcept
230 {
231  Derived output;
232 
233  // If there is no magnetic field, bailout quickly
234  output.H = std::hypot(u, v, w);
235  if (output.H == 0) {
236  output = FromPreDerived(0, 0, 0);
237  } else {
238  // XYZ vectors normalized
239  const Eigen::Vector3d n = los_xyz_by_za_local(z, a);
240  const Eigen::Vector3d ev = ev_xyz_by_za_local(z, a);
241  const Eigen::Vector3d nH = los_xyz_by_uvw_local(u, v, w);
242 
243  // Normalized vector (which are also the magnetic field derivatives)
244  output.dH_dv = nH[0];
245  output.dH_du = nH[1];
246  output.dH_dw = nH[2];
247 
248  // Compute theta (and its derivatives if possible)
249  const Numeric cos_theta = n.dot(nH);
250  const Numeric sin_theta = std::sqrt(1 - Constant::pow2(cos_theta));
251  output.theta = std::acos(cos_theta);
252  if (sin_theta not_eq 0) {
253  const Eigen::Vector3d dtheta = (nH * cos_theta - n) / (output.H * sin_theta);
254  output.dtheta_dv = dtheta[0];
255  output.dtheta_du = dtheta[1];
256  output.dtheta_dw = dtheta[2];
257  } else {
258  output.dtheta_dv = 0;
259  output.dtheta_du = 0;
260  output.dtheta_dw = 0;
261  }
262 
263  // Compute eta (and its derivatives if possible)
264  const Eigen::Vector3d inplane = nH - nH.dot(n) * n;
265  const Numeric y = ev.cross(inplane).dot(n);
266  const Numeric x = ev.dot(inplane);
267  output.eta = std::atan2(y, x);
268  if (x not_eq 0 or y not_eq 0) {
269  const Eigen::Vector3d deta = n.cross(nH) / (output.H * (Constant::pow2(x) + Constant::pow2(y)));
270  output.deta_dv = deta[0];
271  output.deta_du = deta[1];
272  output.deta_dw = deta[2];
273  } else {
274  output.deta_dv = 0;
275  output.deta_du = 0;
276  output.deta_dw = 0;
277  }
278  }
279 
280  return output;
281 }

314  Numeric eta) noexcept {
315  const Numeric ST = std::sin(theta), CT = std::cos(theta), ST2 = ST * ST,
316  CT2 = CT * CT, ST2C2E = ST2 * std::cos(2 * eta),
317  ST2S2E = ST2 * std::sin(2 * eta);
318 
319  AllPolarizationVectors pv;
320  pv.sm = PolarizationVector(

322  pv.pi =

324  pv.sp = PolarizationVector(

326  return pv;

329 AllPolarizationVectors AllPolarization_dtheta(
330  Numeric theta, const Numeric eta) noexcept {
331  const Numeric ST = std::sin(theta), CT = std::cos(theta),
332  C2E = std::cos(2 * eta), S2E = std::sin(2 * eta), dST = CT,
333  dST2 = 2 * ST * dST, dCT = -ST, dST2C2E = dST2 * C2E,
334  dST2S2E = dST2 * S2E, dCT2 = 2 * CT * dCT;

336  AllPolarizationVectors pv;

338  dCT2, dST2C2E, dST2S2E, 2 * dCT, 4 * dCT, 2 * dST2S2E, -2 * dST2C2E);
339  pv.pi = PolarizationVector(
340  dST2, -dST2C2E, -dST2S2E, 0, 0, -2 * dST2S2E, 2 * dST2C2E);
341  pv.sp = PolarizationVector(
342  dCT2, dST2C2E, dST2S2E, -2 * dCT, -4 * dCT, 2 * dST2S2E, -2 * dST2C2E);

377 void sum(PropagationMatrix& pm, const ComplexVectorView& abs, const PolarizationVector& polvec, const bool do_phase) ARTS_NOEXCEPT {
