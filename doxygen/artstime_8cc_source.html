79 ArrayOfIndex time_steps(const ArrayOfTime& times, const TimeStep& DT)
80 {
81  Index N = times.nelem();
82  ARTS_USER_ERROR_IF (N < 2,
83  "Can only find time steps for 2-long or longer time grids");
84 
85  // algorithm only works with absolute times
86  const TimeStep dt = std::chrono::abs(DT);
87 
88  ArrayOfIndex time_steps{0};
89 
90  if (N > 2) {
91  Time tupp = next_even(times[0], dt);
92  time_steps.emplace_back(1);
93 
94  for (; time_steps.back() < N; time_steps.back()++) {
95  if (not (times[time_steps.back()] < tupp)) {
96  tupp = next_even(tupp, dt);
97  time_steps.emplace_back(time_steps.back());
98  }
99  }
100  } else {
101  time_steps.emplace_back(N);
102  }
103  return time_steps;
104 }

153 Time mean_time(const ArrayOfTime& ts, Index s, Index E)
154 {
155  Index e=0;
156  if (e == -1) e = ts.nelem();
157  ARTS_USER_ERROR_IF (e < 0 or e > ts.nelem(),

162 
163  Time::InternalTimeStep dt(0);
164  for (Index i=s+1; i<e; i++)
165  dt += (ts[i] - ts[s]) / (e - s);
166  return ts[s] + dt;
