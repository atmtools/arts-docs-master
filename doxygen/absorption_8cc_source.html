83 void SpeciesAuxData::setParam(const String& artstag,
84  const String& auxtype,
85  const ArrayOfGriddedField1& auxdata) {
86  // Global species lookup data:
87  using global_data::species_data;
88 
89  // We need a species index sorted by Arts identifier. Keep this in a
90  // static variable, so that we have to do this only once. The ARTS
91  // species index is ArtsMap[<Arts String>].
92  static map<String, SpecIsoMap> ArtsMap;
93 
94  // Remember if this stuff has already been initialized:
95  static bool hinit = false;
96 
97  if (!hinit) {
98  for (Index i = 0; i < species_data.nelem(); ++i) {
99  const SpeciesRecord& sr = species_data[i];
100  for (Index j = 0; j < sr.Isotopologue().nelem(); ++j) {
101  SpecIsoMap indicies(i, j);
102  String buf = sr.Name() + "-" + sr.Isotopologue()[j].Name();
103  ArtsMap[buf] = indicies;
104  }
105  }
106  hinit = true;
107  }
108 
109  Index species;
110  Index isotopologue;
111 
112  // ok, now for the cool index map:
113  // is this arts identifier valid?
114  const map<String, SpecIsoMap>::const_iterator i = ArtsMap.find(artstag);
115  ARTS_USER_ERROR_IF (i == ArtsMap.end(),
116  "ARTS Tag: ", artstag, " is unknown.")
117 
118  SpecIsoMap id = i->second;
119 
120  // Set mspecies:
121  species = id.Speciesindex();
122 
123  // Set misotopologue:
124  isotopologue = id.Isotopologueindex();
125 
126  Index this_auxtype = 0;
127 
128  while (this_auxtype < AT_FINAL_ENTRY &&
129  auxtype != SpeciesAuxTypeNames[this_auxtype])
130  this_auxtype++;
131 
132  if (this_auxtype != AT_FINAL_ENTRY) {
133  setParam(species, isotopologue, (AuxType)this_auxtype, auxdata);
134  } else {
135  ostringstream os;
136  os << "Unknown SpeciesAuxData type: " << auxtype;
137  std::runtime_error(os.str());
138  }
139 }

163 bool SpeciesAuxData::ReadFromStream(String& artsid,
164  istream& is,
165  Index nparams,
166  const Verbosity& verbosity) {
167  CREATE_OUT3;
168 
169  // Global species lookup data:
170  using global_data::species_data;
171 
172  // We need a species index sorted by Arts identifier. Keep this in a
173  // static variable, so that we have to do this only once. The ARTS
174  // species index is ArtsMap[<Arts String>].
175  static map<String, SpecIsoMap> ArtsMap;
176 
177  // Remember if this stuff has already been initialized:
178  static bool hinit = false;
179 
180  if (!hinit) {
181  out3 << " ARTS index table:\n";
182 
183  for (Index i = 0; i < species_data.nelem(); ++i) {
184  const SpeciesRecord& sr = species_data[i];
185 
186  for (Index j = 0; j < sr.Isotopologue().nelem(); ++j) {
187  SpecIsoMap indicies(i, j);
188  String buf = sr.Name() + "-" + sr.Isotopologue()[j].Name();
189 
190  ArtsMap[buf] = indicies;
191 
192  // Print the generated data structures (for debugging):
193  // The explicit conversion of Name to a c-String is
194  // necessary, because setw does not work correctly for
195  // stl Strings.
196  const Index& i1 = ArtsMap[buf].Speciesindex();
197  const Index& i2 = ArtsMap[buf].Isotopologueindex();
198 
199  out3 << " Arts Identifier = " << buf << " Species = " << setw(10)
200  << setiosflags(ios::left) << species_data[i1].Name().c_str()
201  << "iso = " << species_data[i1].Isotopologue()[i2].Name().c_str()
202  << "\n";
203  }
204  }
205  hinit = true;
206  }
207 
208  // This always contains the rest of the line to parse. At the
209  // beginning the entire line. Line gets shorter and shorter as we
210  // continue to extract stuff from the beginning.
211  String line;
212 
213  // Look for more comments?
214  bool comment = true;
215 
216  while (comment) {
217  // Return true if eof is reached:
218  if (is.eof()) return true;
219 
220  // Throw runtime_error if stream is bad:
221  ARTS_USER_ERROR_IF (!is, "Stream bad.");
222 
223  // Read line from file into linebuffer:
224  getline(is, line);
225 
226  // It is possible that we were exactly at the end of the file before
227  // calling getline. In that case the previous eof() was still false
228  // because eof() evaluates only to true if one tries to read after the
229  // end of the file. The following check catches this.
230  if (line.nelem() == 0 && is.eof()) return true;
231 
232  // @ as first character marks catalogue entry
233  char c;
234  extract(c, line, 1);
235 
236  // check for empty line
237  if (c == '@') {
238  comment = false;
239  }
240  }
241 
242  // read the arts identifier String
243  istringstream icecream(line);
244 
245  icecream >> artsid;
246 
247  if (artsid.length() != 0) {
248  Index mspecies;
249  Index misotopologue;
250 
251  // ok, now for the cool index map:
252  // is this arts identifier valid?
253  const map<String, SpecIsoMap>::const_iterator i = ArtsMap.find(artsid);
254  ARTS_USER_ERROR_IF (i == ArtsMap.end(),
255  "ARTS Tag: ", artsid, " is unknown.")
256 
257  SpecIsoMap id = i->second;
258 
259  // Set mspecies:
260  mspecies = id.Speciesindex();
261 
262  // Set misotopologue:
263  misotopologue = id.Isotopologueindex();
264 
265  ArrayOfGriddedField1 ratios;
266  ratios.resize(1);
267  // Extract accuracies:
268  try {
269  Numeric p = NAN;
270  std::vector<Numeric> aux;
271  for (Index ip = 0; ip < nparams; ip++) {
272  icecream >> double_imanip() >> p;
273  aux.push_back(p);
274  }
275 
276  Vector grid;
277  if (aux.size() > 1)
278  nlinspace(grid, 1, (Numeric)aux.size(), aux.size());
279  else
280  grid = Vector(1, .1);
281 
282  ratios[0].set_grid(0, grid);
283  ratios[0].data = aux;
284  mparams[mspecies][misotopologue] = ratios;
285  } catch (const runtime_error&) {
286  ARTS_USER_ERROR ("Error reading SpeciesAuxData.");
287  }
288  }
289 
290  // That's it!
291  return false;
292 }

294 void checkIsotopologueRatios(const ArrayOfArrayOfSpeciesTag& abs_species,
295  const SpeciesAuxData& isoratios) {
296  using global_data::species_data;
297 
298  // Check total number of species:
299  ARTS_USER_ERROR_IF (species_data.nelem() != isoratios.nspecies(),
300  "Number of species in SpeciesAuxData (", isoratios.nspecies(),
301  ") " "does not fit builtin species data (",
302  species_data.nelem(), ").")
303 
304  // For the selected species, we check all isotopes by looping over the
305  // species data. (Trying to check only the isotopes actually used gets
306  // quite complicated, actually, so we do the simple thing here.)
307 
308  // Loop over the absorption species:
309  for (Index i = 0; i < abs_species.nelem(); i++) {
310  // sp is the index of this species in the internal lookup table
311  const Index sp = abs_species[i][0].Species();
312 
313  // Get handle on species data for this species:
314  const SpeciesRecord& this_sd = species_data[sp];
315 
316  // Check number of isotopologues:
317  ARTS_USER_ERROR_IF (this_sd.Isotopologue().nelem() != isoratios.nisotopologues(sp),
318  "Incorrect number of isotopologues in isotopologue data.\n"
319  "Species: ", this_sd.Name(), ".\n"
320  "Number of isotopes in SpeciesAuxData (",
321  isoratios.nisotopologues(sp), ") "
322  "does not fit builtin species data (",
323  this_sd.Isotopologue().nelem(), ").")
324 
325  for (Index iso = 0; iso < this_sd.Isotopologue().nelem(); ++iso) {
326  // For "real" species (not representing continau) the isotopologue
327  // ratio must not be NAN or below zero.
328  if (!this_sd.Isotopologue()[iso].isContinuum()) {
329  ARTS_USER_ERROR_IF (std::isnan(isoratios.getParam(sp, iso)[0].data[0]) ||
330  isoratios.getParam(sp, iso)[0].data[0] < 0.,
331  "Invalid isotopologue ratio.\n"
332  "Species: ", this_sd.Name(), "-",
333  this_sd.Isotopologue()[iso].Name(), "\n"
334  "Ratio: ", isoratios.getParam(sp, iso)[0].data[0])
335  }
336  }
337  }
338 }

594 void xsec_species(Matrix& xsec,
595  Matrix& source,
596  Matrix& phase,
597  ArrayOfMatrix& dxsec_dx,
598  ArrayOfMatrix& dsource_dx,
599  ArrayOfMatrix& dphase_dx,
600  const ArrayOfRetrievalQuantity& jacobian_quantities,
601  const Vector& f_grid,
602  const Vector& abs_p,
603  const Vector& abs_t,
604  const EnergyLevelMap& abs_nlte,
605  const Matrix& abs_vmrs,
606  const ArrayOfArrayOfSpeciesTag& abs_species,
607  const AbsorptionLines& band,
608  const Numeric& isot_ratio,
609  const SpeciesAuxData::AuxType& partfun_type,
610  const ArrayOfGriddedField1& partfun_data) {
611  // Size of problem
612  const Index np = abs_p.nelem(); // number of pressure levels
613  const Index nf = f_grid.nelem(); // number of Dirac frequencies
614  const Index nl = band.NumLines(); // number of lines in the catalog
615  const Index nj = jacobian_quantities.nelem(); // number of partial derivatives
616  const Index nt = source.nrows(); // number of energy levels in NLTE
617 
618  // Type of problem
619  const bool do_nonlte = nt;
620 
621  Linefunctions::InternalData scratch(nf, nj);
622  Linefunctions::InternalData sum(nf, nj);
623 
624  // Test if the size of the problem is 0
625  if (not np or not nf or not nl) return;
626 
627  // Constant for all lines
628  const Numeric QT0 = single_partition_function(band.T0(), partfun_type, partfun_data);
629 
630  ArrayOfString fail_msg;
631  bool do_abort = false;
632 
633 #pragma omp parallel for if (!arts_omp_in_parallel() && np > 1) \
634  firstprivate(scratch, sum)
635  for (Index ip = 0; ip < np; ip++) {
636  if (do_abort) continue;
637  try {
638  // Constants for this level
639  const Numeric& temperature = abs_t[ip];
640  const Numeric& pressure = abs_p[ip];
641 
642  // Constants for this level
643  const Numeric QT =
644  single_partition_function(temperature, partfun_type, partfun_data);
645  const Numeric dQTdT = dsingle_partition_function_dT(
646  temperature,
647  partfun_type,
648  partfun_data);
649  const Numeric DC =
650  Linefunctions::DopplerConstant(temperature, band.SpeciesMass());
651  const Numeric dDCdT = Linefunctions::dDopplerConstant_dT(temperature, DC);
652  const Vector line_shape_vmr =
653  band.BroadeningSpeciesVMR(abs_vmrs(joker, ip), abs_species);
654 
655  Linefunctions::set_cross_section_of_band(scratch,
656  sum,
657  f_grid,
658  band,
659  jacobian_quantities,
660  line_shape_vmr,
661  abs_nlte[ip],
662  pressure,
663  temperature,
664  isot_ratio,
665  0,
666  DC,
667  dDCdT,
668  QT,
669  dQTdT,
670  QT0,
671  false);
672 
673  // absorption cross-section
674  MapToEigen(xsec).col(ip).noalias() += sum.F.real();
675  for (Index j = 0; j < nj; j++) {
676  if (not propmattype_index(jacobian_quantities, j)) continue;
677  MapToEigen(dxsec_dx[j]).col(ip).noalias() += sum.dF.col(j).real();
678  }
679 
680  // phase cross-section
681  if (not phase.empty()) {
682  MapToEigen(phase).col(ip).noalias() += sum.F.imag();
683  for (Index j = 0; j < nj; j++) {
684  if (not propmattype_index(jacobian_quantities, j)) continue;
685  MapToEigen(dphase_dx[j]).col(ip).noalias() += sum.dF.col(j).imag();
686  }
687  }
688 
689  // source ratio cross-section
690  if (do_nonlte) {
691  MapToEigen(source).col(ip).noalias() += sum.N.real();
692  for (Index j = 0; j < nj; j++) {
693  if (not propmattype_index(jacobian_quantities, j)) continue;
694  MapToEigen(dsource_dx[j]).col(ip).noalias() += sum.dN.col(j).real();
695  }
696  }
697  } catch (const std::runtime_error& e) {
698  ostringstream os;
699  os << "Runtime-error in cross-section calculation at p_abs index " << ip
700  << ": \n";
701  os << e.what();
