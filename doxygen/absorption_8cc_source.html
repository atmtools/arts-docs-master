86 Numeric wavenumber_to_joule(Numeric e) {

124 void xsec_species(Matrix& xsec,
125  Matrix& source,
126  Matrix& phase,
127  ArrayOfMatrix& dxsec_dx,
128  ArrayOfMatrix& dsource_dx,
129  ArrayOfMatrix& dphase_dx,
130  const ArrayOfRetrievalQuantity& jacobian_quantities,
131  const Vector& f_grid,
132  const Vector& abs_p,
133  const Vector& abs_t,
134  const EnergyLevelMap& abs_nlte,
135  const Matrix& abs_vmrs,
136  const ArrayOfArrayOfSpeciesTag& abs_species,
137  const AbsorptionLines& band,
138  const Numeric& isot_ratio) {
139  // Size of problem
140  const Index np = abs_p.nelem(); // number of pressure levels
141  const Index nf = f_grid.nelem(); // number of Dirac frequencies
142  const Index nl = band.NumLines(); // number of lines in the catalog
143  const Index nj = jacobian_quantities.nelem(); // number of partial derivatives
144  const Index nt = source.nrows(); // number of energy levels in NLTE
145 
146  // Type of problem
147  const bool do_nonlte = nt;
148 
149  Linefunctions::InternalData scratch(nf, nj);
150  Linefunctions::InternalData sum(nf, nj);
151 
152  // Test if the size of the problem is 0
153  if (not np or not nf or not nl) return;
154 
155  // Constant for all lines
156  const Numeric QT0 = single_partition_function(band.T0(), band.Isotopologue());
157 
158  ArrayOfString fail_msg;
159  bool do_abort = false;
160 
161 #pragma omp parallel for if (!arts_omp_in_parallel() && np > 1) \
162  firstprivate(scratch, sum)
163  for (Index ip = 0; ip < np; ip++) {
164  if (do_abort) continue;
165  try {
166  // Constants for this level
167  const Numeric& temperature = abs_t[ip];
168  const Numeric& pressure = abs_p[ip];
169 
170  // Constants for this level
171  const Numeric QT =
172  single_partition_function(temperature, band.Isotopologue());
173  const Numeric dQTdT = dsingle_partition_function_dT(
174  temperature, band.Isotopologue());
175  const Numeric DC =
176  Linefunctions::DopplerConstant(temperature, band.SpeciesMass());
177  const Numeric dDCdT = Linefunctions::dDopplerConstant_dT(temperature, DC);
178  const Vector line_shape_vmr =
179  band.BroadeningSpeciesVMR(abs_vmrs(joker, ip), abs_species);
180 
181  Linefunctions::set_cross_section_of_band(scratch,
182  sum,
183  f_grid,
184  band,
185  jacobian_quantities,
186  line_shape_vmr,
187  abs_nlte[ip],
188  pressure,
189  temperature,
190  isot_ratio,
191  0,
192  DC,
193  dDCdT,
194  QT,
195  dQTdT,
196  QT0,
197  false);
198 
199  // absorption cross-section
200  MapToEigen(xsec).col(ip).noalias() += sum.F.real();
201  for (Index j = 0; j < nj; j++) {
202  if (not jacobian_quantities[j].propmattype()) continue;
203  MapToEigen(dxsec_dx[j]).col(ip).noalias() += sum.dF.col(j).real();
204  }
205 
206  // phase cross-section
207  if (not phase.empty()) {
208  MapToEigen(phase).col(ip).noalias() += sum.F.imag();
209  for (Index j = 0; j < nj; j++) {
210  if (not jacobian_quantities[j].propmattype()) continue;
211  MapToEigen(dphase_dx[j]).col(ip).noalias() += sum.dF.col(j).imag();
212  }
213  }
214 
215  // source ratio cross-section
216  if (do_nonlte) {
217  MapToEigen(source).col(ip).noalias() += sum.N.real();
218  for (Index j = 0; j < nj; j++) {
219  if (not jacobian_quantities[j].propmattype()) continue;
220  MapToEigen(dsource_dx[j]).col(ip).noalias() += sum.dN.col(j).real();
221  }
222  }
223  } catch (const std::runtime_error& e) {
224  ostringstream os;
225  os << "Runtime-error in cross-section calculation at p_abs index " << ip
226  << ": \n";
227  os << e.what();
228 #pragma omp critical(xsec_species_cross_sections)
229  {
230  do_abort = true;
231  fail_msg.push_back(os.str());
232  }
233  }
234  }
235 
236  ARTS_USER_ERROR_IF (do_abort,
237  "Error messages from failed cases:\n", fail_msg)
238 }
