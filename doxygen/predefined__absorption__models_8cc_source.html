218  using Constant::inv_sqrt_pi;
219  using Constant::pow2;
220  using Constant::pow3;
221 
222  // Central frequency [Hz]
223  constexpr std::array<Numeric, nlines_mpm2020> f0 = {
224  1.18750334E+11, 5.6264774E+10, 6.2486253E+10, 5.8446588E+10, 6.0306056E+10,
225  5.9590983E+10, 5.9164204E+10, 6.0434778E+10, 5.8323877E+10, 6.1150562E+10,
226  5.7612486E+10, 6.1800158E+10, 5.6968211E+10, 6.2411220E+10, 5.6363399E+10,
227  6.2997984E+10, 5.5783815E+10, 6.3568526E+10, 5.5221384E+10, 6.4127775E+10,
228  5.4671180E+10, 6.4678910E+10, 5.4130025E+10, 6.5224078E+10, 5.3595775E+10,
229  6.5764779E+10, 5.3066934E+10, 6.6302096E+10, 5.2542418E+10, 6.6836834E+10,
230  5.2021429E+10, 6.7369601E+10, 5.1503360E+10, 6.7900868E+10, 5.0987745E+10,
231  6.8431006E+10, 5.0474214E+10, 6.8960312E+10, 3.68498246E+11, 4.24763020E+11,

233 
234  // Intensity [1 / Pa] (rounded to 10 digits because at most 9 digits exist in f0)
235  constexpr std::array<Numeric, nlines_mpm2020> intens = {
236  1.591521878E-21, 1.941172240E-21, 4.834543970E-21, 4.959264029E-21, 7.010386457E-21,
237  7.051673348E-21, 8.085012578E-21, 8.108262250E-21, 8.145673278E-21, 8.149757320E-21,
238  7.396406085E-21, 7.401923754E-21, 6.162286575E-21, 6.168475265E-21, 4.749226167E-21,
239  4.754435107E-21, 3.405982896E-21, 3.408455562E-21, 2.282498656E-21, 2.283934341E-21,
240  1.432692459E-21, 1.433513473E-21, 8.439995690E-22, 8.443521837E-22, 4.672706507E-22,
241  4.676049313E-22, 2.435008301E-22, 2.437304596E-22, 1.195038747E-22, 1.196873412E-22,
242  5.532759045E-23, 5.537261239E-23, 2.416832398E-23, 2.418989865E-23, 9.969285671E-24,

246  // Temperature intensity modifier
247  constexpr std::array<Numeric, nlines_mpm2020> a2 = {
248  0.01, 0.014, 0.083, 0.083, 0.207, 0.207, 0.387, 0.386,
249  0.621, 0.621, 0.910, 0.910, 1.255, 1.255, 1.654, 1.654,
250  2.109, 2.108, 2.618, 2.617, 3.182, 3.181, 3.800, 3.800,
251  4.474, 4.473, 5.201, 5.200, 5.983, 5.982, 6.819, 6.818,
252  7.709, 7.708, 8.653, 8.652, 9.651, 9.650, 0.048, 0.044,
253  0.049, 0.145, 0.141, 0.145};
254 
255  // Line shape model in SI units
256  constexpr auto lsm = init_mpm2020_lsm();
257 
258  // Reference temperature [K]
259  constexpr Numeric t0 = 300;
260 
261  // QuantumIdentifier if we need it
262  constexpr std::array<QuantumIdentifier, nlines_mpm2020> qids = init_mpm2020_qids();
263  constexpr Numeric mass = qids.front().Isotopologue().mass;
264 
265  // Model setting
266  const bool do_temp_deriv = do_temperature_jacobian(jacs);
267 
268  // Per pressure level
269  for (Index ip=0; ip<p.nelem(); ip++) {
270  const Numeric theta = t0 / t[ip];
271  const Numeric theta_m1 = theta - 1;
272  const Numeric theta_3 = pow3(theta);
273  const Numeric GD_div_F0 = Linefunctions::DopplerConstant(t[ip], mass);
274 
275  for (std::size_t i=0; i<nlines_mpm2020; i++) {
276  const Numeric invGD = 1 / (GD_div_F0 * f0[i]);
277  const Numeric fac = sqrt_pi * invGD;
278  const Numeric ST = theta_3 * p[ip] * intens[i] * std::exp(-a2[i] * theta_m1);
279  const Numeric G0 = (1 + 0.1*water_vmr[ip]) * p[ip] * lsm[i].G0().at(t[ip], t0);
280  const Numeric Y = p[ip] * lsm[i].Y().at(t[ip], t0);
281  const Numeric G = pow2( p[ip]) * lsm[i].G().at(t[ip], t0);
282  const Numeric DV = pow2(p[ip]) * lsm[i].DV().at(t[ip], t0);
283 
284  const Numeric dinvGD_dT = do_temp_deriv ? - invGD * Linefunctions::dDopplerConstant_dT(t[ip], GD_div_F0) : 0;
285  const Numeric dST_dT = do_temp_deriv ? (a2[i]*t0 - 3*t[ip]) / pow2(t[ip]) * ST : 0;
286  const Numeric dG0_dT = do_temp_deriv ? (1 + 0.1*water_vmr[ip]) * p[ip] * lsm[i].G0().dT(t[ip], t0) : 0;
287  const Numeric dY_dT = do_temp_deriv ? p[ip] * lsm[i].Y().dT(t[ip], t0) : 0;
288  const Numeric dG_dT = do_temp_deriv ? pow2(p[ip]) * lsm[i].G().dT(t[ip], t0) : 0;
289  const Numeric dDV_dT = do_temp_deriv ? pow2(p[ip]) * lsm[i].DV().dT(t[ip], t0) : 0;
290 
291  for (Index j=0; j<f.nelem(); j++) {
292  const Complex z = Complex(f0[i] + DV - f[j], G0) * invGD;
293  const Complex Fv = fac * Faddeeva::w(z);
294  const Complex Flm = 1 / Complex(G0, f[j] + f0[i] + DV);
295 
296  const Complex abs = std::real(
297  Complex(1 + G, Y) * Fv +
298  Complex(1 + G, -Y) * Flm);
299 
300  xsec(j, ip) += ST * pow2(f[j]) * abs.real();
301 
302  if (jacs.nelem()) {
303  const Complex dw = 2 * (Complex(0, fac * inv_sqrt_pi) - z * Fv);
304  const Complex dm = - pi * pow2(Flm);
305 
306  for (Index iq=0; iq<jacs.nelem(); iq++) {
307  const auto& deriv = jacs[iq];
308 
309  if (not deriv.propmattype()) continue;
310 
311 
312  if (deriv == Jacobian::Atm::Temperature) {
313  const Complex dFv = dw * (invGD * Complex(dDV_dT, dG0_dT) - dinvGD_dT) + Fv * dinvGD_dT;
314  const Complex dFlm = dm * Complex(dG0_dT, dDV_dT);
315  dxsec[iq](j, ip) += pow2(f[j]) * (ST * std::real(
316  Complex(1 + G, Y) * dFv + Complex(dG_dT, dY_dT) * Fv +
317  Complex(1 + G, -Y) * dFlm + Complex(G, -dY_dT) * Flm) + abs.real() * dST_dT);
318  } else if (is_frequency_parameter(deriv)) {
319  const Complex dFv = - dw * invGD;
320  const Complex dFlm = Complex(0, 1) * dm;
321  dxsec[iq](j, ip) += ST * (pow2(f[j]) * std::real(
322  Complex(1 + G, Y) * dFv +
323  Complex(1 + G, -Y) * dFlm) + 2 * abs.real() * f[j]);
324  } else if (deriv.Target().needQuantumIdentity()) {
325  const Absorption::QuantumIdentifierLineTarget lt(deriv.QuantumIdentity(), qids[i]);
326 
327  //NOTE: This is a special case where each line must be seen as a "Band" by themselves.
328  //NOTE: (cont) This is because we never check for "Line" unless a full Absorption::Lines
329  //NOTE: (cont) is used in the QuantumIdentifierLineTarget struct.
330  if (lt not_eq Absorption::QuantumIdentifierLineTargetType::Band) continue;
331 
332  if (deriv == Jacobian::Line::ShapeG0X0) {
333  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(
334  Complex(1 + G, Y) * Complex(0, 1) * dw * invGD +
335  Complex(1 + G, -Y) * dm) *
336  lsm[i].G0().dX0(t[ip], t0);
337  } else if (deriv == Jacobian::Line::ShapeG0X1) {
338  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(
339  Complex(1 + G, Y) * Complex(0, 1) * dw * invGD +
340  Complex(1 + G, -Y) * dm) *
341  lsm[i].G0().dX1(t[ip], t0);
342  } else if (deriv == Jacobian::Line::ShapeDVX0) {
343  const Complex dFv = dw * invGD;
344  const Complex dFlm = Complex(0, 1) * dm;
345  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(
346  Complex(1 + G, Y) * dFv +
347  Complex(1 + G, -Y) * dFlm) *
348  lsm[i].DV().dX0(t[ip], t0);
349  } else if (deriv == Jacobian::Line::ShapeDVX1) {
350  const Complex dFv = dw * invGD;
351  const Complex dFlm = Complex(0, 1) * dm;
352  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(
353  Complex(1 + G, Y) * dFv +
354  Complex(1 + G, -Y) * dFlm) *
355  lsm[i].DV().dX1(t[ip], t0);
356  } else if (deriv == Jacobian::Line::ShapeDVX2) {
357  const Complex dFv = dw * invGD;
358  const Complex dFlm = Complex(0, 1) * dm;
359  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(
360  Complex(1 + G, Y) * dFv +
361  Complex(1 + G, -Y) * dFlm) *
362  lsm[i].DV().dX2(t[ip], t0);
363  } else if (deriv == Jacobian::Line::ShapeGX0) {
364  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(Fv + Flm) *
365  lsm[i].G().dX0(t[ip], t0);
366  } else if (deriv == Jacobian::Line::ShapeYX0) {
367  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(Fv + Flm) *
368  lsm[i].Y().dX0(t[ip], t0);
369  } else if (deriv == Jacobian::Line::ShapeGX1) {
370  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(Fv - Flm) *
371  lsm[i].G().dX1(t[ip], t0);
372  } else if (deriv == Jacobian::Line::ShapeYX1) {
373  dxsec[iq](j, ip) += ST * pow2(f[j]) * std::real(Fv + Flm) *
