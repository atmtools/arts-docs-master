77 ostream& operator<<(ostream& os, const TokVal& a) {
78  // This is needed for nice formating:
79  bool first = true;
80 
81  switch (a.mtype) {
82  case String_t:
83  os << "\"" << a.ms << "\"";
84  break;
85  case Index_t:
86  os << a.mn;
87  break;
88  case Numeric_t:
89  os << a.mx;
90  break;
91  case Array_String_t:
92  os << "[";
93  for (Index i = 0; i < a.msv.nelem(); ++i) {
94  if (first)
95  first = false;
96  else
97  os << ",";
98  os << "\"" << a.msv[i] << "\"";
99  }
100  os << "]";
101  break;
102  case Array_Index_t:
103  os << "[";
104  for (Index i = 0; i < a.mnv.nelem(); ++i) {
105  if (first)
106  first = false;
107  else
108  os << ",";
109 
110  os << a.mnv[i];
111  }
112  os << "]";
113  break;
114  case Vector_t:
115  os << "[";
116  for (Index i = 0; i < a.mxv.nelem(); ++i) {
117  if (first)
118  first = false;
119  else
120  os << ",";
121 
122  os << a.mxv[i];
123  }
124  os << "]";
125  break;
126  case Matrix_t:
127  os << "[";
128  for (Index i = 0; i < a.mm.nrows(); ++i) {
129  for (Index j = 0; i < a.mm.ncols(); ++j) {
130  if (first)
131  first = false;
132  else {
133  if (j == 0)
134  os << ";";
135  else
136  os << ",";
137  }
138 
139  os << a.mm(i, j);
140  }
141  }
142  os << "]";
143  break;
144  default:
145  cerr << "Undefined token type.\n";
146  arts_exit();
147  }
148  return os;
149 }
