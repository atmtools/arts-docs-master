401 void ReadHITRAN(ArrayOfAbsorptionLines& abs_lines,
402  const String& hitran_file,

416 {
417  abs_lines.resize(0);
418 
419  // Global numbers
420  const std::vector<QuantumNumberType> global_nums = string2vecqn(globalquantumnumbers);
421 
422  // Local numbers
423  const std::vector<QuantumNumberType> local_nums = string2vecqn(localquantumnumbers);
424 
425  // HITRAN type
426  const Options::HitranType hitran_version = Options::toHitranTypeOrThrow(hitran_type);
427 
428  // Hitran data
429  ifstream is;
430  open_input_file(is, hitran_file);
431 
432  bool go_on = true;
433  while (go_on) {
434  Absorption::SingleLineExternal sline;
435  switch (hitran_version) {
436  case Options::HitranType::Post2012:
437  sline = Absorption::ReadFromHitran2012Stream(is);
438  break;
439  case Options::HitranType::From2004To2012:
440  sline = Absorption::ReadFromHitran2004Stream(is);
441  break;
442  case Options::HitranType::Pre2004:
443  sline = Absorption::ReadFromHitran2001Stream(is);
444  break;
445  case Options::HitranType::Online:
446  sline = Absorption::ReadFromHitranOnlineStream(is);
447  break;
448  default:
449  ARTS_ASSERT(false, "The HitranType enum class has to be fully updated!\n");
450  }
451 
452  if (sline.bad) {
453  if (is.eof()) {
454  break;
455  } else {
456  ARTS_USER_ERROR("Cannot read line ", nelem(abs_lines) + 1);
457  }
458  } else if (sline.line.F0() < fmin) {
459  continue; // Skip this line
460  } else if (sline.line.F0() > fmax) {
461  break; // We assume sorted so quit here
462  }
463 
464  // Set Zeeman if implemented
465  sline.line.Zeeman() = Zeeman::GetAdvancedModel(sline.quantumidentity);
466 
467  // Get the global quantum number identifier
468  QuantumIdentifier global_qid(sline.quantumidentity.Isotopologue(), Quantum::IdentifierType::Transition);
469  for(auto qn: global_nums) {
470  global_qid.Lower()[qn] = sline.quantumidentity.Lower()[qn];
471  global_qid.Upper()[qn] = sline.quantumidentity.Upper()[qn];
472  }
473

475  sline.line.LowerQuantumNumbers().reserve(local_nums.size());
476  sline.line.UpperQuantumNumbers().reserve(local_nums.size());
477  for(auto qn: local_nums) {
478  sline.line.LowerQuantumNumbers().emplace_back(sline.quantumidentity.Lower()[qn]);
479  sline.line.UpperQuantumNumbers().emplace_back(sline.quantumidentity.Upper()[qn]);
480  }
481 
482  // Either find a line like this in the list of lines or start a new Lines
483  auto band = std::find_if(abs_lines.begin(), abs_lines.end(), [&](const auto& li){return li.MatchWithExternal(sline, global_qid);});
484  if (band not_eq abs_lines.end()) {
485  band -> AppendSingleLine(sline.line);
486  } else {
487  abs_lines.emplace_back(sline.selfbroadening, sline.bathbroadening, sline.cutoff,
488  sline.mirroring, sline.population, sline.normalization,
489  sline.lineshapetype, sline.T0, sline.cutofffreq,
490  sline.linemixinglimit, global_qid, local_nums, sline.species,
491  std::vector<AbsorptionSingleLine>{sline.line});
492  }
493  }
494 
495  abs_linesSetNormalization(abs_lines, normalization_option, verbosity);
496  abs_linesSetMirroring(abs_lines, mirroring_option, verbosity);
497  abs_linesSetPopulation(abs_lines, population_option, verbosity);

499  abs_linesSetCutoff(abs_lines, cutoff_option, cutoff_value, verbosity);
