89 Index Absorption::Lines::LineShapePos(const Species::Species spec) const noexcept {
90  // Is always first if this is self and self broadening exists
91  if(mselfbroadening and spec == mquantumidentity.Species()) {
92  return 0;
93  }
94 
95  // First and last might be artificial so they should not be checked
96  const Index s = mselfbroadening;
97  const Index e = mbroadeningspecies.nelem() - mbathbroadening;
98  for(Index i=s; i<e; i++) {
99  if(spec == mbroadeningspecies[i]) {
100  return i;
101  }
102  }
103

105  if(mbathbroadening)
106  return mbroadeningspecies.nelem()-1;
107  else

120 Numeric Absorption::Lines::ShapeParameter_dInternal(size_t k, Numeric T, Numeric P, const Vector& vmrs, const RetrievalQuantity& derivative) const noexcept {
121  const auto self = derivative.Mode() == LineShape::self_broadening;
122  const auto bath = derivative.Mode() == LineShape::bath_broadening;
123  const auto& ls = mlines[k].LineShape();
124 
125  if(derivative.QuantumIdentity().Species() not_eq Species() or
126  derivative.QuantumIdentity().Isotopologue() not_eq Isotopologue())
127  return 0;
128  else if(self and mselfbroadening)
129  return ls.GetInternalDeriv(
130  T, mT0, P, 0, vmrs, derivative.LineType());
131  else if(self)
132  return ls.GetInternalDeriv(
133  T, mT0, P, LineShapePos(derivative.QuantumIdentity().Species()), vmrs, derivative.LineType());
134  else if(bath and mbathbroadening)
135  return ls.GetInternalDeriv(
136  T, mT0, P, ls.nelem() - 1, vmrs, derivative.LineType());
137  else if(bath)
138  return 0;
139  else
140  return ls.GetInternalDeriv(
141  T, mT0, P, LineShapePos(derivative.QuantumIdentity().Species()), vmrs, derivative.LineType());

144 Absorption::SingleLineExternal Absorption::ReadFromArtscat3Stream(istream& is) {
145  // Default data and values for this type
146  SingleLineExternal data;
147  data.selfbroadening = true;
148  data.bathbroadening = true;
149  data.lineshapetype = LineShape::Type::VP;
150  data.species.resize(2);
151 
152  // This always contains the rest of the line to parse. At the
153  // beginning the entire line. Line gets shorter and shorter as we
154  // continue to extract stuff from the beginning.
155  String line;
156 
157  // Look for more comments?
158  bool comment = true;
159 
160  while (comment) {
161  // Return true if eof is reached:
162  if (is.eof()) return data;
163 
164  // Throw runtime_error if stream is bad:
165  ARTS_USER_ERROR_IF (!is, "Stream bad.");
166 
167  // Read line from file into linebuffer:
168  getline(is, line);
169 
170  // It is possible that we were exactly at the end of the file before
171  // calling getline. In that case the previous eof() was still false
172  // because eof() evaluates only to true if one tries to read after the
173  // end of the file. The following check catches this.
174  if (line.nelem() == 0 && is.eof()) return data;
175 
176  // @ as first character marks catalogue entry
177  char c;
178  extract(c, line, 1);
179 
180  // check for empty line
181  if (c == '@') {
182  comment = false;
183  }
184  }
185 
186  // read the arts identifier String
187  istringstream icecream(line);
188 
189  String artsid;
190  icecream >> artsid;
191 
192  if (artsid.length() != 0) {
193  // Set the species
194  const auto isotopologue = Species::Tag(Species::update_isot_name(artsid));
195  ARTS_USER_ERROR_IF (isotopologue.is_joker() or isotopologue.type not_eq Species::TagType::Plain,
196  "A line catalog species can only be of the form \"Plain\", meaning it\nhas the form SPECIES-ISONUM.\n"
197  "Your input contains: ", artsid, ". which we cannot interpret as a plain species")
198  data.quantumidentity = Quantum::Identifier(isotopologue.Isotopologue(), Quantum::IdentifierType::Transition);
199 
200  // Extract center frequency:
201  icecream >> double_imanip() >> data.line.F0();
202 
203  Numeric psf;
204  // Extract pressure shift:
205  icecream >> double_imanip() >> psf;
206 
207  // Extract intensity:
208  icecream >> double_imanip() >> data.line.I0();
209 
210  // Extract reference temperature for Intensity in K:
211  icecream >> double_imanip() >> data.T0;
212 
213  // Extract lower state energy:
214  icecream >> data.line.E0();
215 
216  // Extract air broadening parameters:
217  Numeric agam, sgam;
218  icecream >> double_imanip() >> agam;
219  icecream >> double_imanip() >> sgam;
220 
221  // Extract temperature coefficient of broadening parameters:
222  Numeric nair, nself;
223  icecream >> double_imanip() >> nair;
224  icecream >> double_imanip() >> nself;
225 
226  // Extract reference temperature for broadening parameter in K:
227  Numeric tgam;
228  icecream >> double_imanip() >> tgam;
229 
230  // Extract the aux parameters:
231  Index naux;
232  icecream >> naux;
233 
234  // resize the aux array and read it
235  ArrayOfNumeric maux;
236  maux.resize(naux);
237 
238  for (Index j = 0; j < naux; j++) {
239  icecream >> maux[j];
240  //cout << "maux" << j << " = " << maux[j] << "\n";
241  }
242 
243  // Extract accuracies:
244  try {
245  Numeric unused_numeric;
246  icecream >> double_imanip() >> unused_numeric /*mdf*/;
247  icecream >> double_imanip() >> unused_numeric /*mdi0*/;
248  icecream >> double_imanip() >> unused_numeric /*dagam*/;
249  icecream >> double_imanip() >> unused_numeric /*dsgam*/;
250  icecream >> double_imanip() >> unused_numeric /*dnair*/;
251  icecream >> double_imanip() >> unused_numeric /*dnself*/;
252  icecream >> double_imanip() >> unused_numeric /*dpsf*/;
253  } catch (const std::runtime_error&) {
254  }
255 
256  // Fix if tgam is different from ti0
257  if (tgam != data.T0) {
258  agam = agam * pow(tgam / data.T0, nair);
259  sgam = sgam * pow(tgam / data.T0, nself);
260  psf = psf * pow(tgam / data.T0, (Numeric).25 + (Numeric)1.5 * nair);
261  }
262 
263  // Set line shape computer
264  data.line.LineShape() = LineShape::Model(sgam, nself, agam, nair, psf);
265  }
266 
267  // That's it!
268  data.bad = false;
269  return data;
270 }
