59 void GriddedField::copy_grids(const GriddedField& gf) {
60  ARTS_ASSERT(gf.get_dim() == dim);
61 
62  for (Index i = 0; i < dim; i++) {
63  switch (gf.get_grid_type(i)) {
64  case GRID_TYPE_NUMERIC:
65  mgridtypes[i] = GRID_TYPE_NUMERIC;
66  mnumericgrids[i] = gf.get_numeric_grid(i);
67  mstringgrids[i].resize(0);
68  break;
69  case GRID_TYPE_STRING:
70  mgridtypes[i] = GRID_TYPE_STRING;
71  mstringgrids[i] = gf.get_string_grid(i);
72  mnumericgrids[i].resize(0);
73  break;
74  }
75  }
76 }
