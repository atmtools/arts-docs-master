82  Type& mtype,

84  bool& bath,
85  Model& m,
86  ArrayOfSpecies& species,
87  const QuantumIdentifier& qid) {
88  // Set or reset variables
89  mtype = Type::VP;
90  self = true;
91  bath = false;
92  m.mdata = std::vector<SingleSpeciesModel>(7);
93  species = ArrayOfSpecies(7);
94 
95  // Set species
96  species[1] = Species::fromShortName("N2");
97  species[2] = Species::fromShortName("O2");
98  species[3] = Species::fromShortName("H2O");
99  species[4] = Species::fromShortName("CO2");
100  species[5] = Species::fromShortName("H2");
101  species[6] = Species::fromShortName("He");
102 
103  // Temperature types
104  for (auto& v : m.mdata) {
105  v.G0().type = TemperatureModel::T1;
106  v.D0().type = TemperatureModel::T5;
107  }
108 
109  // G0 main coefficient
110  for (auto& v : m.mdata) is >> v.G0().X0;
111 
112  // G0 exponent is same as D0 exponent
113  for (auto& v : m.mdata) {
114  is >> v.G0().X1;
115  v.D0().X1 = v.G0().X1;
116  }
117 
118  // D0 coefficient
119  m.mdata.front().D0().X0 = 0;
120  for (int k = 1; k < 7; k++)
121  is >> m.mdata[k].D0().X0;
122 
123  // Special case when self is part of this list, it needs to be removed
124  for (int k = 1; k < 7; k++) {
125  if (qid.Species() == species[k]) {
126  ARTS_USER_ERROR_IF(m.mdata.front().G0().X0 not_eq m.mdata[k].G0().X0 or
127  m.mdata.front().G0().X1 not_eq m.mdata[k].G0().X1 or
128  m.mdata.front().D0().X1 not_eq m.mdata[k].D0().X1,
129  "Species is ", qid.Isotopologue(), " and this is a broadening species in ARTSCAT-4.\n"
130  "Despite this, values representing self and ", qid.Isotopologue(), " does not match "
131  "in input string\n")
132  m.mdata.front().D0().X0 = m.mdata[k].D0().X0;
133  m.Remove(k, species);
134  break;
135  }
136  }
137 
138  return is;
139 }

141 std::istream& LineShape::from_linefunctiondata(std::istream& data,
142  Type& mtype,
143  bool& self,
144  bool& bath,
145  Model& m,
146  ArrayOfSpecies& species) {
147  self = bath = false;
148  Index specs;
149  String s;
150 
151  // The first tag should give the line shape scheme
152  data >> mtype;
153  check_enum_error(mtype, "Bad Data");
154 
155  // Order of elements for line shape
156  const auto shapeparams =
157  LegacyLineFunctionData::lineshapetag2variablesvector(toString(mtype));
158 
159  // The second tag should give the line mixing scheme
160  data >> s;
161 
162  // Order of elements for line mixing
163  const auto mixingparams =
164  LegacyLineFunctionData::linemixingtag2variablesvector(s);
165 
166  // The third tag should contain the number of species
167  data >> specs;
168  species.resize(specs);
169  m.mdata.resize(specs);
170 
171  ARTS_USER_ERROR_IF (not specs and mtype not_eq Type::DP,
172  "Need at least one species for non-Doppler line shapes");
173 
174  // For all species, we need to set the methods to compute them
175  for (Index i = 0; i < specs; i++) {
176  // This should be a species tag or one of the specials, SELF or BATH
177  data >> s;
178  if (s == self_broadening) {
179  // If the species is self, then we need to flag this
180  self = true;
181  ARTS_USER_ERROR_IF (i not_eq 0,
182  "Self broadening must be first, it is not\n");
183  } else if (s == bath_broadening) {
184  // If the species is air, then we need to flag this
185  bath = true;
186  species[i] = Species::Species::Bath;
187  ARTS_USER_ERROR_IF (i not_eq specs - 1,
188  "Air/bath broadening must be last, it is not\n");
189  } else {
190  // Otherwise, we hope we find a species
191  species[i] = Species::fromShortName(s);
192  ARTS_USER_ERROR_IF (not good_enum(species[i]),
193  "Encountered ", s, " in a position where a species should have been "
194  "defined.\nPlease check your pressure broadening data structure and ensure "
195  "that it follows the correct conventions.\n")
196  }
197 
198  // For all parameters
199  for (auto& params : {shapeparams, mixingparams}) {
200  for (auto& param : params) {
201  data >> s; // Should contain a temperature tag
202 
203  const auto type = toTemperatureModel(s);
204  const Index ntemp =
205  LegacyLineFunctionData::temperaturemodel2legacynelem(type);
206 
207  m.mdata[i].Data()[Index(param)].type = type;
208  if (ntemp <= ModelParameters::N) {
209  switch (ntemp) {
210  case 1:
211  data >> m.mdata[i].Data()[Index(param)].X0;
212  break;
213  case 2:
214  data >> m.mdata[i].Data()[Index(param)].X0 >>
215  m.mdata[i].Data()[Index(param)].X1;
216  break;
217  case 3:
218  data >> m.mdata[i].Data()[Index(param)].X0 >>
219  m.mdata[i].Data()[Index(param)].X1 >>
220  m.mdata[i].Data()[Index(param)].X2;
221  break;
222  case 0:
223  break;
224  default:
225  ARTS_USER_ERROR (
226  "Unknown number of input parameters in Legacy mode.");
227  }
228  } else { // Has to be the only allowed interpolation case
229  ARTS_USER_ERROR_IF (ntemp > 12,
230  "Too many input parameters in interpolation results Legacy mode.");
231  Numeric temp;
232  data >> temp; // should be 200
233  data >> temp; // should be 250
234  data >> temp; // should be 296
235  data >> temp; // should be 340
236  data >> m.mdata[i].Y().X0 >> m.mdata[i].Y().X1 >> m.mdata[i].Y().X2 >> m.mdata[i].Y().X3
237  >> m.mdata[i].G().X0 >> m.mdata[i].G().X1 >> m.mdata[i].G().X2 >> m.mdata[i].G().X3;
238  }
239  }
240  }
241  }
242 
243  return data;
244 }

459

461  const Index back = n - 1; // Last index
462 
463  // Initialize list of VMRS to 0
464  Vector line_vmrs(n, 0);
465 
466  // We need to know if bath is an actual species
467  const bool bath = lineshape_species.back() == Species::Species::Bath;
468 
469  // Loop species
470  for (Index i = 0; i < n - bath; i++) {
471  const Species::Species target = lineshape_species[i];
472 
473  // Find species in list or do nothing at all
474  Index this_species_index = -1;
475  for (Index j = 0; j < atmospheric_species.nelem(); j++) {
476  if (atmospheric_species[j].Species() == target) {
477  this_species_index = j;
478  }
479  }
480 
481  // Set to non-zero in-case species exists
482  if (this_species_index not_eq -1) {
483  line_vmrs[i] = atmospheric_vmrs[this_species_index];
484  }
485  }
486 
487  // Renormalize, if bath-species exist this is automatic.
488  if (bath) {
489  line_vmrs[back] = 1.0 - line_vmrs.sum();
490  } else if(line_vmrs.sum() == 0) { // Special case
491  } else {
492  line_vmrs /= line_vmrs.sum();
493  }
494 
495  return line_vmrs;
496 }

583  return LineShape::Model();

586

591  Index i=-100000;
592

596  if(std::any_of(names.cbegin(), names.cend(),
597  [part](auto x){return part == x;})) {
598  i=-1;

603  tm = toTemperatureModel(part);
604  }
