87 constexpr std::string_view normalizationtype2metadatastring(NormalizationType in) {

89  case NormalizationType::None:
90  return "No re-normalization in the far wing will be applied.\n";
91  case NormalizationType::VVH:
92  return "van Vleck and Huber far-wing renormalization will be applied, "
93  "i.e. F ~ (f tanh(hf/2kT))/(f0 tanh(hf0/2kT))\n";
94  case NormalizationType::VVW:
95  return "van Vleck and Weisskopf far-wing renormalization will be applied, "
96  "i.e. F ~ (f/f0)^2\n";
97  case NormalizationType::RQ:
98  return "Rosenkranz quadratic far-wing renormalization will be applied, "
99  "i.e. F ~ hf0/2kT sinh(hf0/2kT) (f/f0)^2\n";

109 ENUMCLASS(PopulationType, char,
110  LTE, // Assume band is in LTE
111  NLTE, // Assume band is in NLTE and the upper-to-lower ratio is known
112  VibTemps, // Assume band is in NLTE described by vibrational temperatures and LTE at other levels
113  ByHITRANRosenkranzRelmat, // Assume band needs to compute relaxation matrix to derive HITRAN Y-coefficients
