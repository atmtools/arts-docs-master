44 void abs_xsec_agenda_checkedCalc(Workspace& ws _U_,
45  Index& abs_xsec_agenda_checked,
46  // WS Input:
47  const ArrayOfArrayOfSpeciesTag& abs_species,
48  const Agenda& abs_xsec_agenda,
49  const Verbosity&) {
50  bool needs_continua = false;
51  bool needs_cia = false;
52  bool needs_hxsec = false;
53 
54  for (Index sp = 0; sp < abs_species.nelem(); sp++) {
55  for (Index tgs = 0; tgs < abs_species[sp].nelem(); tgs++) {
56  switch (abs_species[sp][tgs].Type()) {
57  case Species::TagType::Plain:
58  break;
59  case Species::TagType::Zeeman:
60  break;
61  case Species::TagType::Predefined:
62  needs_continua = true;
63  break;
64  case Species::TagType::Cia:
65  needs_cia = true;
66  break;
67  case Species::TagType::FreeElectrons:
68  break;
69  case Species::TagType::Particles:
70  break;
71  case Species::TagType::HitranXsec:
72  needs_hxsec = true;
73  break;
74  default:
75  ARTS_USER_ERROR ("Unknown species type: ", abs_species[sp][tgs].Type())
76  break;
77  }
78  }
79  }
80 
81  ARTS_USER_ERROR_IF (needs_continua &&
82  !abs_xsec_agenda.has_method("abs_xsec_per_speciesAddConts"),
83  "*abs_species* contains continuum species but *abs_xsec_agenda*\n"
84  "does not contain *abs_xsec_per_speciesAddConts*.");
85 
86  ARTS_USER_ERROR_IF (needs_cia && !abs_xsec_agenda.has_method("abs_xsec_per_speciesAddCIA"),
87  "*abs_species* contains CIA species but *abs_xsec_agenda*\n"
88  "does not contain *abs_xsec_per_speciesAddCIA*.");
89 
90  ARTS_USER_ERROR_IF (needs_hxsec &&
91  !abs_xsec_agenda.has_method("abs_xsec_per_speciesAddHitranXsec"),
92  "*abs_species* contains HITRAN xsec species but *abs_xsec_agenda*\n"
93  "does not contain *abs_xsec_per_speciesAddHitranXsec*.");
94 
95  if (find_first_species_tag(abs_species, SpeciesTag("O2-MPM2020")) >= 0) {
96  ARTS_USER_ERROR_IF (not abs_xsec_agenda.has_method("abs_xsec_per_speciesAddPredefinedO2MPM2020"),
97  "*abs_species contains \"O2-MPM2020\" but *abs_xsec_agenda*\n"
98  "does not contain *abs_xsec_per_speciesAddPredefinedO2MPM2020*.");
99  }
100 
101  // If here, all OK
102  abs_xsec_agenda_checked = 1;
103 }

105 /* Workspace method: Doxygen documentation will be auto-generated */
106 void atmfields_checkedCalc(Index& atmfields_checked,
107  const Index& atmosphere_dim,
108  const Vector& p_grid,
109  const Vector& lat_grid,
110  const Vector& lon_grid,
111  const ArrayOfArrayOfSpeciesTag& abs_species,
112  const Tensor3& t_field,
113  const Tensor4& vmr_field,
114  const Tensor3& wind_u_field,
115  const Tensor3& wind_v_field,
116  const Tensor3& wind_w_field,
117  const Tensor3& mag_u_field,
118  const Tensor3& mag_v_field,
119  const Tensor3& mag_w_field,
120  const Index& abs_f_interp_order,
121  const Index& negative_vmr_ok,
122  const Verbosity&) {
123  // Consistency between dim, grids and atmospheric fields/surfaces
124  chk_if_in_range("atmosphere_dim", atmosphere_dim, 1, 3);
125  chk_atm_grids(atmosphere_dim, p_grid, lat_grid, lon_grid);
126  chk_atm_field("t_field", t_field, atmosphere_dim, p_grid, lat_grid, lon_grid);
127  chk_atm_field("vmr_field",
128  vmr_field,
129  atmosphere_dim,
130  abs_species.nelem(),
131  p_grid,
132  lat_grid,
133  lon_grid);
134 
135  // More for vmr_field.
136  ARTS_USER_ERROR_IF (!negative_vmr_ok && abs_species.nelem() && min(vmr_field) < 0,
137  "All values in *vmr_field* must be >= 0.");
138 
139  // More for t_field.
140  ARTS_USER_ERROR_IF (min(t_field) <= 0,
141  "All temperatures in *t_field* must be > 0.");
142 
143  // Winds
144  if (wind_w_field.npages() > 0) {
145  chk_atm_field("wind_w_field",
146  wind_w_field,
147  atmosphere_dim,
148  p_grid,
149  lat_grid,
150  lon_grid);
151  }
152  if (atmosphere_dim < 3 && wind_v_field.npages() > 0) {
153  chk_atm_field("wind_v_field",
154  wind_v_field,
155  atmosphere_dim,
156  p_grid,
157  lat_grid,
158  lon_grid);
159  }
160  if (atmosphere_dim > 2) {
161  if (wind_u_field.npages() > 0) {
162  if (wind_v_field.npages() > 0) {
163  bool chk_poles = false;
164  chk_atm_field("wind_u_field",
165  wind_u_field,
166  atmosphere_dim,
167  p_grid,
168  lat_grid,
169  lon_grid,
170  chk_poles);
171  chk_atm_field("wind_v_field",
172  wind_v_field,
173  atmosphere_dim,
174  p_grid,
175  lat_grid,
176  lon_grid,
177  chk_poles);
178  chk_atm_vecfield_lat90("wind_v_field",
179  wind_v_field,
180  "wind_u_field",
181  wind_u_field,
182  atmosphere_dim,
183  lat_grid);
184  } else {
185  chk_atm_field("wind_u_field",
186  wind_u_field,
187  atmosphere_dim,
188  p_grid,
189  lat_grid,
190  lon_grid);
191  }
192  } else {
193  if (wind_v_field.npages() > 0) {
194  chk_atm_field("wind_v_field",
195  wind_v_field,
196  atmosphere_dim,
197  p_grid,
198  lat_grid,
199  lon_grid);
200  }
201  }
202  }
203 
204  // If any of the wind fields exist, abs_f_interp_order must not be zero.
205  if (wind_u_field.npages() > 0 || wind_v_field.npages() > 0 ||
206  wind_w_field.npages() > 0) {
207  ARTS_USER_ERROR_IF (abs_f_interp_order == 0,
208  "You have a wind field set, but abs_f_interp_order zero.\n"
209  "This is not allowed. Though abs_f_interp_order only is\n"
210  "required and has an effect if absorption lookup tables\n"
211  "are used, for safety reasons you also have to set it >0\n"
212  "in case of on-the-fly absorption.")
213  }
214 
215  // Magnetic field
216  if (mag_w_field.npages() > 0) {
217  chk_atm_field("mag_w_field (vertical magfield component)",
218  mag_w_field,
219  atmosphere_dim,
220  p_grid,
221  lat_grid,
222  lon_grid);
223  }
224  if (mag_u_field.npages() > 0) {
225  if (mag_v_field.npages() > 0) {
226  bool chk_poles = false;
227  chk_atm_field("mag_v_field",
228  mag_v_field,
229  atmosphere_dim,
230  p_grid,
231  lat_grid,
232  lon_grid,
233  chk_poles);
234  chk_atm_field("mag_u_field",
235  mag_u_field,
236  atmosphere_dim,
237  p_grid,
238  lat_grid,
239  lon_grid,
240  chk_poles);
241  chk_atm_vecfield_lat90("mag_v_field",
242  mag_v_field,
243  "mag_u_field",
244  mag_u_field,
245  atmosphere_dim,
246  lat_grid);
247  } else {
248  chk_atm_field("mag_u_field",
249  mag_u_field,
250  atmosphere_dim,
251  p_grid,
252  lat_grid,
253  lon_grid);
254  }
255  } else {
256  if (mag_v_field.npages() > 0) {
257  chk_atm_field("mag_v_field",
258  mag_v_field,
259  atmosphere_dim,
260  p_grid,
261  lat_grid,
262  lon_grid);
263  }
264  }
265 
266  // If here, all OK
267  atmfields_checked = 1;
268 }

270 /* Workspace method: Doxygen documentation will be auto-generated */
271 void atmgeom_checkedCalc(Index& atmgeom_checked,
272  const Index& atmosphere_dim,
273  const Vector& p_grid,
274  const Vector& lat_grid,
275  const Vector& lon_grid,
276  const Tensor3& z_field,
277  const Vector& refellipsoid,
278  const Matrix& z_surface,
279  const Vector& lat_true,
280  const Vector& lon_true,
281  const Numeric& max500hpa_gradient,
282  const Verbosity&) {
283  // A repetition from atmfields_checked, but we do this to make the two parts
284  // independent (the other option would be to demand atmfields_checkec == 1)
285  chk_if_in_range("atmosphere_dim", atmosphere_dim, 1, 3);
286  chk_atm_grids(atmosphere_dim, p_grid, lat_grid, lon_grid);
287 
288  // *refellipsoid*
289  ARTS_USER_ERROR_IF (refellipsoid.nelem() != 2,
290  "The WSV *refellispoid* must be a vector of "
291  "length 2*.");
292  ARTS_USER_ERROR_IF (refellipsoid[0] <= 0,
293  "The first element of *refellipsoid* must "
294  "be > 0.");
295  ARTS_USER_ERROR_IF (refellipsoid[1] < 0 || refellipsoid[1] > 1,
296  "The second element of *refellipsoid* must be "
297  "inside [0,1].");
298  ARTS_USER_ERROR_IF (atmosphere_dim == 1 && refellipsoid[1] != 0,
299  "For 1D, the second element of *refellipsoid* "
300  "(the eccentricity) must be 0.");
301 
302  chk_atm_field("z_field", z_field, atmosphere_dim, p_grid, lat_grid, lon_grid);
303  chk_atm_surface("z_surface", z_surface, atmosphere_dim, lat_grid, lon_grid);
304 
305  // Check that z_field has strictly increasing pages.
306  for (Index row = 0; row < z_field.nrows(); row++) {
307  for (Index col = 0; col < z_field.ncols(); col++) {
308  ostringstream os;
309  os << "z_field (for latitude nr " << row << " and longitude nr " << col
310  << ")";
311  chk_if_increasing(os.str(), z_field(joker, row, col));
312  }
313  }
314 
315  // Check that there is no gap between the surface and lowest pressure
316  // level
317  // (A copy of this code piece is found in z_fieldFromHSE. Make this to an
318  // internal function if used in more places.)
319  for (Index row = 0; row < z_surface.nrows(); row++) {
320  for (Index col = 0; col < z_surface.ncols(); col++) {
321  ARTS_USER_ERROR_IF (z_surface(row, col) < z_field(0, row, col) ||
322  z_surface(row, col) >= z_field(z_field.npages() - 1, row, col),
323  "The surface altitude (*z_surface*) cannot be outside\n"
324  "of the altitudes in *z_field*.\n"
325  "z_surface: ", z_surface(row, col), "\n"
326  "min of z_field: ", z_field(0, row, col), "\n"
327  "max of z_field: ", z_field(z_field.npages() - 1, row, col),
328  "\n",
329  (atmosphere_dim > 1) ? var_string("\nThis was found to be the case for:\n", "latitude ", lat_grid[row]) : var_string(),
330  (atmosphere_dim > 2) ? var_string("\nlongitude ", lon_grid[col]) : var_string())
331  }
332  }
333 
334  // A rough check of gradients in the 500 hPa level (or closest pressure level)
335  // This check is just run in the latitude direction, along some longitudes to
336  // save time
337  if (atmosphere_dim > 1) {
338  // Find 500 hPa
339  Index ip = -1; Numeric dpmin = 99e99;
340  for (Index i=0; i<p_grid.nelem(); i++) {
341  const Numeric dp = abs(p_grid[i] - 500e2);
342  if (dp < dpmin) {
343  ip = i;
344  dpmin = dp;
345  }
346  }
347  Numeric maxgrad = -1;
348  for (Index ilon=0; ilon<max(1,lon_grid.nelem()); ilon += 10) {
349  for (Index ilat=1; ilat<lat_grid.nelem(); ilat++) {
350  const Numeric grad = abs((z_field(ip,ilat,ilon)-z_field(ip,ilat-1,ilon)) /
351  (lat_grid[ilat]-lat_grid[ilat-1]));
352  if (grad > maxgrad)
353  maxgrad = grad;
354  }
355  }
356  // Scale to change over 100 km
357  maxgrad *= 100.0/111.0;
358  if (maxgrad > max500hpa_gradient) {
359  ostringstream os;
360  os << "A check of the altitude of the " << p_grid[ip]/100
361  << " hPa level has been made.\nThe maximum gradient found matches "
362  << maxgrad << " m/100km, that exceeds\nthe set limit of "
363  << max500hpa_gradient << " m/100km (by GIN *max500hpa_gradient*).\n"
364  << "Please check the smoothness of *z_field*.";
365  throw runtime_error(os.str());
366  }
367  }
368 
369  // lat/lon true
370  if (atmosphere_dim < 3 && (lat_true.nelem() || lon_true.nelem())) {
371  if (atmosphere_dim == 1) {
372  ARTS_USER_ERROR_IF (lat_true.nelem() != 1,
373  "For 1D, *lat_true* must have length 1.");
374  ARTS_USER_ERROR_IF (lon_true.nelem() != 1,
375  "For 1D, *lon_true* must have length 1.");
376  } else if (atmosphere_dim == 2) {
377  ARTS_USER_ERROR_IF (lat_true.nelem() != lat_grid.nelem(),
378  "For 2D, *lat_true* must have same length as *lat_grid*.");
379  ARTS_USER_ERROR_IF (lon_true.nelem() != lat_grid.nelem(),
380  "For 2D, *lon_true* must have same length as *lat_grid*.");
381  }
382  ARTS_USER_ERROR_IF (lon_true.nelem() != lat_true.nelem(),
383  "If *lat_true* is set, also *lon_true* must be "
384  "set (and have the same length).");
385  ARTS_USER_ERROR_IF (min(lat_true) < -90 || max(lat_true) > 90,
386  "Values in *lat_true* must be inside [-90,90].");
387  ARTS_USER_ERROR_IF (min(lon_true) < -180 || max(lon_true) > 360,
388  "Values in *lon_true* must be inside [-180,360].");
389  }
390 
391  // If here, all OK
392  atmgeom_checked = 1;
393 }

395 /* Workspace method: Doxygen documentation will be auto-generated */
396 void cloudbox_checkedCalc(Index& cloudbox_checked,
397  const Index& atmfields_checked,
398  const Index& atmosphere_dim,
399  const Vector& p_grid,
400  const Vector& lat_grid,
401  const Vector& lon_grid,
402  const Tensor3& z_field,
403  const Matrix& z_surface,
404  const Tensor3& wind_u_field,
405  const Tensor3& wind_v_field,
406  const Tensor3& wind_w_field,
407  const Index& cloudbox_on,
408  const ArrayOfIndex& cloudbox_limits,
409  const Tensor4& pnd_field,
410  const ArrayOfTensor4& dpnd_field_dx,
411  const ArrayOfRetrievalQuantity& jacobian_quantities,
412  const ArrayOfArrayOfSingleScatteringData& scat_data,
413  const ArrayOfString& scat_species,
414  const Matrix& particle_masses,
415  const ArrayOfArrayOfSpeciesTag& abs_species,
416  const Index& negative_pnd_ok,
417  const Verbosity&) {
418  ARTS_USER_ERROR_IF (atmfields_checked != 1,
419  "The atmospheric fields must be flagged to have "
420  "passed a consistency check (atmfields_checked=1).");
421 
422  chk_if_bool("cloudbox_on", cloudbox_on);
423 
424  if (cloudbox_on) {
425  // Winds, must be empty variables (i.e. no winds allowed)
426  ARTS_USER_ERROR_IF (!wind_w_field.empty() ||
427  !wind_v_field.empty() ||
428  !wind_u_field.empty(),
429  "The scattering methods are not (yet?) handling winds. For this\n"
430  "reason, the WSVs for wind fields must all be empty with an\n."
431  "active cloudbox.");
432 
433  // no "particles" in abs_species if cloudbox is on (they act on the same
434  // scat_data! and there is no good reason to have some particles as
435  // abs-only, if we anyway do a scattering calculation.).
436  Index has_absparticles = 0;
437  for (Index sp = 0; sp < abs_species.nelem() && has_absparticles < 1; sp++) {
438  if (abs_species[sp].Particles()) {
439  has_absparticles = 1;
440  }
441  }
442  ARTS_USER_ERROR_IF (has_absparticles,
443  "For scattering calculations (cloudbox is on),"
444  "abs_species is not allowed to contain\n"
445  "'particles' (absorbing-only particles)!");
446 
447  // Cloudbox limits
448  ARTS_USER_ERROR_IF (cloudbox_limits.nelem() != atmosphere_dim * 2,
449  "The array *cloudbox_limits* has incorrect length.\n"
450  "For atmospheric dim. = ", atmosphere_dim,
451  " the length shall be ", atmosphere_dim * 2, " but it is ",
452  cloudbox_limits.nelem(), ".")
453  ARTS_USER_ERROR_IF (cloudbox_limits[1] <= cloudbox_limits[0] || cloudbox_limits[0] < 0 ||
454  cloudbox_limits[1] >= p_grid.nelem(),
455  "Incorrect value(s) for cloud box pressure limit(s) found."
456  "\nValues are either out of range or upper limit is not "
457  "greater than lower limit.\nWith present length of "
458  "*p_grid*, OK values are 0 - ", p_grid.nelem() - 1,
459  ".\nThe pressure index limits are set to ", cloudbox_limits[0],
460  " - ", cloudbox_limits[1], ".")
461 
462  Index nlat = 1, nlon = 1;
463 
464  if (atmosphere_dim > 1) {
465  nlat = lat_grid.nelem();
466  ARTS_USER_ERROR_IF (cloudbox_limits[3] <= cloudbox_limits[2] || cloudbox_limits[2] < 1 ||
467  cloudbox_limits[3] >= nlat - 1,
468  "Incorrect value(s) for cloud box latitude limit(s) found."
469  "\nValues are either out of range or upper limit is not "
470  "greater than lower limit.\nWith present length of "
471  "*lat_grid*, OK values are 1 - ", nlat - 2,
472  ".\nThe latitude index limits are set to ", cloudbox_limits[2],
473  " - ", cloudbox_limits[3], ".")
474  ARTS_USER_ERROR_IF ((lat_grid[cloudbox_limits[2]] - lat_grid[0] < LAT_LON_MIN) &&
475  (atmosphere_dim == 2 || (atmosphere_dim == 3 && lat_grid[0] > -90)),
476  "Too small distance between cloudbox and lower end of "
477  "latitude grid.\n"
478  "This distance must be ", LAT_LON_MIN, " degrees.\n"
479  "Cloudbox ends at ", lat_grid[cloudbox_limits[2]],
480  " and latitude grid starts at ", lat_grid[0], ".")
481  ARTS_USER_ERROR_IF ((lat_grid[nlat - 1] - lat_grid[cloudbox_limits[3]] < LAT_LON_MIN) &&
482  (atmosphere_dim == 2 ||
483  (atmosphere_dim == 3 && lat_grid[nlat - 1] < 90)),
484  "Too small distance between cloudbox and upper end of "
485  "latitude grid.\n"
486  "This distance must be ", LAT_LON_MIN, " degrees.\n"
487  "Cloudbox ends at ", lat_grid[cloudbox_limits[3]],
488  " and latitude grid ends at ", lat_grid[nlat - 1], ".")
489  }
490 
491  if (atmosphere_dim > 2) {
492  nlon = lon_grid.nelem();
493  ARTS_USER_ERROR_IF (cloudbox_limits[5] <= cloudbox_limits[4] || cloudbox_limits[4] < 1 ||
494  cloudbox_limits[5] >= nlon - 1,
495  "Incorrect value(s) for cloud box longitude limit(s) found"
496  ".\nValues are either out of range or upper limit is not "
497  "greater than lower limit.\nWith present length of "
498  "*lon_grid*, OK values are 1 - ", nlon - 2,
499  ".\nThe longitude limits are set to ", cloudbox_limits[4],
500  " - ", cloudbox_limits[5], ".")
501  if (lon_grid[nlon - 1] - lon_grid[0] < 360) {
502  const Numeric latmax = max(abs(lat_grid[cloudbox_limits[2]]),
503  abs(lat_grid[cloudbox_limits[3]]));
504  const Numeric lfac = 1 / cos(DEG2RAD * latmax);
505  ARTS_USER_ERROR_IF (lon_grid[cloudbox_limits[4]] - lon_grid[0] < LAT_LON_MIN / lfac,
506  "Too small distance between cloudbox and lower end of"
507  "the longitude\ngrid. This distance must here be ",
508  LAT_LON_MIN / lfac, " degrees.")
509  ARTS_USER_ERROR_IF (lon_grid[nlon - 1] - lon_grid[cloudbox_limits[5]] <
510  LAT_LON_MIN / lfac,
511  "Too small distance between cloudbox and upper end of"
512  "the longitude\ngrid. This distance must here be ",
513  LAT_LON_MIN / lfac, " degrees.")
514  }
515  }
516 
517  // Check with respect to z_surface
518  for (Index o = 0; o < nlon; o++) {
519  for (Index a = 0; a < nlat; a++) {
520  ARTS_USER_ERROR_IF (z_field(cloudbox_limits[1], a, o) <= z_surface(a, o),
521  "The upper vertical limit of the cloudbox must be above "
522  "the surface altitude (for all latitudes and longitudes).");
523  }
524  }
525 
526  // Check pnd_field
527  //
528  const Index np = TotalNumberOfElements(scat_data);
529  // Dummy variables to mimic grids of correct size
530  Vector g1(cloudbox_limits[1] - cloudbox_limits[0] + 1), g2(0), g3(0);
531  if (atmosphere_dim > 1) {
532  g2.resize(cloudbox_limits[3] - cloudbox_limits[2] + 1);
533  }
534  if (atmosphere_dim > 2) {
535  g3.resize(cloudbox_limits[5] - cloudbox_limits[4] + 1);
536  }
537 
538  chk_atm_field("pnd_field", pnd_field, atmosphere_dim, np, g1, g2, g3);
539 
540  ARTS_USER_ERROR_IF (!negative_pnd_ok && min(pnd_field) < 0,
541  "Negative values in *pnd_field* not allowed.");
542 
543  // No non-zero pnd at lower boundary unless lower boundary is at or below
544  // surface
545  for (Index a = 0; a < g2.nelem(); a++) {
546  for (Index o = 0; o < g3.nelem(); o++) {
547  ARTS_USER_ERROR_IF (max(pnd_field(joker, 0, a, o)) > 0 &&
548  z_field(cloudbox_limits[0], a, o) > z_surface(a, o),
549  "A non-zero value found in *pnd_field* at the"
550  " lower altitude limit of the cloudbox (but the "
551  "position is not at or below the surface altitude).");
552  }
553  }
554 
555  // No non-zero pnd at upper boundary unless upper boundary is top of
556  // atmosphere
