53 ArrayOfVector caha(const ArrayOfVector& data, const Vector& tcvec, const Vector& thvec, const Index first_c_index)
54 {
55  assert(first_c_index >= 0);
56 
57  /* Must compute the positions due to c-Index offset
58  *
59  * The start position should be on C or H
60  * if first_c_index > 1 then start at H two steps early
61  *
62  * The cold position is always a multiple of 4 from
63  * first_c_index. In case first_c_index is > 3, we
64  * must take the modulus of the position
65  *
66  * The hot position is either two indices before or
67  * two indices behind the cold position, whichever
68  * is lowest and positive and below 4 */
69  const Index start = first_c_index - ((first_c_index > 1) ? 2 : 0);
70  const Index cpos = first_c_index % 4;
71  const Index hpos = cpos + ((cpos < 2) ? 2 : -2);
72 
73  // Initialize pointers and values of interest
74  const Vector * pc = nullptr;
75  const Vector * pa = nullptr;
76  const Vector * ph = nullptr;
77  Numeric tc=0, th=0;
78 
79  // Output vector
80  ArrayOfVector out;
81 
82  // For all data entries
83  for (Index i=start; i<data.nelem(); i++) {
84  // Cycle is CAHA → 4 steps
85  const Index pos = i % 4;
86 
87  // Set the values when we are at the correct position
88  if (pos == cpos) {
89  pc = &data[i];
90  tc = tcvec[i];
91  } else if (pos == hpos) {
92  ph = &data[i];
93  th = thvec[i];
94  } else {
95  pa = &data[i];
96  }
97 
98  /* Add new measurement if we are at C or H since we have
99  * then completed a new calibration cycle
100  *
101  * Note that nullptr is false */
102  if ((pos == cpos or pos == hpos) and pc and pa and ph) {
103  out.emplace_back(calibration(*pc, *pa, *ph, tc, th));
104  }
105  }
106 
107  return out;
108 }

112 VectorView avg(VectorView y, const ArrayOfVector& ys, const Index start, const Index end_tmp)
113 {
114  // True end
115  const Index end = end_tmp >= 0 ? end_tmp : 1 + ys.nelem() + end_tmp;

117  // Scale

120  // Reset

122 
123  // Compute

125  for (Index i=0; i<y.nelem(); i++)
126  y[i] += ys[k][i] * scale;

238 Numeric median(const ConstVectorView v, const ArrayOfIndex& pos)
239 {
240  // Size of problem
241  const Index n = pos.nelem() ? pos.nelem() : v.nelem();
242 
243  // Create a vector to sort
244  ArrayOfNumeric calc(n);
245  for (Index i=0; i<n; i++)
246  if (pos.nelem())
247  calc[i] = v[pos[i]];
248  else
249  calc[i] = v[i];
250 
251  // Sort the vector
252  std::sort(calc.begin(), calc.end());
253 
254  // Return the median
255  if (n % 2)
256  return calc[n/2];
257  else
258  return (calc[(n-1)/2] + calc[n/2]) / 2;
259 }

275 
276  // Sort the vector
277  std::sort(calc.begin(), calc.end());
278 
279  // Remove non-normal
280  auto newend = std::remove_if(calc.begin(), calc.end(), [](auto x){return not isnormal_or_zero(x);});

282 
283  // Return the median
284  if (numnormal) {
285  if (numnormal % 2) {
286  return calc[numnormal/2];
287  } else {
288  return (calc[(numnormal-1)/2] + calc[numnormal/2]) / 2;
289  }
290  } else {
291  return std::numeric_limits<Numeric>::quiet_NaN();

297 ArrayOfIndex focus_doublescaling(const Vector& x, const Numeric x0, const Numeric dx)
298 {
299  ArrayOfIndex scale(x.size());
300 
301  // Scaling function, rounds down an Index
302  auto scaling = [](Numeric X, Numeric X0, Numeric DX) {
303  const Index p = 1 + Index(std::abs(X - X0) / DX); // floor + 1 is close enough
304  Index this_scale = 0;
305  while ((p >> this_scale) > 1) this_scale++;
306  return 1 << this_scale;
307  };
308 
309  for (Index i=0; i<x.size(); i++) scale[i] = scaling(x[i], x0, dx);
310  return scale;
311 }

313 std::pair<Index, Index> find_first_and_last_1(const ArrayOfIndex& x)
314 {
315  Index first=x.nelem() - 1;
316  Index last=0;
317  for (Index i=0; i<x.nelem(); i++) {

328 Vector focus(const Vector& x, const ArrayOfIndex& scaling)
329 {
330  const Index N = x.size();
331  Index first_i, last_i;
332 
333  // Find the first and last one in scaling
334  const auto [firstone, lastone] = find_first_and_last_1(scaling);
335 
336  // All the central numbers
337  std::vector<Numeric> out;
338  for (Index i=firstone; i<=lastone; i++) out.emplace_back(x[i]);
339 
340  // All the last numbers appended
341  first_i = lastone + 1;
342  while (first_i not_eq N) {
343  last_i = first_i + scaling[first_i];
344  if (last_i > N) last_i = N;
345  out.emplace_back(mean(x[Range(first_i, last_i-first_i)]));
346  first_i = last_i;
347  }
348 
349  // All the first numbers prepended
350  std::vector<Numeric> fx(0);
351  last_i = firstone;
352  while (last_i > 0) {
353  first_i = last_i - scaling[last_i];
354  if (first_i < 0) first_i = 0;
355  out.insert(out.begin(), mean(x[Range(first_i, last_i-first_i)]));
356  last_i = first_i;
357  }
358 
359  return out;
