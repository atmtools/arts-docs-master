47 double_istream& double_istream::parse_on_fail(double& x, bool neg) {
48  const char* exp[] = {"", "inf", "Inf", "nan", "NaN"};
49  const char* e = exp[0];
50  int l = 0;
51  char inf[4] = "\0\0\0";
52  char* c = inf;
53  if (neg) *c++ = '-';
54  in.clear();
55  if (!(in >> *c).good()) return *this;
56  switch (*c) {
57  case 'i':
58  e = exp[l = 1];
59  break;
60  case 'I':
61  e = exp[l = 2];
62  break;
63  case 'n':
64  e = exp[l = 3];
65  break;
66  case 'N':
67  e = exp[l = 4];
68  break;
69  }
70  while (*c == *e) {
71  if ((e - exp[l]) == 2) break;
72  ++e;
73  if (!(in >> *++c).good()) break;
74  }
75  if (in.good() && *c == *e) {
76  switch (l) {
77  case 1:
78  case 2:
79  x = std::numeric_limits<double>::infinity();
80  break;
81  case 3:
82  case 4:
83  x = std::numeric_limits<double>::quiet_NaN();
84  break;
85  }
86  if (neg) x = -x;
87  return *this;
88  } else if (!in.good()) {
89  if (!in.fail()) return *this;
90  in.clear();
91  --c;
92  }
93  do {
94  in.putback(*c);
95  } while (c-- != inf);
96  in.setstate(std::ios_base::failbit);
97  return *this;
98 }
