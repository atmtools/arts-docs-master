95 constexpr QuantumNumberType string2quantumnumbertype(const std::string_view s) {
96  QuantumNumberType out = toQuantumNumberType(s);
97  if (out == QuantumNumberType::FINAL) {
98  if (s.find("F#") < s.length()) out = QuantumNumberType::F; // HITRAN has many names for F
99  else if (s == "K") out = QuantumNumberType::Ka; // HITRAN name
100  else if (s == "v") out = QuantumNumberType::v1; // HITRAN name
101  else if (s == "l") out = QuantumNumberType::l1; // HITRAN name
102  else if (s == "ElecStateLabel") out = QuantumNumberType::ElectronState; // HITRAN name
103  }
104  return out;
105 }
