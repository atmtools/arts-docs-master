379  template <int N>
380  Eigen::Matrix<Numeric, N, 1> second_order_integration_source(const Eigen::Matrix<Numeric, N, N> T, const Eigen::Matrix<Numeric, N, 1> far, const Eigen::Matrix<Numeric, N, 1> close,
381  const Eigen::Matrix<Numeric, N, N> Kfar, const Eigen::Matrix<Numeric, N, N> Kclose, const Numeric r)
382  const noexcept {
383  static_assert (N>0 and N<5, "Bad size N");
384 
385  const auto I = Eigen::Matrix<Numeric, N, N>::Identity();
386  if (T(0, 0) < 0.99) {
387  const auto od = 0.5 * r * (Kfar + Kclose);
388  const Eigen::Matrix<Numeric, N, 1> second = od.inverse() * ((I - (I + od) * T) * far + (od - I + T) * close);
389  if ((far[0] < close[0] and Kfar(0, 0) > Kclose(0, 0)) or (far[0] > close[0] and Kfar(0, 0) < Kclose(0, 0))) {
390  const Eigen::Matrix<Numeric, N, 1> second_limit = 0.5 * (I - T) * (far + close);
391 
392  // FIXME: This is the equation given in the source material...
393  // const Eigen::Matrix<Numeric, N, 1> second_limit = 0.25 * r * (Kfar * far + Kclose * close);
394 
395  if (second_limit[0] > second[0]) {
396  return second;
397  } else {
398  return second_limit;
399  }
400  } else {
401  return second;
402  }
403  } else {
404  return 0.5 * (I - T) * (far + close);
405  }
406  }

409  template <int N>
410  Eigen::Matrix<Numeric, N, 1> second_order_integration_dsource(size_t i,
411  const TransmissionMatrix& dx,
412  const Eigen::Matrix<Numeric, N, 1> far,
413  const Eigen::Matrix<Numeric, N, 1> close,
414  const Eigen::Matrix<Numeric, N, 1> d,
415  bool isfar) const noexcept {
416  static_assert (N > 0 and N < 5, "Bad size N");
417 
418  const auto I = Eigen::Matrix<Numeric, N, N>::Identity();
419  const auto T = TraMat<N>(i);
420  const auto dTdx = dx.TraMat<N>(i);
421  const Eigen::Matrix<Numeric, N, 1> first = 0.5 * (I - T) * (far + close);
422  if (T(0, 0) < 0.99) {
423  const auto od = OptDepth<N>(i);
424  const auto doddx = dx.OptDepth<N>(i);
425  const Eigen::Matrix<Numeric, N, 1> second = od.inverse() * ((I - (I + od) * T) * far + (od - I + T) * close);
426  if (first[0] > second[0]) {
427  if (isfar) {
428  return od.inverse() * ((I - (I + od) * T) * d
429  - (I + od) * dTdx * far
430  + (doddx + T) * close
431  - doddx * second);
432  }
433  else
434  return od.inverse() * (
435  - (I + od) * dTdx * far
436  + (od - I + T) * d
437  + (doddx + T) * close
438  - doddx * second);
439  } else {
440  return 0.5 * ((I - T) * d - dTdx * (far + close));
441  }
442  } else {
443  return 0.5 * ((I - T) * d - dTdx * (far + close));
444  }
445  }
