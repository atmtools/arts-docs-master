645  if (iy_unit_radar == "1") {
646  } else if (iy_unit_radar == "Ze") {
647  ze_cfac(cfac, f_grid, ze_tref, k2);
648  } else if (iy_unit_radar == "dBZe") {
649  ze_cfac(cfac, f_grid, ze_tref, k2);

651  } else {
652  ARTS_USER_ERROR (
653  "For this method, *iy_unit_radar* must be set to \"1\", "
654  "\"Ze\" or \"dBZe\".");
655  }

658  // Various initializations
659  //---------------------------------------------------------------------------
660 
661  // Variables to handle conversion from Stokes to instrument_pol
662  ArrayOfIndex npolcum(nf + 1);
663  npolcum[0] = 0;
664  ArrayOfArrayOfVector W(nf);
665  for (Index i = 0; i < nf; i++) {
666  const Index ni = instrument_pol_array[i].nelem();
667  npolcum[i + 1] = npolcum[i] + ni;
668  W[i].resize(ni);
669  for (Index j = 0; j < ni; j++) {
670  W[i][j].resize(stokes_dim);
671  stokes2pol(W[i][j], stokes_dim, instrument_pol_array[i][j], 0.5);
672  }
673  }
674 
675  // Resize and init *y* and *y_XXX*
676  //
677  const Index ntot = npos * npolcum[nf] * nbins;
678  y.resize(ntot);
679  y = NAN;
680  y_f.resize(ntot);
681  y_pol.resize(ntot);
682  y_pos.resize(ntot, sensor_pos.ncols());
683  y_los.resize(ntot, sensor_los.ncols());
684  y_geo.resize(ntot, 5);
685  y_geo = NAN; // Will be replaced if relavant data are provided
686 
687  // y_aux
688  y_aux.resize(naux);
689  for (Index i = 0; i < naux; i++) {
690  y_aux[i].resize(ntot);
691  y_aux[i] = NAN;
692  }
693 
694  // Jacobian variables
695  //
696  Index j_analytical_do = 0;
697  Index njq = 0;
698  ArrayOfArrayOfIndex jacobian_indices;
699  //
700  if (jacobian_do) {
701  bool any_affine;
702  jac_ranges_indices(jacobian_indices, any_affine, jacobian_quantities, true);
703 
704  jacobian.resize(ntot,
705  jacobian_indices[jacobian_indices.nelem() - 1][1] + 1);
706  jacobian = 0;
707  njq = jacobian_quantities.nelem();
708  //
709  FOR_ANALYTICAL_JACOBIANS_DO(j_analytical_do = 1;)
710  } else {
711  jacobian.resize(0, 0);
712  }
713 
714  //---------------------------------------------------------------------------
715  // The calculations
716  //---------------------------------------------------------------------------
717 
718  // Loop positions
719  for (Index p = 0; p < npos; p++) {
720  // RT part
721  ArrayOfTensor3 diy_dx;
722  Matrix iy;
723  Ppath ppath;
724  ArrayOfMatrix iy_aux;
725  const Index iy_id = (Index)1e6 * p;
726  //
727  iy_radar_agendaExecute(ws,
728  iy,
729  iy_aux,
730  ppath,
731  diy_dx,
732  iy_aux_vars,
733  iy_id,
734  cloudbox_on,
735  jacobian_do,
736  sensor_pos(p, joker),
737  sensor_los(p, joker),
738  iy_radar_agenda);

741  const Index np = ppath.np;
742  ARTS_USER_ERROR_IF (np == 1,
743  "A path consisting of a single point found. "
744  "This is not allowed.");
745  error_if_limb_ppath(ppath);
746  ARTS_USER_ERROR_IF (iy.nrows() != nf * np,
747  "The size of *iy* returned from *iy_radar_agenda* "
748  "is not correct (for this method).");

751  Vector range(np);
752  if (is_z) {
753  range = ppath.pos(joker, 0);
754  } else { // Calculate round-trip time
755  range[0] = 2 * ppath.end_lstep / SPEED_OF_LIGHT;
756  for (Index i = 1; i < np; i++) {
757  range[i] = range[i - 1] + ppath.lstep[i - 1] *
758  (ppath.ngroup[i - 1] + ppath.ngroup[i]) /

760  }
761  }

763  const Numeric range_end2 = max(range[0], range[np - 1]);

769  {
770  // Bin limits

772  Numeric blim2 = min(range_bins[b + 1], range_end2);

778  // want the average, so divide weights with bin width
779  hbin /= (blim2 - blim1);
780

782  // Pick out part of iy for frequency
783  Matrix I = iy(Range(iv * np, np), joker);

785  if (j_analytical_do) {
786  FOR_ANALYTICAL_JACOBIANS_DO(
787  dI[iq] = diy_dx[iq](joker, Range(iv * np, np), joker);)
788  }

790  for (Index a = 0; a < naux; a++) {
791  A[a] = iy_aux[a](Range(iv * np, np), joker);
792  }
793 
794  // Variables to hold data for one freq and one pol
795  Vector refl(np);
796  ArrayOfMatrix drefl(njq);
797  if (j_analytical_do) {
798  FOR_ANALYTICAL_JACOBIANS_DO(drefl[iq].resize(dI[iq].npages(), np);)
799  }

801  for (Index a = 0; a < naux; a++) {
802  auxvar[a].resize(np);
803  }
804 
805  for (Index ip = 0; ip < instrument_pol_array[iv].nelem(); ip++) {
806  // Apply weights on each Stokes element
807  mult(refl, I, W[iv][ip]);
808  if (j_analytical_do) {
809  FOR_ANALYTICAL_JACOBIANS_DO(for (Index k = 0;
810  k < drefl[iq].nrows();
811  k++) {
812  mult(drefl[iq](k, joker), dI[iq](k, joker, joker), W[iv][ip]);
813  })
814  }
815  for (Index a = 0; a < naux; a++) {
816  if (iy_aux_vars[a] == "Backscattering") {
817  mult(auxvar[a], A[a], W[iv][ip]);
818  } else {
819  for (Index j = 0; j < np; j++) {
820  auxvar[a][j] = A[a](j, 0);
821  }
822  }
823  }
824 
825  // Apply bin weight vector to get final values.
826  Index iout = nbins * (p * npolcum[nf] + npolcum[iv] + ip) + b;
827  y[iout] = cfac[iv] * (hbin * refl);
828  //
829  if (j_analytical_do) {
830  FOR_ANALYTICAL_JACOBIANS_DO(
831  for (Index k = 0; k < drefl[iq].nrows(); k++) {
832  jacobian(iout, jacobian_indices[iq][0] + k) =
833  cfac[iv] * (hbin * drefl[iq](k, joker));
834  if (iy_unit_radar == "dBZe") {
835  jacobian(iout, jacobian_indices[iq][0] + k) *=
836  jfac / max(y[iout], ze_min);
837  }
838  })
839  }
840 
841  if (iy_unit_radar == "dBZe") {
842  y[iout] = y[iout] <= ze_min ? dbze_min : 10 * log10(y[iout]);
843  }
844 
845  // Same for aux variables
846  for (Index a = 0; a < naux; a++) {
847  if (iy_aux_vars[a] == "Backscattering") {
848  y_aux[a][iout] = cfac[iv] * (hbin * auxvar[a]);
849  if (iy_unit_radar == "dBZe") {
850  y_aux[a][iout] = y_aux[a][iout] <= ze_min
851  ? dbze_min
852  : 10 * log10(y_aux[a][iout]);
853  }
854  } else {
855  y_aux[a][iout] = hbin * auxvar[a];
856  }
857  }
858  }
859  } // Frequency
860  }
861  }
862 
863  // Other aux variables
864  //
865  Vector geo_pos;
866  geo_pos_agendaExecute(ws, geo_pos, ppath, geo_pos_agenda);
867  ARTS_USER_ERROR_IF (geo_pos.nelem() && geo_pos.nelem() != atmosphere_dim,
868  "Wrong size of *geo_pos* obtained from "
869  "*geo_pos_agenda*.\nThe length of *geo_pos* must "
870  "be zero or equal to *atmosphere_dim*.");
871  //
872  for (Index b = 0; b < nbins; b++) {
873  for (Index iv = 0; iv < nf; iv++) {
874  for (Index ip = 0; ip < instrument_pol_array[iv].nelem(); ip++) {
875  const Index iout = nbins * (p * npolcum[nf] + npolcum[iv] + ip) + b;
876  y_f[iout] = f_grid[iv];
877  y_pol[iout] = instrument_pol_array[iv][ip];
878  y_pos(iout, joker) = sensor_pos(p, joker);
879  y_los(iout, joker) = sensor_los(p, joker);
880  if (geo_pos.nelem()) {
881  y_geo(iout, joker) = geo_pos;
882  }
883  }
884  }
885  }
886  }

976  Index phase = -1, it = -1; // Dummy values
977  if (dbze > dbze_noise) {

989  it = gp.fd[0] < 0.5 ? gp.idx : gp.idx+1;
990  }
991  }
992 
993  // Calculate attenuation from previous point
994  //
995  // For simplicity, extinction estimated using "roughly" corrected dBZe
996  //

998 
999  // Attenuation due particles
1000  if (do_atten_hyd && dBZe(ip,ilat,ilon) > dbze_noise) {

1002  GridPos gp;

1020  Numeric k_this;
1021  PropagationMatrix propmat;
1022  ArrayOfPropagationMatrix partial_dummy;
1023  StokesVector nlte_dummy;
1024  ArrayOfStokesVector partial_nlte_dummy;
1025  EnergyLevelMap rtp_nlte_local_dummy;
1026  propmat_clearsky_agendaExecute(ws,
1027  propmat,
1028  nlte_dummy,
1029  partial_dummy,
1030  partial_nlte_dummy,
1031  ArrayOfRetrievalQuantity(0),
1032  f_grid,
1033  Vector(3,0),
1034  Vector(0),
1035  p_grid[ip],
1036  t_field(ip,ilat,ilon),
1037  rtp_nlte_local_dummy,
1038  vmr_field(joker,ip,ilat,ilon),
1039  propmat_clearsky_agenda);

1043  (z_field(ip+1,ilat,ilon) - z_field(ip,ilat,ilon));
1044  // This equals -10*log10(exp(-tau)^2)
1045  dbze_corr += 20 * LOG10_EULER_NUMBER * tau;
1046  // k_this can ne be shifted to be "above value"
1047  k_abs_above = k_this;
1048  }
1049  }
1050 
1051  // Invert
1052  if (dBZe(ip,ilat,ilon) > dbze_noise) {
1053  // Correct reflectivity with (updated) attenuation
1054  dbze = dBZe(ip,ilat,ilon) + min(dbze_corr, dbze_max_corr);
1055  // Interpolate inversion table (table holds log10 of water content)
1056  GridPos gp;
1057  gridpos(gp, invtable[phase].get_numeric_grid(GFIELD3_DB_GRID),
1058  dbze, extrap_fac);
1059  Vector itw(2);
1060  interpweights(itw, gp);
1061  Numeric wc = pow(10.0,
1062  (interp(itw, invtable[phase].data(0,joker,it), gp)));
1063  // Apply max and clip values
1064  if (wc > wc_max)
1065  wc = 0;
1066  else if (wc > wc_clip)
1067  wc = wc_clip;
1068  particle_bulkprop_field(phase,ip,ilat,ilon) = wc;
1069  }
1070 
1071  // In clutter zone or below surface
1072  } else {
1073  if (fill_clutter) {
1074  particle_bulkprop_field(0,ip,ilat,ilon) =
1075  particle_bulkprop_field(0,ip+1,ilat,ilon);
1076  particle_bulkprop_field(1,ip,ilat,ilon) =
1077  particle_bulkprop_field(1,ip+1,ilat,ilon);
1078  }
1079  }
1080 
1081  } // presuure
1082  } // R > 0
1083  } // lon
1084  } // lat
1085 }
