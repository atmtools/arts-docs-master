37 void abs_hitran_relmat_dataReadHitranRelmatDataAndLines(HitranRelaxationMatrixData& abs_hitran_relmat_data,
38  ArrayOfArrayOfAbsorptionLines& abs_lines_per_species,
39  const ArrayOfArrayOfSpeciesTag& abs_species,
40  const String& basedir,
41  const Numeric& linemixinglimit,
42  const Numeric& fmin,
43  const Numeric& fmax,
44  const Numeric& stot,
45  const String& mode,
46  const Verbosity&)
47 {
48  using global_data::species_data;
49 
50  lm_hitran_2017::ModeOfLineMixing intmode;
51  if (mode == "VP") intmode = lm_hitran_2017::ModeOfLineMixing::VP;
52  else if (mode == "VP_Y") intmode = lm_hitran_2017::ModeOfLineMixing::VP_Y;
53  else if (mode == "SDVP") intmode = lm_hitran_2017::ModeOfLineMixing::SDVP;
54  else if (mode == "SDVP_Y") intmode = lm_hitran_2017::ModeOfLineMixing::SDVP_Y;
55  else if (mode == "FullW") intmode = lm_hitran_2017::ModeOfLineMixing::FullW;
56  else if (mode == "VP_W") intmode = lm_hitran_2017::ModeOfLineMixing::VP_W;
57  else throw std::runtime_error("Bad mode, see method instruction for valid arguments");
58 
59  if (abs_species.nelem() not_eq abs_lines_per_species.nelem())
60  throw std::runtime_error("Bad size of input species+lines");
61 
62  ArrayOfAbsorptionLines lines;
63  lm_hitran_2017::read(abs_hitran_relmat_data, lines, basedir, linemixinglimit, fmin, fmax, stot, intmode);
64  std::for_each(lines.begin(), lines.end(), [](auto& band){band.sort_by_frequency();}); // Sort so largest frequency is last
65  ArrayOfIndex used(lines.nelem(), false);
66 
67  bool emptied=false;
68  for (Index i=0; i<abs_species.nelem(); i++) {
69 
70  for (Index j=0; j<abs_species[i].nelem(); j++) {
71  if (abs_species[i][j].Species() not_eq SpeciesTag("CO2").Species())
72  continue;
73 
74  if (not emptied) {
75  abs_lines_per_species[i].resize(0);
76  emptied = true;
77  }
78 
79  for (Index k=0; k<lines.nelem(); k++) {
80  if (used[k]) continue;
81 
82  const Numeric lf{abs_species[i][j].Lf() > 0 ? abs_species[i][j].Lf() : -std::numeric_limits<Numeric>::max()};
83  const Numeric uf{abs_species[i][j].Uf() > 0 ? abs_species[i][j].Uf() : std::numeric_limits<Numeric>::max()};
84 
85  // Select lines with correct Isotopologue and one line center within the range
86  if ((abs_species[i][j].Isotopologue() == lines[k].Isotopologue() or
87  abs_species[i][j].Isotopologue() == species_data[SpeciesTag("CO2").Species()].Isotopologue().nelem()) and
88  (lines[k].AllLines().front().F0() <= uf and lines[k].AllLines().back().F0() >= lf)) {
89  used[k] = true; // The lines should not be copied into other places
90  abs_lines_per_species[i].push_back(lines[k]);
91  }
92  }
93  }
94  }
95 }

97 void propmat_clearskyAddHitranLineMixingLines(ArrayOfPropagationMatrix& propmat_clearsky,
98  const HitranRelaxationMatrixData& abs_hitran_relmat_data,
99  const ArrayOfArrayOfAbsorptionLines& abs_lines_per_species,
100  const Vector& f_grid,
101  const ArrayOfArrayOfSpeciesTag& abs_species,
102  const ArrayOfRetrievalQuantity& jacobian_quantities,
103  const SpeciesAuxData& partition_functions,
104  const Numeric& rtp_pressure,
105  const Numeric& rtp_temperature,
106  const Vector& rtp_vmr,
107  const Verbosity&)
108 {
109  if (jacobian_quantities.nelem())
110  throw std::runtime_error("Cannot support any Jacobian at this time");
111  if (abs_species.nelem() not_eq abs_lines_per_species.nelem())
112  throw std::runtime_error("Bad size of input species+lines");
113  if (abs_species.nelem() not_eq rtp_vmr.nelem())
114  throw std::runtime_error("Bad size of input species+vmrs");
115 
116  // vmrs should be [air, water, co2]
117  Vector vmrs(3, 0);
118  for (Index i=0; i<abs_species.nelem(); i++) {
119  auto& specs = abs_species[i];
120  for (auto& spec: specs) {
121  if (SpeciesTag("H2O").Species() == spec.Species()) {
122  vmrs[1] = rtp_vmr[i];
123  }
124  else if (SpeciesTag("CO2").Species() == spec.Species()) {
125  vmrs[2] = rtp_vmr[i];
126  }
127  }
128  }
129  vmrs[0] = 1 - vmrs[1] - vmrs[2];
130 
131  for (Index i=0; i<abs_species.nelem(); i++) {
132  if (abs_lines_per_species[i].nelem() and
133  (abs_lines_per_species[i].front().Population() == Absorption::PopulationType::ByHITRANFullRelmat or
134  abs_lines_per_species[i].front().Population() == Absorption::PopulationType::ByHITRANRosenkranzRelmat))
135  propmat_clearsky[i].Kjj() += lm_hitran_2017::compute(abs_hitran_relmat_data, abs_lines_per_species[i], rtp_pressure, rtp_temperature, vmrs, f_grid, partition_functions);
136  }
137 }

160 
161  for (Index i=0; i<abs_species.nelem(); i++) {

167  const Numeric this_vmr = rtp_vmr[i];
168  const ComplexVector abs = Absorption::LineMixing::linemixing_ecs_absorption(rtp_temperature,

170  this_vmr,
171  line_shape_vmr,
172  line_shape_mass,
173  f_grid,
174  band,
175  partition_functions.getParamType(band.QuantumIdentity()),

183 void propmat_clearskyAddOnTheFlyLineMixingWithZeeman(ArrayOfPropagationMatrix& propmat_clearsky,
184  const ArrayOfArrayOfAbsorptionLines& abs_lines_per_species,
185  const Vector& f_grid,
186  const ArrayOfArrayOfSpeciesTag& abs_species,
187  const ArrayOfRetrievalQuantity& jacobian_quantities,
188  const SpeciesAuxData& partition_functions,
189  const Numeric& rtp_pressure,
190  const Numeric& rtp_temperature,
191  const Vector& rtp_vmr,
192  const Vector& rtp_mag,
193  const Vector& rtp_los,
194  const Index& lbl_checked,
195  const Verbosity&)
196 {
197  if (std::any_of(propmat_clearsky.begin(), propmat_clearsky.end(), [](auto& pm){return pm.StokesDimensions() not_eq 4;}))
198  throw std::runtime_error("Only for stokes dim 4");
199  if (jacobian_quantities.nelem())
200  throw std::runtime_error("Cannot support any Jacobian at this time");
201  if (abs_species.nelem() not_eq abs_lines_per_species.nelem())
202  throw std::runtime_error("Bad size of input species+lines");
203  if (abs_species.nelem() not_eq rtp_vmr.nelem())
204  throw std::runtime_error("Bad size of input species+vmrs");
205  if (not lbl_checked)
206  throw std::runtime_error("Please set lbl_checked true to use this function");
207 
208  // Polarization
209  const auto Z = Zeeman::FromGrids(rtp_mag[0], rtp_mag[1], rtp_mag[2], Conversion::deg2rad(rtp_los[0]), Conversion::deg2rad(rtp_los[1]));
210  const auto polarization_scale_data = Zeeman::AllPolarization(Z.theta, Z.eta);
211 
212  for (Index i=0; i<abs_species.nelem(); i++) {
213  for (auto& band: abs_lines_per_species[i]) {
214  if (band.Population() == Absorption::PopulationType::ByMakarovFullRelmat and band.DoLineMixing(rtp_pressure)) {
215  // vmrs should be for the line
216  const Vector line_shape_vmr = band.BroadeningSpeciesVMR(rtp_vmr, abs_species);
217  const Vector line_shape_mass = band.BroadeningSpeciesMass(rtp_vmr, abs_species);
218  const Numeric this_vmr = rtp_vmr[i];
219  for (Zeeman::Polarization polarization : {Zeeman::Polarization::Pi, Zeeman::Polarization::SigmaMinus, Zeeman::Polarization::SigmaPlus}) {
220  const ComplexVector abs = Absorption::LineMixing::linemixing_ecs_absorption_with_zeeman_perturbations(rtp_temperature,
221  Z.H,
222  rtp_pressure,
223  this_vmr,
224  line_shape_vmr,
225  line_shape_mass,
226  f_grid,
227  polarization,
228  band,
229  partition_functions.getParamType(band.QuantumIdentity()),
230  partition_functions.getParam(band.QuantumIdentity()));
231 
232  auto& pol = Zeeman::SelectPolarization(polarization_scale_data, polarization);
233  auto pol_real = pol.attenuation();
234  auto pol_imag = pol.dispersion();
235  for (Index iv=0; iv<f_grid.nelem(); iv++) {
236  propmat_clearsky[i].Kjj()[iv] += abs[iv].real() * pol_real[0];
237  propmat_clearsky[i].K12()[iv] += abs[iv].real() * pol_real[1];
238  propmat_clearsky[i].K13()[iv] += abs[iv].real() * pol_real[2];
239  propmat_clearsky[i].K14()[iv] += abs[iv].real() * pol_real[3];
240  propmat_clearsky[i].K23()[iv] += abs[iv].imag() * pol_imag[0];
241  propmat_clearsky[i].K24()[iv] += abs[iv].imag() * pol_imag[1];
242  propmat_clearsky[i].K34()[iv] += abs[iv].imag() * pol_imag[2];
243  }
244  }
245  }
246  }
247  }
