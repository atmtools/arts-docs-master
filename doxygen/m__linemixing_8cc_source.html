37 void abs_hitran_relmat_dataReadHitranRelmatDataAndLines(HitranRelaxationMatrixData& abs_hitran_relmat_data,
38  ArrayOfArrayOfAbsorptionLines& abs_lines_per_species,
39  const ArrayOfArrayOfSpeciesTag& abs_species,
40  const String& basedir,
41  const Numeric& linemixinglimit,
42  const Numeric& fmin,
43  const Numeric& fmax,
44  const Numeric& stot,
45  const String& mode,
46  const Verbosity&)
47 {
48  using global_data::species_data;
49 
50  lm_hitran_2017::ModeOfLineMixing intmode;
51  if (mode == "VP") intmode = lm_hitran_2017::ModeOfLineMixing::VP;
52  else if (mode == "VP_Y") intmode = lm_hitran_2017::ModeOfLineMixing::VP_Y;
53  else if (mode == "SDVP") intmode = lm_hitran_2017::ModeOfLineMixing::SDVP;
54  else if (mode == "SDVP_Y") intmode = lm_hitran_2017::ModeOfLineMixing::SDVP_Y;
55  else if (mode == "FullW") intmode = lm_hitran_2017::ModeOfLineMixing::FullW;
56  else if (mode == "VP_W") intmode = lm_hitran_2017::ModeOfLineMixing::VP_W;
57  else ARTS_USER_ERROR ("Bad mode, see method instruction for valid arguments");
58 
59  ARTS_USER_ERROR_IF (abs_species.nelem() not_eq abs_lines_per_species.nelem(),
60  "Bad size of input species+lines");
61 
62  ArrayOfAbsorptionLines lines;
63  lm_hitran_2017::read(abs_hitran_relmat_data, lines, basedir, linemixinglimit, fmin, fmax, stot, intmode);
64  std::for_each(lines.begin(), lines.end(), [](auto& band){band.sort_by_frequency();}); // Sort so largest frequency is last
65  ArrayOfIndex used(lines.nelem(), false);
66 
67  bool emptied=false;
68  for (Index i=0; i<abs_species.nelem(); i++) {
69 
70  for (Index j=0; j<abs_species[i].nelem(); j++) {
71  if (abs_species[i][j].Species() not_eq SpeciesTag("CO2").Species())
72  continue;
73 
74  if (not emptied) {
75  abs_lines_per_species[i].resize(0);
76  emptied = true;
77  }
78 
79  for (Index k=0; k<lines.nelem(); k++) {
80  if (used[k]) continue;
81 
82  const Numeric lf{abs_species[i][j].Lf() > 0 ? abs_species[i][j].Lf() : -std::numeric_limits<Numeric>::max()};
83  const Numeric uf{abs_species[i][j].Uf() > 0 ? abs_species[i][j].Uf() : std::numeric_limits<Numeric>::max()};
84 
85  // Select lines with correct Isotopologue and one line center within the range
86  if ((abs_species[i][j].Isotopologue() == lines[k].Isotopologue() or
87  abs_species[i][j].Isotopologue() == species_data[SpeciesTag("CO2").Species()].Isotopologue().nelem()) and
88  (lines[k].AllLines().front().F0() <= uf and lines[k].AllLines().back().F0() >= lf)) {
89  used[k] = true; // The lines should not be copied into other places
90  abs_lines_per_species[i].push_back(lines[k]);
91  }
92  }
93  }
94  }
95 }

97 void propmat_clearskyAddHitranLineMixingLines(PropagationMatrix& propmat_clearsky,
98  const HitranRelaxationMatrixData& abs_hitran_relmat_data,
99  const ArrayOfArrayOfAbsorptionLines& abs_lines_per_species,
100  const Vector& f_grid,
101  const ArrayOfArrayOfSpeciesTag& abs_species,
102  const ArrayOfRetrievalQuantity& jacobian_quantities,
103  const SpeciesAuxData& partition_functions,
104  const Numeric& rtp_pressure,
105  const Numeric& rtp_temperature,
106  const Vector& rtp_vmr,
107  const Verbosity&)
108 {
109  ARTS_USER_ERROR_IF (jacobian_quantities.nelem(),
110  "Cannot support any Jacobian at this time");
111  ARTS_USER_ERROR_IF (abs_species.nelem() not_eq abs_lines_per_species.nelem(),
112  "Bad size of input species+lines");
113  ARTS_USER_ERROR_IF (abs_species.nelem() not_eq rtp_vmr.nelem(),
114  "Bad size of input species+vmrs");
115 
116  // vmrs should be [air, water, co2]
117  Vector vmrs(3, 0);
118  for (Index i=0; i<abs_species.nelem(); i++) {
119  auto& specs = abs_species[i];
120  for (auto& spec: specs) {
121  if (SpeciesTag("H2O").Species() == spec.Species()) {
122  vmrs[1] = rtp_vmr[i];
123  }
124  else if (SpeciesTag("CO2").Species() == spec.Species()) {
125  vmrs[2] = rtp_vmr[i];
126  }
127  }
128  }
129  vmrs[0] = 1 - vmrs[1] - vmrs[2];
130 
131  for (Index i=0; i<abs_species.nelem(); i++) {
132  if (abs_lines_per_species[i].nelem() and
133  (abs_lines_per_species[i].front().Population() == Absorption::PopulationType::ByHITRANFullRelmat or
134  abs_lines_per_species[i].front().Population() == Absorption::PopulationType::ByHITRANRosenkranzRelmat))
135  propmat_clearsky.Kjj() += lm_hitran_2017::compute(abs_hitran_relmat_data, abs_lines_per_species[i], rtp_pressure, rtp_temperature, vmrs, f_grid, partition_functions);
136  }

139 void propmat_clearskyAddOnTheFlyLineMixing(PropagationMatrix& propmat_clearsky,
140  ArrayOfPropagationMatrix& dpropmat_clearsky_dx,
141  const ArrayOfArrayOfAbsorptionLines& abs_lines_per_species,
142  const Vector& f_grid,
143  const ArrayOfArrayOfSpeciesTag& abs_species,
144  const ArrayOfRetrievalQuantity& jacobian_quantities,
145  const SpeciesAuxData& partition_functions,
146  const Numeric& rtp_pressure,
147  const Numeric& rtp_temperature,
148  const Vector& rtp_vmr,
149  const Index& lbl_checked,
150  const Verbosity&)
151 {
152  ARTS_USER_ERROR_IF (abs_species.nelem() not_eq abs_lines_per_species.nelem(),
153  "Bad size of input species+lines");
154  ARTS_USER_ERROR_IF (abs_species.nelem() not_eq rtp_vmr.nelem(),
155  "Bad size of input species+vmrs");
156  ARTS_USER_ERROR_IF (not lbl_checked,
157  "Please set lbl_checked true to use this function");
158 
159  for (Index i=0; i<abs_species.nelem(); i++) {
160  for (auto& band: abs_lines_per_species[i]) {
161  if (band.Population() == Absorption::PopulationType::ByMakarovFullRelmat and band.DoLineMixing(rtp_pressure)) {
162  // vmrs should be for the line
163  const Vector line_shape_vmr = band.BroadeningSpeciesVMR(rtp_vmr, abs_species);
164  const Vector line_shape_mass = band.BroadeningSpeciesMass(rtp_vmr, abs_species);
165  const Numeric this_vmr = rtp_vmr[i];
166  const auto [abs, dabs] = Absorption::LineMixing::ecs_absorption(rtp_temperature,
167  rtp_pressure,
168  this_vmr,
169  line_shape_vmr,
170  line_shape_mass,
171  f_grid,
172  band,
173  partition_functions.getParamType(band.QuantumIdentity()),
174  partition_functions.getParam(band.QuantumIdentity()),
175  jacobian_quantities);
176  propmat_clearsky.Kjj() += abs.real();
177 
178  // Sum up the resorted Jacobian
179  for (Index j=0; j<jacobian_quantities.nelem(); j++) {
180  if (not propmattype_index(jacobian_quantities, j)) continue;
181  if (is_special_vmr(jacobian_quantities[j], abs_species[i])) {
182  dpropmat_clearsky_dx[j].Kjj() += abs.real();
183  } else {
184  dpropmat_clearsky_dx[j].Kjj() += dabs[j].real();
185  }
186  }
187  }
188  }
189  }

192 void propmat_clearskyAddOnTheFlyLineMixingWithZeeman(PropagationMatrix& propmat_clearsky,
193  ArrayOfPropagationMatrix& dpropmat_clearsky_dx,
194  const ArrayOfArrayOfAbsorptionLines& abs_lines_per_species,
195  const Vector& f_grid,
196  const ArrayOfArrayOfSpeciesTag& abs_species,
197  const ArrayOfRetrievalQuantity& jacobian_quantities,
198  const SpeciesAuxData& partition_functions,
199  const Numeric& rtp_pressure,
200  const Numeric& rtp_temperature,
201  const Vector& rtp_vmr,
202  const Vector& rtp_mag,
203  const Vector& rtp_los,
204  const Index& lbl_checked,
205  const Verbosity&)
206 {
207  ARTS_USER_ERROR_IF (propmat_clearsky.StokesDimensions() not_eq 4,
208  "Only for stokes dim 4");
209  ARTS_USER_ERROR_IF (abs_species.nelem() not_eq abs_lines_per_species.nelem(),
210  "Bad size of input species+lines");
211  ARTS_USER_ERROR_IF (abs_species.nelem() not_eq rtp_vmr.nelem(),
212  "Bad size of input species+vmrs");
213  ARTS_USER_ERROR_IF (not lbl_checked,
214  "Please set lbl_checked true to use this function");
215 
216  // Polarization
217  const auto Z = Zeeman::FromGrids(rtp_mag[0], rtp_mag[1], rtp_mag[2], Conversion::deg2rad(rtp_los[0]), Conversion::deg2rad(rtp_los[1]));
218  const auto polarization_scale_data = Zeeman::AllPolarization(Z.theta, Z.eta);
219  const auto polarization_scale_dtheta_data = Zeeman::AllPolarization_dtheta(Z.theta, Z.eta);
220  const auto polarization_scale_deta_data = Zeeman::AllPolarization_deta(Z.theta, Z.eta);
221 
222  for (Index i=0; i<abs_species.nelem(); i++) {
223  for (auto& band: abs_lines_per_species[i]) {
224  if (band.Population() == Absorption::PopulationType::ByMakarovFullRelmat and band.DoLineMixing(rtp_pressure)) {
225  // vmrs should be for the line
226  const Vector line_shape_vmr = band.BroadeningSpeciesVMR(rtp_vmr, abs_species);
227  const Vector line_shape_mass = band.BroadeningSpeciesMass(rtp_vmr, abs_species);
228  const Numeric this_vmr = rtp_vmr[i];
229  for (Zeeman::Polarization polarization : {Zeeman::Polarization::Pi, Zeeman::Polarization::SigmaMinus, Zeeman::Polarization::SigmaPlus}) {
230  const auto [abs, dabs] = Absorption::LineMixing::ecs_absorption_zeeman(rtp_temperature,
231  Z.H,
232  rtp_pressure,
233  this_vmr,
234  line_shape_vmr,
235  line_shape_mass,
236  f_grid,
237  polarization,
238  band,
239  partition_functions.getParamType(band.QuantumIdentity()),
240  partition_functions.getParam(band.QuantumIdentity()),
241  jacobian_quantities);
242 
243  // Sum up the propagation matrix
244  Zeeman::sum(propmat_clearsky, abs, Zeeman::SelectPolarization(polarization_scale_data, polarization));
245 
246  // Sum up the resorted Jacobian
247  for (Index j=0; j<jacobian_quantities.nelem(); j++) {
248  if (not propmattype_index(jacobian_quantities, j)) continue;
249  if (jacobian_quantities[j] == Jacobian::Atm::MagneticU) {
250  Zeeman::dsum(dpropmat_clearsky_dx[j], abs, dabs[j],
251  Zeeman::SelectPolarization(polarization_scale_data, polarization),
252  Zeeman::SelectPolarization(polarization_scale_dtheta_data, polarization),
253  Zeeman::SelectPolarization(polarization_scale_deta_data, polarization),
254  Z.dH_du, Z.dtheta_du, Z.deta_du);
255  } else if (jacobian_quantities[j] == Jacobian::Atm::MagneticV) {
256  Zeeman::dsum(dpropmat_clearsky_dx[j], abs, dabs[j],
257  Zeeman::SelectPolarization(polarization_scale_data, polarization),
258  Zeeman::SelectPolarization(polarization_scale_dtheta_data, polarization),
259  Zeeman::SelectPolarization(polarization_scale_deta_data, polarization),
260  Z.dH_dv, Z.dtheta_dv, Z.deta_dv);
261  } else if (jacobian_quantities[j] == Jacobian::Atm::MagneticW) {
262  Zeeman::dsum(dpropmat_clearsky_dx[j], abs, dabs[j],
263  Zeeman::SelectPolarization(polarization_scale_data, polarization),
264  Zeeman::SelectPolarization(polarization_scale_dtheta_data, polarization),
265  Zeeman::SelectPolarization(polarization_scale_deta_data, polarization),
266  Z.dH_dw, Z.dtheta_dw, Z.deta_dw);
267  } else if (is_special_vmr(jacobian_quantities[j], abs_species[i])) {
268  Zeeman::sum(dpropmat_clearsky_dx[j], abs, Zeeman::SelectPolarization(polarization_scale_data, polarization));
269  } else {
270  dpropmat_clearsky_dx[j].Kjj() += dabs[j].real();
271  }
272  }
273  }
274  }
275  }
276  }
