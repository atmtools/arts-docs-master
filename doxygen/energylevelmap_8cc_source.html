31 bool EnergyLevelMap::OK() const noexcept {
32  if (not (mvalue.nbooks() == mlevels.nelem() and
33  (mvib_energy.nelem() == mlevels.nelem() or mvib_energy.nelem() == 0))) {
34  return false; // Bad dimensions, vibrational energies and IDs and data of strange size
35  }
36 
37  if (mtype == EnergyLevelMapType::Tensor3_t) {
38  } else if (mtype == EnergyLevelMapType::Vector_t) {
39  if (mvalue.npages() not_eq 1 or mvalue.nrows() not_eq 1) {
40  return false; // Bad dimensions for vector type
41  }
42  } else if (mtype == EnergyLevelMapType::Numeric_t) {
43  if (mvalue.npages() not_eq 1 or mvalue.nrows() not_eq 1 or mvalue.ncols() not_eq 1) {
44  return false; // Bad dimensions for numeric type
45  }
46  } else if (mtype == EnergyLevelMapType::None_t) {
47  if (mvalue.npages() not_eq 0 or mvalue.nrows() not_eq 0 or mvalue.ncols() not_eq 0) {
48  return false; // Bad dimensions for none type
49  }
50  }
51 
52  for (auto& e: mvib_energy) if (e < 0) return false; // Bad energies
53 
54  return true;
55 }

58 Output2 EnergyLevelMap::get_ratio_params(
59  const AbsorptionLines& band, const Index& line_index) const
60 {
61  ARTS_USER_ERROR_IF (mtype not_eq EnergyLevelMapType::Numeric_t,
62  "Must have Numeric_t, input type is bad");
63 
64  Output2 x{/*.r_low=*/0, /*.r_upp=*/0};
65 
66  bool found1=false;
67  bool found2=false;
68  for (size_t i=0; i<mlevels.size(); i++) {
69  const Absorption::QuantumIdentifierLineTarget lt = Absorption::QuantumIdentifierLineTarget(mlevels[i], band, line_index);
70  if (lt == Absorption::QuantumIdentifierLineTargetType::Level and lt.lower) {
71  found1 = true;
72  x.r_low = mvalue(i, 0, 0, 0);
73  }
74 
75  if (lt == Absorption::QuantumIdentifierLineTargetType::Level and lt.upper) {
76  found2 = true;
77  x.r_upp = mvalue(i, 0, 0, 0);
78  }
79 
80  if (found1 and found2)
81  break;
82  }
83  return x;
84 }

94  Output4 x{/*.E_low=*/0, /*.E_upp=*/0, /*.T_low=*/T, /*.T_upp=*/T};
95 
96  bool found1=false;
97  bool found2=false;
98  for (Index i=0; i<mlevels.nelem(); i++) {
99  const Absorption::QuantumIdentifierLineTarget lt = Absorption::QuantumIdentifierLineTarget(mlevels[i], band, line_index);
100

102  found1 = true;

108  found2 = true;

113  if (found1 and found2) {
114  break;
115  }
116  }

149 EnergyLevelMap EnergyLevelMap::InterpToGridPos(Index atmosphere_dim, const ArrayOfGridPos& p, const ArrayOfGridPos& lat, const ArrayOfGridPos& lon) const
150 {
151  if (mtype == EnergyLevelMapType::None_t)
152  return EnergyLevelMap();
153  ARTS_USER_ERROR_IF (mtype not_eq EnergyLevelMapType::Tensor3_t,
154  "Must have Tensor3_t, input type is bad");
155 
156  EnergyLevelMap elm(EnergyLevelMapType::Vector_t, 1, 1, p.nelem(), *this);
157 
158  Matrix itw_field;
159  interp_atmfield_gp2itw(itw_field, atmosphere_dim, p, lat, lon);
160 
161  const Index nnlte = mlevels.nelem();
162  for (Index itnlte = 0; itnlte < nnlte; itnlte++)
163  interp_atmfield_by_itw(elm.mvalue(itnlte, 0, 0, joker), atmosphere_dim,
164  mvalue(itnlte, joker, joker, joker),
165  p, lat, lon, itw_field);
166  return elm;
167 }
