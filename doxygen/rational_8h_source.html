363 constexpr Rational numeric2rational(Numeric x, size_t maxdec=4) noexcept {
364  Index nom=0, denom=1;
365 
366  // Keep track of sign independently
367  const bool signchange = x < 0;
368  x = signchange ? -x : x;
369 
370  // Add numbers by keeping the floor
371  size_t i=0;
372  do {
373  const Index xi=Index(x);
374  nom += xi;
375  x = 10 * (x - Numeric(xi));
376  nom *= 10;
377  denom *= 10;
378  i++;
379  } while (i<=maxdec);
380 
381  // Fix possible rounding error
382  if (x >= 5)
383  nom += 10;
384 
385  // Change sign or not
386  if (signchange)
387  return Rational(-nom, denom);
388  else
389  return Rational(nom, denom);
390 }
