370 constexpr Rational numeric2rational(Numeric x, size_t maxdec=4) {
371  Index nom=0, denom=1;
372 
373  // Keep track of sign independently
374  const bool signchange = x < 0;
375  x = x < 0 ? -x : x;
376 
377  // Add numbers by keeping the floor
378  size_t i=0;
379  do {
380  const Index xi=Index(x);
381  nom += xi;
382  x = 10 * (x - Numeric(xi));
383  nom *= 10;
384  denom *= 10;
385  i++;
386  } while (i<=maxdec);
387 
388  // Fix possible rounding error
389  if (x >= 5)
390  nom += 10;
391 
392  // Change sign or not
393  if (signchange)
394  return Rational(-nom, denom);
395  else
396  return Rational(nom, denom);
397 }
